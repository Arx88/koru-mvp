   102→function escapeHtml(s: string): string {
   103→  return s.replace(/&/g, "&amp;").replace(/</g, "&lt;").replace(/>/g, "&gt;");
   104→}
   105→
   106→function inlineMd(s: string): string {
   107→  // s ya está escapado
   108→  s = s.replace(/\[([^\]]+)\]\(([^)]+)\)/g, '<a href="$2" target="_blank" rel="noopener noreferrer">$1</a>');
   109→  s = s.replace(/\*\*([^*]+)\*\*/g, "<strong>$1</strong>");
   110→  s = s.replace(/\*([^*]+)\*/g, "<em>$1</em>");
   111→  return s;
   112→}
   113→
   114→function renderMarkdown(text: string): string {
   115→  if (!text.trim()) return "";
   116→  const lines = text.split("\n");
   117→  const out: string[] = [];
   118→  let inList = false;
   119→  let para: string[] = [];
   120→  const flushPara = () => {
   121→    if (para.length > 0) {
   122→      out.push(`<p>${para.map(p => inlineMd(escapeHtml(p))).join("<br />")}</p>`);
   123→      para = [];
   124→    }
   125→  };
   126→  for (const line of lines) {
   127→    const listMatch = line.match(/^\s*-\s+(.*)$/);
   128→    if (listMatch) {
   129→      flushPara();
   130→      if (!inList) { out.push("<ul>"); inList = true; }
   131→      out.push(`<li>${inlineMd(escapeHtml(listMatch[1]))}</li>`);
   132→      continue;
   133→    }
   134→    if (inList) { out.push("</ul>"); inList = false; }
   135→    para.push(line);
   136→  }
   137→  flushPara();
   138→  if (inList) out.push("</ul>");
   139→  return out.join("");
   140→}

   375→  async function handleSave() {

   421→        title: title.trim() || url.trim() || "Enlace",

   423→        url: url.trim(),
   424→        notes: notes.trim() || undefined,

   432→      const rutinaText = [
   433→        `Cadencia: ${rutinaCadence}`,
   434→        `Meta: ${target} ${rutinaUnit || ""}`.trim(),
   435→        rutinaAnchorTime ? `Ancla: ${rutinaAnchorTime}` : null,

   438→      // habilita logHabit, streaks, rutinas ancladas, etc.

   440→        title.trim() || "Rutina",
   441→        "repeat",
   442→        rutinaCadence as Habit["cadence"],
   443→        target,
   444→        rutinaUnit || undefined,
   445→        rutinaAnchorTime || undefined,
   446→      );
   447→      // 🔴 TIER S: creamos una Routine que ancla el hábito recién creado a
   448→      // un horario + días de la semana. daysOfWeek: L-V si cadence es
   449→      // "mon-fri", todos (0-6) si "daily", [1] si "weekly", [0] para custom.
   450→      const daysOfWeek: number[] = rutinaCadence === "mon-fri"
   451→        ? [1, 2, 3, 4, 5]
   452→        : rutinaCadence === "daily"
   453→          ? [0, 1, 2, 3, 4, 5, 6]
   454→          : rutinaCadence === "weekly"
   455→            ? [1]
   456→            : [0];
   457→      createRoutine(
   458→        title.trim() || "Rutina",
   459→        rutinaAnchorTime || "08:00",
   460→        [habitId],
   461→        daysOfWeek,
   462→      );
   463→      await persist({
   464→        title: title.trim() || "Rutina sin título",
   465→        collection: finalCollection,
   466→        notes: rutinaText,
   467→        kind,
   468→        sourceBlock: buildSavedRecordBlock({
   469→          title: title.trim() || "Rutina",
   470→          collection: finalCollection,
   471→          notes: rutinaText,
   472→          kind,
   473→        }),
   474→      });
   475→      return;
   476→    }
   477→
   478→    if (selected === "ejercicio") {
   479→      const weeks = parseInt(weeksTotal) || 0;
   480→      const sessionsText = sessions
   481→        .filter(s => s.dayLabel.trim() || s.exercises.some(e => e.name.trim()))
   482→        .map(s => {
   483→          const ex = s.exercises.filter(e => e.name.trim());
   484→          const exText = ex
   485→            .map(e => `  - ${e.name}: ${e.sets || 0}x${e.reps || 0}${e.weight ? ` @ ${e.weight}kg` : ""}`)
   486→            .join("\n");
   487→          return `${s.dayLabel || "Día"}:\n${exText}`;
   488→        })
   489→        .join("\n\n");
   490→      const ejText = `Semanas: ${weeks}\n\n${sessionsText}`;
   491→      // 🔴 GAP-3: además del LifeRecord, persistimos un ExercisePlan durable —
   492→      // habilita logWorkout, avance de sesión, completado del plan, etc.
   493→      const planSessions: Omit<ExerciseSession, "id" | "order">[] = sessions
   494→        .filter(s => s.dayLabel.trim() || s.exercises.some(e => e.name.trim()))
   495→        .map(s => ({
   496→          dayLabel: s.dayLabel || "Día",
   497→          exercises: s.exercises
   498→            .filter(e => e.name.trim())
   499→            .map(e => ({
   500→              exercise: e.name,

   653→        notes: planText || undefined,
   654→        kind,
   655→        sourceBlock: buildSavedRecordBlock({
   656→          title: title.trim() || "Plan",
   657→          collection: finalCollection,
   658→          notes: planText,
   659→          kind,
   660→        }),
   661→      });

   663→    }

   728→      case "rutinaTarget": setRutinaTarget(v); break;
   729→      case "rutinaUnit": setRutinaUnit(v); break;

   749→  function insertWrap(wrap: string, placeholder: string) {
   750→    const ta = notesRef.current;
   751→    if (!ta) return;
   752→    const start = ta.selectionStart;
   753→    const end = ta.selectionEnd;
   754→    const sel = notes.slice(start, end) || placeholder;
   755→    const next = notes.slice(0, start) + wrap + sel + wrap + notes.slice(end);
   756→    setNotes(next);
   757→    requestAnimationFrame(() => {
   758→      ta.focus();
   759→      const pos = start + wrap.length;
   760→      ta.setSelectionRange(pos, pos + sel.length);
   761→    });
   762→  }
   763→
   764→  function insertPrefix(prefix: string) {
   765→    const ta = notesRef.current;
   766→    if (!ta) return;
   767→    const start = ta.selectionStart;
   768→    const before = notes.slice(0, start);
   769→    const lineStart = before.lastIndexOf("\n") + 1;
   770→    const next = notes.slice(0, lineStart) + prefix + notes.slice(lineStart);
   771→    setNotes(next);
   772→    requestAnimationFrame(() => {
   773→      ta.focus();
   774→      const pos = start + prefix.length;
   775→      ta.setSelectionRange(pos, pos);
   776→    });
   777→  }
   778→
   779→  function insertLink() {
   780→    const ta = notesRef.current;
   781→    if (!ta) return;
   782→    const start = ta.selectionStart;
   783→    const end = ta.selectionEnd;
   784→    const sel = notes.slice(start, end) || "texto";
   785→    const link = `[${sel}](https://)`;
   786→    const next = notes.slice(0, start) + link + notes.slice(end);
   787→    setNotes(next);
   788→    requestAnimationFrame(() => {
   789→      ta.focus();
   790→      const urlStart = start + sel.length + 3; // "[" + sel + "]("
   791→      ta.setSelectionRange(urlStart, urlStart + 8); // selecciona "https://"
   792→    });
   793→  }

  1485→              <input
  1486→                type="text"
  1487→                value={tagInput}
  1488→                onChange={(e) => setTagInput(e.target.value)}
  1489→                onKeyDown={(e) => {
  1490→                  if (e.key === "Enter" || e.key === ",") {
  1491→                    e.preventDefault();
  1492→                    addTag(tagInput);
  1493→                  } else if (e.key === "Backspace" && !tagInput && tags.length > 0) {
  1494→                    setTags(tags.slice(0, -1));
  1495→                  }
  1496→                }}
  1497→                placeholder="Agregar etiqueta y Enter…"
  1498→              />
  1499→              {tagSuggestions.length > 0 && (
  1500→                <div className="koru-create-tag-suggest">
  1501→                  {tagSuggestions.map(t => (
  1502→                    <button key={t} type="button" className="koru-create-tag-suggest-item" onClick={() => addTag(t)}>
  1503→                      {t}
  1504→                    </button>
  1505→                  ))}
  1506→                </div>
  1507→              )}
  1508→              {tags.length > 0 && (
  1509→                <div className="koru-create-tag-chips">
  1510→                  {tags.map(t => (
  1511→                    <span key={t} className="koru-create-tag-chip">
  1512→                      {t}
  1513→                      <button type="button" aria-label={`Quitar ${t}`} onClick={() => removeTag(t)}>×</button>
  1514→                    </span>
  1515→                  ))}
  1516→                </div>
  1517→              )}
  1518→            </div>
  1519→
  1520→            {/* CARPETA — con smart defaults hint */}
  1521→            <label className="koru-create-field">
  1522→              <span className="koru-create-field-label">Carpeta</span>
  1523→              {lastUsedCollection && (
  1524→                <span className="koru-create-last-used">Último usado: {lastUsedCollection}</span>
  1525→              )}
  1526→              <input
  1527→                type="text"
  1528→                value={collection}
  1529→                onChange={(e) => setCollection(e.target.value)}
  1530→                placeholder={currentTpl?.collection}
  1531→              />
  1532→            </label>
  1533→
  1534→            <div className="koru-create-actions">
  1535→              <button
  1536→                type="button"
  1537→                className="koru-create-action-cancel"
  1538→                onClick={() => setSelected(null)}
  1539→              >
  1540→                Cancelar
  1541→              </button>
  1542→              <button
  1543→                type="button"
  1544→                className="koru-create-action-save"
  1545→                onClick={selected === "receta" ? handleSaveReceta : handleSave}
  1546→                disabled={
  1547→                  saving ||
  1548→                  (!title.trim() && selected !== "gasto") ||
  1549→                  // 🔴 TIER S: plan requiere al menos un step con título.
  1550→                  (selected === "plan" && !planSteps.some(s => s.title.trim()))
  1551→                }
  1552→                style={{ background: currentTpl?.accent }}
  1553→              >
  1554→                {saving ? (
  1555→                  <><Mat>hourglass_top</Mat> Guardando…</>
  1556→                ) : (
  1557→                  <><Mat>check</Mat> Guardar</>
  1558→                )}
  1559→              </button>
  1560→            </div>
  1561→          </div>
  1562→        )}
  1563→      </div>
  1564→    </div>,
  1565→    document.body,
  1566→  );
  1567→}
