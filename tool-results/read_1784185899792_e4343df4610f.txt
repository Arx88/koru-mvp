   342→   342→          }
   343→   343→        }
   344→   344→      }

   348→   348→

   379→   379→            status: "done" as const,
   380→   380→            mascotState: "happy" as const,
   381→   381→          };
   382→   382→          // Marcar como mostrado para no repetir
   383→   383→          commitDomainState((prev) => ({
   384→   384→            ...prev,
   385→   385→            nudges: (prev.nudges ?? []).map(n =>
   386→   386→              n.id === nudge.id ? { ...n, title: `[proactive_shown] ${n.title}` } : n
   387→   387→            ),
   388→   388→          }));
   389→   389→          commitChatTurns((prev) => [...prev, proactiveTurn].slice(-120));
   390→   390→        }
   391→   391→      }
   392→   392→    };
   393→   393→    // Fase 2.12: pausar heartbeat cuando el tab no está visible.
   394→   394→    // Antes corría cada 60s sin importar si el usuario estaba mirando.
   395→   395→    // Ahora se pausa al ocultar el tab y reanuda al volver, ahorrando CPU.
   396→   396→    let timer = window.setInterval(runHeartbeat, 60_000);

   856→   856→  async function submitEntry(
   857→   857→    text: string,
   858→   858→    transcriptSource: DailyEntry["transcriptSource"] = "typed",
   859→   859→    history: KoruChatTurn[] = chatTurnsRef.current,
   860→   860→  ) {
   861→   861→    setProcessing(true);
   862→   862→    let koruTurnId: string | null = null;
   863→   863→    // Trackea los items creados durante el stream para preservar sus ids al
   864→   864→    // reemplazar con el resultado final. Sin esto, el resultado final crea items
   865→   865→    // con ids nuevos → el WebNavCardA se desmonta/remonta → pierde su estado
   866→   866→    // (animación de progreso, visibleCount) → reinicia a "Iniciando búsqueda...".
   867→   867→    const streamedItemsByType = new Map<string, KoruTurnItem>();
   868→   868→    try {
   869→   869→      const previousState = domainStateRef.current;
   870→   870→      const domainHistory = history
   871→   871→        .filter((turn) => turn.text.trim().length > 0)
   872→   872→        .slice(-12)
   873→   873→        .map((turn) => ({
   874→   874→          role: turn.role === "koru" ? ("assistant" as const) : ("user" as const),
   875→   875→          content: turn.text,
   876→   876→          createdAt: turn.createdAt,
   877→   877→        }));
   878→   878→      const agentResult = await runBackendAgentTurn(text, previousState, domainHistory, selectedModel ?? undefined, (chunk) => {
   879→   879→        // Fase real del pipeline: tomamos el último stateEvent para la píldora/barra.
   880→   880→        if (chunk.stateEvents?.length) {
   881→   881→          const last = chunk.stateEvents[chunk.stateEvents.length - 1];
   882→   882→          setPhase(last.kind === "done" ? null : last.kind);
   883→   883→        }
   884→   884→        const blocksToItems = (blocks: UiBlock[]): KoruTurnItem[] =>
   885→   885→          blocks.map((block) => ({
   886→   886→            id: createId("item"),
   887→   887→            kind: "action" as const,
   888→   888→            tag: block.type,
   889→   889→            text: (block as any).title ?? (block as any).query ?? "",
   890→   890→            status: "working" as const,
   891→   891→            uiBlock: block,
   892→   892→          }));
   893→   893→        if (!koruTurnId) {
   894→   894→          koruTurnId = createId("turn");
   895→   895→          const koruTurn: KoruChatTurn = {
   896→   896→            id: koruTurnId,
   897→   897→            role: "koru",
   898→   898→            text: chunk.reply,
   899→   899→            createdAt: new Date().toISOString(),
   900→   900→            items: blocksToItems(chunk.uiBlocks),
   901→   901→            status: "working" as const,
   902→   902→            mascotState: chunk.mascotState ?? "working",
   903→   903→          };
   904→   904→          (koruTurn.items ?? []).forEach((it) => { if (it.uiBlock?.type) streamedItemsByType.set(it.uiBlock.type, it); });
   905→   905→          commitChatTurns((prev) => [...prev, koruTurn].slice(-120));
   906→   906→        } else {
   907→   907→          const isDone = chunk.stateEvents?.some((e) => e.kind === "done");
   908→   908→          commitChatTurns((prev) =>
   909→   909→            prev.map((turn) => {
   910→   910→              if (turn.id !== koruTurnId) return turn;
   911→   911→              const existingItems = turn.items ?? [];
   912→   912→              const newBlocks = chunk.uiBlocks ?? [];
   913→   913→              // Merge por TIPO de block (no por índice). Así, si el orden cambia
   914→   914→              // entre chunks (ej: aparece un data_card antes que el web_nav),
   915→   915→              // cada item mantiene su identidad y estado (animación de progreso,
   916→   916→              // contadores internos). Antes se matcheaba por idx, lo que hacía
   917→   917→              // que un cambio de orden desmontara el componente y reiniciara la
   918→   918→              // animación ("Iniciando búsqueda..." colgado).
   919→   919→              const consumedExisting = new Set<string>();
   920→   920→              const mergedItems: KoruTurnItem[] = newBlocks.map((block) => {
   921→   921→                const existing = existingItems.find(
   922→   922→                  (it) => it.uiBlock?.type === block.type && !consumedExisting.has(it.uiBlock!.type),
   923→   923→                );
   924→   924→                if (existing) {
   925→   925→                  consumedExisting.add(existing.uiBlock!.type);
   926→   926→                  return { ...existing, uiBlock: block, text: (block as any).title ?? (block as any).query ?? "" };
   927→   927→                }
   928→   928→                return {
   929→   929→                  id: createId("item"),
   930→   930→                  kind: "action" as const,
   931→   931→                  tag: block.type,
   932→   932→                  text: (block as any).title ?? (block as any).query ?? "",
   933→   933→                  status: "working" as const,
   934→   934→                  uiBlock: block,
   935→   935→                };
   936→   936→              });
   937→   937→              mergedItems.forEach((it) => { if (it.uiBlock?.type) streamedItemsByType.set(it.uiBlock.type, it); });
   938→   938→              return {
   939→   939→                ...turn,
   940→   940→                text: chunk.reply,
   941→   941→                items: mergedItems,
   942→   942→                mascotState: chunk.mascotState ?? turn.mascotState,
   943→   943→                status: isDone ? ("done" as const) : ("working" as const),
   944→   944→              };
   945→   945→            }),
   946→   946→          );
   947→   947→        }
   948→   948→      });
   949→   949→      const result = applyBackendTurnToState(previousState, text, transcriptSource, agentResult);
   950→   950→      commitDomainState(result.state);
   951→   951→
   952→   952→      // 🔴 Programar notificaciones para nuevos commitments con dueAt
   953→   953→      const newCommitments = (result.state.commitments ?? []).filter(
   954→   954→        (c) => c.dueAt && !previousState.commitments?.find((pc) => pc.id === c.id)
   955→   955→      );
   956→   956→      if (newCommitments.length > 0) {
   957→   957→        const userId = result.state.userId ?? "default";
   958→   958→        for (const c of newCommitments) {
   959→   959→          scheduleReminderNotification(c as any, userId);
   960→   960→          // 🔴 NUEVO: programar setTimeout preciso para que la notificación
   961→   961→          // dispare EXACTO cuando llega la hora, no en el próximo heartbeat tick.
   962→   962→          schedulePreciseTimeout(c.id, c.dueAt!, c.title, userId);
   963→   963→        }
   964→   964→        // 🔴 Request notification permission — best effort
   965→   965→        requestNotificationPermission().catch(() => {});
   966→   966→      }
   967→   967→      if (koruTurnId) {
   968→   968→        // Preservar ids de los items del stream: si el resultado final tiene un item
   969→   969→        // del mismo tipo que uno ya renderizado en el stream, mantener su id (e
   970→   970→        // identidad de React) para no desmontar/remontar el componente y perder su
   971→   971→        // estado interno (animación de progreso del WebNavCardA, visibleCount, etc.).
   972→   972→        const consumedTypes = new Set<string>();
   973→   973→        const stableItems = result.items.map((item) => {
   974→   974→          const type = item.uiBlock?.type ?? item.tag;
   975→   975→          if (type && !consumedTypes.has(type)) {
   976→   976→            const streamed = streamedItemsByType.get(type);
   977→   977→            if (streamed) {
   978→   978→              consumedTypes.add(type);
   979→   979→              return { ...item, id: streamed.id };
   980→   980→            }
   981→   981→          }
   982→   982→          return item;
   983→   983→        });
   984→   984→        commitChatTurns((prev) =>
   985→   985→          prev.map((turn) =>
   986→   986→            turn.id === koruTurnId
   987→   987→              ? { ...turn, text: agentResult.reply, items: stableItems, status: "done" as const, mascotState: agentResult.mascotState ?? "idle" }
   988→   988→              : turn,
   989→   989→          ),
   990→   990→        );
   991→   991→        // 🔴 Voice: si voice está activado, Koru habla la respuesta
   992→   992→        if (voiceEnabled && agentResult.reply) {
   993→   993→          speak(agentResult.reply);
   994→   994→        }
   995→   995→        // 🔴 Memory toast: si hay memoryCandidates en la respuesta, mostrar toast.
   996→   996→        // Buscar en memoryCandidates top-level Y dentro de toolResults (save_memory tool).
   997→   997→        const topCandidates = (agentResult.memoryCandidates ?? []);
   998→   998→        const toolCandidates = (agentResult.toolResults ?? []).flatMap((tr: any) => {
   999→   999→          const data = tr.data;
  1000→  1000→          if (data?.type === "memory_capture" && Array.isArray(data.memoryCandidates)) return data.memoryCandidates;
  1001→  1001→          if (data?.type === "personal_capture" && Array.isArray(data.memoryCandidates)) return data.memoryCandidates;
  1002→  1002→          return [];
  1003→  1003→        });
  1004→  1004→        const allCandidates = [...topCandidates, ...toolCandidates].filter(
  1005→  1005→          (m: any) => m && typeof m.text === "string" && m.text.length > 4,
  1006→  1006→        );
  1007→  1007→        if (allCandidates.length > 0 && !previousState.ephemeralMode && previousState.durableMemoryEnabled) {
  1008→  1008→          const first = allCandidates[0];
  1009→  1009→          // Solo mostrar si NO existe ya una memoria con el mismo texto (deduplicación visual)
  1010→  1010→          const normalizedNew = first.text.toLowerCase().replace(/[áéíóú]/g, (m: string) => ({ á: "a", é: "e", í: "i", ó: "o", ú: "u" }[m]!)).replace(/[^a-z0-9\s]/g, "").trim();
  1011→  1011→          const alreadyExists = previousState.memories.some((existing: any) => {
  1012→  1012→            const norm = existing.text.toLowerCase().replace(/[áéíóú]/g, (m: string) => ({ á: "a", é: "e", í: "i", ó: "o", ú: "u" }[m]!)).replace(/[^a-z0-9\s]/g, "").trim();
  1013→  1013→            return norm === normalizedNew;
  1014→  1014→          });
  1015→  1015→          if (!alreadyExists) {
  1016→  1016→            setMemoryToast({
  1017→  1017→              id: `toast_${Date.now()}`,
  1018→  1018→              kind: first.kind,
  1019→  1019→              text: first.text,
  1020→  1020→            });
  1021→  1021→            // Auto-dismiss después de 6 segundos
  1022→  1022→            setTimeout(() => {
  1023→  1023→              setMemoryToast((current) => (current?.text === first.text ? null : current));
  1024→  1024→            }, 6000);
  1025→  1025→          }
  1026→  1026→        }
  1027→  1027→      }
  1028→  1028→      writeAuditEvent({
  1029→  1029→        type: "turn_analyzed",
  1030→  1030→        engine: "backend_agent_loop",
  1031→  1031→        input: text,
  1032→  1032→        transcriptSource,
  1033→  1033→        historyUsed: domainHistory,
  1034→  1034→        response: agentResult.reply,
  1035→  1035→        understanding: agentResult.understanding,
  1036→  1036→        suggestedActions: agentResult.suggestedActions,
  1037→  1037→        toolResults: agentResult.toolResults,
  1038→  1038→        provider: agentResult.provider,
  1039→  1039→        model: agentResult.model,
  1040→  1040→        fallbackReason: agentResult.fallbackReason,
  1041→  1041→        items: auditTurnItems(result.items),
  1042→  1042→        state: auditStateSnapshot(result.state),
  1043→  1043→        delta: auditStateDelta(previousState, result.state),
  1044→  1044→      });
  1045→  1045→      return { response: agentResult.reply, items: result.items, state: result.state, mascotState: agentResult.mascotState, koruTurnId };
  1046→  1046→    } finally {
  1047→  1047→      setProcessing(false);
  1048→  1048→      setActivity(null);
  1049→  1049→      setPhase(null);
  1050→  1050→    }
  1051→  1051→  }

  1254→  1254→          const next = {

  1256→  1256→            actions: prev.actions.map((action) => (action.id === original.id ? patchedAction : action)),
  1257→  1257→            updatedAt: finishedAt,

  1271→  1271→            state: auditStateSnapshot(next),
  1272→  1272→            delta: auditStateDelta(prev, next),
  1273→  1273→          });
  1274→  1274→          return next;
  1275→  1275→        });
  1276→  1276→        replaceActionInChat(patchedItem);
  1277→  1277→      } finally {

  1280→  1280→    })();

  1284→  1284→  function reviewAction(id: string, approve: boolean): KoruTurnItem | null {
  1285→  1285→    const currentState = domainStateRef.current;
  1286→  1286→    const original = currentState.actions.find((action) => action.id === id);
  1287→  1287→    if (approve && (original?.kind === "web_research" || original?.kind === "world_signal")) {
  1288→  1288→      return runReadonlyWebAction(original);
  1289→  1289→    }
  1290→  1290→
  1291→  1291→    const nextState = approve ? approveAndExecuteAction(currentState, id) : rejectAction(currentState, id);
  1292→  1292→    const reviewed = nextState.actions.find((action) => action.id === id);
  1293→  1293→    if (!reviewed) return null;
  1294→  1294→    commitDomainState(nextState);
  1295→  1295→    const item = actionToTurnItem(reviewed);
  1296→  1296→    replaceActionInChat(item);
  1297→  1297→    writeAuditEvent({
  1298→  1298→      type: "action_reviewed",
  1299→  1299→      actionId: id,
  1300→  1300→      approved: approve,
  1301→  1301→      item: auditTurnItems([item])[0],
  1302→  1302→      state: auditStateSnapshot(nextState),
  1303→  1303→      delta: auditStateDelta(currentState, nextState),
  1304→  1304→    });
  1305→  1305→    appendKoruConfirmation(approve ? actionConfirmationText(item) : "Lo suelto. No hice nada con eso.");
  1306→  1306→
  1307→  1307→    return item;
  1308→  1308→  }
