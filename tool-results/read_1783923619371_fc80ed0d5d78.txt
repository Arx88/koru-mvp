   466→function plainLower(value: string): string {
   467→  return value.normalize("NFD").replace(/[\u0300-\u036f]/g, "").toLowerCase();
   468→}

   470→function timeFromText(value: string): string | undefined {
   471→  const match = /\b(?:a\s+las|las)\s+(\d{1,2})(?::(\d{2}))?\b/i.exec(plainLower(value));
   472→  if (!match) return undefined;
   473→  const hour = Math.max(0, Math.min(23, Number(match[1])));
   474→  const minute = match[2] ? Math.max(0, Math.min(59, Number(match[2]))) : 0;
   475→  return `${hour.toString().padStart(2, "0")}:${minute.toString().padStart(2, "0")}`;
   476→}

   478→function createId(prefix: string): string {
   479→  if (typeof crypto !== "undefined" && "randomUUID" in crypto) {
   480→    return `${prefix}_${crypto.randomUUID().replace(/-/g, "").slice(0, 12)}`;
   481→  }
   482→  return `${prefix}_${Math.random().toString(36).slice(2, 12)}`;
   483→}

   485→/**
   486→ * Genera iniciales de un nombre de equipo (máx 3 letras).
   487→ * "Spain" → "SPA", "Real Madrid" → "MAD", "Boca Juniors" → "BJU", "PSG" → "PSG".
   488→ */
   489→function initialsFromName(name: string): string {
   490→  const clean = String(name ?? "").trim().toUpperCase();
   491→  if (!clean) return "???";
   492→  // Si ya es sigla (PSG, MLS, etc), devolverla
   493→  if (/^[A-Z]{2,4}$/.test(clean) && clean.length <= 4) return clean;
   494→  const words = clean.split(/\s+/).filter(w => w.length > 0);
   495→  if (words.length === 1) return words[0].slice(0, 3);
   496→  // Tomar primera letra de las primeras 3 palabras, saltando artículos
   497→  const filtered = words.filter(w => !["DE", "DEL", "LA", "LAS", "EL", "LOS", "Y", "FC", "CF"].includes(w));

   501→/**
   502→ * Formatea una fecha ISO a "DD/MM YYYY" o "Hoy" / "Ayer" según corresponda.
   503→ */
   504→function formatMatchDate(iso: string): string {
   505→  try {
   506→    const d = new Date(iso);
   507→    if (isNaN(d.getTime())) return "";
   508→    const now = new Date();
   509→    const todayStart = new Date(now.getFullYear(), now.getMonth(), now.getDate());
   510→    const dStart = new Date(d.getFullYear(), d.getMonth(), d.getDate());
   511→    const diffDays = Math.round((todayStart.getTime() - dStart.getTime()) / (24 * 60 * 60 * 1000));
   512→    if (diffDays === 0) return "Hoy";
   513→    if (diffDays === 1) return "Ayer";
   514→    if (diffDays === -1) return "Mañana";
   515→    const day = String(d.getDate()).padStart(2, "0");
   516→    const month = String(d.getMonth() + 1).padStart(2, "0");
   517→    return `${day}/${month}`;
   518→  } catch { return ""; }
   519→}

   529→function extractJsonBlock(text: string): string {
   530→  const start = text.indexOf("{");
   531→  if (start === -1) return text;
   532→  let depth = 0;
   533→  let inString = false;
   534→  let escapeNext = false;
   535→  for (let i = start; i < text.length; i++) {
   536→    const c = text[i];
   537→    if (escapeNext) { escapeNext = false; continue; }
   538→    if (c === "\\") { escapeNext = true; continue; }
   539→    if (c === '"' && !inString) { inString = true; continue; }
   540→    if (c === '"' && inString) { inString = false; continue; }
   541→    if (inString) continue;
   542→    if (c === "{") depth++;
   543→    if (c === "}") { depth--; if (depth === 0) return text.slice(start, i + 1); }
   544→  }
   545→  return text;
   546→}

   548→function safeJsonObjectFromContent(raw: string): Record<string, unknown> {
   549→  const direct = safeJsonParse(raw);
   550→  if (direct.reply !== undefined || direct.uiBlocks !== undefined) return direct;
   551→  const extracted = safeJsonParse(extractJsonBlock(raw));
   552→  if (extracted.reply !== undefined || extracted.uiBlocks !== undefined) return extracted;
   553→  // Dirty extraction: fields individually with regex
   554→  const reply = extractStringField(raw, "reply");
   555→  const mascotState = extractStringField(raw, "mascotState") || extractStringField(raw, "mascot_state");
   556→  if (reply && reply.length > 3) {
   557→    return { reply, mascotState: mascotState || "idle", uiBlocks: [] };
   558→  }
   559→  return {};
   560→}

   562→function extractStringField(raw: string, field: string): string | undefined {
   563→  const idx = raw.toLowerCase().indexOf(`"${field.toLowerCase()}"`);
   564→  if (idx === -1) return undefined;
   565→  let start = raw.indexOf('"', idx + field.length + 2);
   566→  if (start === -1) return undefined;
   567→  start++;
   568→  let i = start;
   569→  let escaped = false;
   570→  while (i < raw.length) {
   571→    const c = raw[i];
   572→    if (escaped) { escaped = false; i++; continue; }
   573→    if (c === '\\') { escaped = true; i++; continue; }
   574→    if (c === '"') break;
   575→    i++;
   576→  }
   577→  return raw.slice(start, i);
   578→}

   624→async function callMinimax(
   625→  config: ProviderConfig,
   626→  messages: ChatMessage[],
   627→  timeoutMs: number,
   628→  toolsEnabled = true,
   629→): Promise<ProviderResult> {
   630→  const accessToken = config.minimaxAccessToken;
   631→  if (!accessToken) throw new Error("MiniMax access token not configured");
   632→  logger.info("callMinimax", "Requesting MiniMax", { model: "MiniMax-M2.7", msgCount: messages.length, toolsEnabled });
   633→  const minimaxMessages = messages.map((m) => {
   634→    if (m.role === "tool") {
   635→      let content = m.content ?? "";
   636→      try {
   637→        const data = JSON.parse(content);
   638→        if (data.type === "search" && Array.isArray(data.sources)) {
   639→          const formatted = data.sources
   640→            .map((s: any, i: number) => {
   641→              const text = s.content || s.snippet || "";
   642→              return `${i + 1}. ${s.title} (${s.domain})\n${text}`;
   643→            })
   644→            .filter((s: string) => s.trim().length > 3)
   645→            .join("\n\n");
   646→          content = formatted || `Búsqueda: ${data.title || ""}`;
   647→        } else if (data.type === "weather") {
   648→          content = `Clima - Ciudad: ${data.city || "?"}, Ahora: ${data.now || "?"}, Rango: ${data.range || "?"}, Lluvia: ${data.rain || "?"}, Viento: ${data.wind || "?"}`;
   649→        }
   650→      } catch {
   651→        // mantener contenido original si no es JSON
   652→      }
   653→      return { role: "user" as const, content: `Resultado de herramienta (${m.tool_call_id ?? "unknown"}):\n${content}` };
   654→    }
   655→    if (m.role === "assistant" && m.tool_calls) {
   656→      return {
   657→        role: "assistant" as const,
   658→        content: m.content ?? `Voy a usar herramientas: ${m.tool_calls.map((t) => t.function.name).join(", ")}`,
   659→      };
   660→    }
   661→    return { role: m.role, content: m.content ?? "" };
   662→  });
   663→  const response = await fetchWithTimeout("https://api.minimax.io/v1/chat/completions", {
   664→    method: "POST",
   665→    headers: {
   666→      Authorization: `Bearer ${accessToken}`,
   667→      "Content-Type": "application/json",
   668→      Accept: "application/json",
   669→    },
   670→    body: JSON.stringify({
   671→      model: "MiniMax-M2.7",
   672→      messages: minimaxMessages,
   673→      ...(toolsEnabled ? { tools: ALL_TOOL_DEFINITIONS, tool_choice: "auto" } : {}),
   674→      temperature: 0.25,
   675→      top_p: 0.95,
   676→      max_tokens: 8192,
   677→      stream: false,
   678→      reasoning_split: true,
   679→    }),
   680→  }, timeoutMs);
   681→  const data = await response.json().catch(() => ({}));
   682→  const choice = asRecord(asArray(asRecord(data).choices)[0]);
   683→  const assistantMsg = asRecord(choice.message);
   684→  const content = asString(assistantMsg.content) ?? "";
   685→  const toolCalls = asArray(assistantMsg.tool_calls);
   686→  const reasoningContent = asString(assistantMsg.reasoning_content) ?? "";
   687→  logger.info("callMinimax", `Response HTTP ${response.status}`, { contentPreview: content.slice(0, 500), reasoningPreview: reasoningContent.slice(0, 200), hasTools: toolCalls.length > 0, usage: dump(data.usage, 300) });
   688→  if (!response.ok || !hasUsableAssistantMessage(data)) {
   689→    logger.error("callMinimax", `MiniMax returned ${response.status}`, { body: dump(data, 1000) });
   690→    throw new Error(`MiniMax returned ${response.status}`);
   691→  }
   692→  return {
   693→    provider: "minimax",
   694→    model: asString(asRecord(data).model) ?? "MiniMax-M2.7",
   695→    message: assistantMsg as ProviderMessage,
   696→  };
   697→}

   773→    response = await fetchWithTimeout(providerUrl(config.nvidiaBaseUrl, "/v1/chat/completions"), {
   774→      method: "POST",
   775→      headers,
   776→      body: JSON.stringify(body),

   782→      method: "POST",
   783→      headers,
   784→      body: JSON.stringify(body),
   785→    }, timeoutMs);
   786→  }
   787→  const data = await response.json().catch(() => ({}));
   788→  if (!response.ok || !hasUsableAssistantMessage(data)) {
   789→    throw new Error(`NVIDIA returned ${response.status}`);
   790→  }
   791→  const choice = asRecord(asArray(asRecord(data).choices)[0]);
   792→  return {
   793→    provider: "nvidia",
   794→    model: asString(asRecord(data).model) ?? config.nvidiaModel,
   795→    message: asRecord(choice.message) as ProviderMessage,
   796→  };

   938→function selectModelForInput(

   944→  // Si no hay fast/medium model configurado, usar el default
   945→  if (!config.nvidiaFastModel && !config.nvidiaMediumModel) return undefined;
   946→

   956→    return config.nvidiaModel; // nvidiaModel = ultra por default
   957→  }
   958→
   959→  // Normal → medium si está disponible, sino ultra
   960→  if (config.nvidiaMediumModel) {
   961→    logger.info("modelRouter", "Using medium model", { model: config.nvidiaMediumModel, reason: "normal input" });
   962→    return config.nvidiaMediumModel;
   963→  }

   984→  if (preferredProvider === "bluesminds" && bluesmindsAvailable) {
   985→    try {
   986→      return await callBlueSminds(config, messages, Math.min(60_000, timeoutMs), toolsEnabled);
   987→    } catch (err: any) {
   988→      if (isRateLimitError(err)) throw err;
   989→      logger.warn("callProvider", "Preferred BlueSminds failed, falling through", { reason: err?.message });
   990→    }
   991→  }
   992→
   993→  if (preferredProvider === "minimax" && config.minimaxAccessToken) {
   994→    try {
   995→      const result = await callMinimax(config, messages, timeoutMs, toolsEnabled);
   996→      if (providerResultIsValid(result)) {
   997→        logger.info("callProvider", "Preferred MiniMax succeeded");
   998→        return result;
   999→      }
  1000→      logger.warn("callProvider", "Preferred MiniMax responded but invalid, falling through");
  1001→    } catch (err: any) {
  1002→      logger.warn("callProvider", "Preferred MiniMax failed, falling through", { reason: err?.message });
  1003→    }
  1004→  }
  1005→
  1006→  if (preferredProvider === "nvidia" && nvidiaAvailable) {
  1007→    try {
  1008→      const result = await callNvidia(config, messages, nvidiaTimeoutMs(config, isOllama, timeoutMs), toolsEnabled, availableTools, modelOverride);
  1009→      if (providerResultIsValid(result)) return result;
  1010→      logger.warn("callProvider", "Preferred NVIDIA responded but invalid, falling back");
  1011→    } catch (err: any) {
  1012→      if (isRateLimitError(err)) throw err;
  1013→      logger.warn("callProvider", "Preferred NVIDIA failed, falling back", { reason: err?.message });
  1014→    }
  1015→  }
  1016→
  1017→  if (preferredProvider === "openrouter" && config.openRouterKeys.length) {
  1018→    return callOpenRouter(config, messages, Math.min(115_000, timeoutMs), toolsEnabled);
  1019→  }
  1020→
  1021→  // FLUJO NORMAL (sin preferencia o preferencia fallida)
  1022→  // BlueSminds es el proveedor principal si está configurado.
  1023→  if (bluesmindsAvailable && (!preferredProvider || preferredProvider === "bluesminds")) {
  1024→    try {
  1025→      const result = await callBlueSminds(config, messages, Math.min(60_000, timeoutMs), toolsEnabled);
  1026→      if (providerResultIsValid(result)) {
  1027→        logger.info("callProvider", "BlueSminds succeeded");
  1028→        return result;
  1029→      }
  1030→      logger.warn("callProvider", "BlueSminds responded but invalid, falling through");
  1031→    } catch (err: any) {
  1032→      if (isRateLimitError(err)) throw err;
  1033→      logger.warn("callProvider", "BlueSminds failed, falling through", { reason: err?.message });
  1034→    }
  1035→  }
  1036→
  1037→  // Si el usuario eligió un provider específico que no es MiniMax, no lo intentamos.
  1038→  if (config.minimaxAccessToken && (!preferredProvider || preferredProvider === "minimax")) {
  1039→    try {
  1040→      const result = await callMinimax(config, messages, timeoutMs, toolsEnabled);
  1041→      if (providerResultIsValid(result)) {
  1042→        logger.info("callProvider", "MiniMax succeeded");
  1043→        return result;
  1044→      }
  1045→      logger.warn("callProvider", "MiniMax responded but invalid, falling through");
  1046→    } catch (err: any) {
  1047→      logger.warn("callProvider", "MiniMax failed, falling through", { reason: err?.message });
  1048→    }
  1049→  }
  1050→
  1051→  if (!nvidiaAvailable) {
  1052→    return callOpenRouter(config, messages, Math.min(115_000, timeoutMs), toolsEnabled);
  1053→  }
  1054→
  1055→  // Si el usuario eligió OpenRouter, saltamos NVIDIA en el flujo normal
  1056→  if (!preferredProvider || preferredProvider !== "openrouter") {
  1057→    try {
  1058→      const result = await callNvidia(config, messages, nvidiaTimeoutMs(config, isOllama, timeoutMs), toolsEnabled, availableTools, modelOverride);
  1059→      if (providerResultIsValid(result)) return result;
  1060→      logger.warn("callProvider", "NVIDIA responded but invalid, falling back");
  1061→    } catch (err: any) {
  1062→      if (isRateLimitError(err)) throw err;
  1063→      logger.warn("callProvider", "NVIDIA failed, falling back to OpenRouter", { reason: err?.message });
  1064→    }
  1065→  }
  1066→
  1067→  if (config.openRouterKeys.length) {
  1068→    return callOpenRouter(config, messages, Math.min(115_000, timeoutMs), toolsEnabled);
  1069→  }
  1070→
  1071→  throw new Error("Ningún proveedor de IA respondió. Verificá la conexión o las credenciales.");
  1072→}

  1183→async function searchDuckDuckGo(query: string): Promise<AssistantSource[]> {
  1184→  const response = await fetchWithTimeout(`https://duckduckgo.com/html/?q=${encodeURIComponent(query)}`, {
  1185→    headers: {
  1186→      "User-Agent": "Mozilla/5.0 KoruAgent/1.0",
  1187→      Accept: "text/html",
  1188→    },
  1189→  }, 10_000);
  1190→  const html = await response.text();
  1191→  const sources: AssistantSource[] = [];

  1193→  let match: RegExpExecArray | null;
  1194→  while ((match = resultRe.exec(html)) && sources.length < 6) {
  1195→    const rawUrl = match[1];
  1196→    let url = rawUrl;
  1197→    try {
  1198→      const parsed = new URL(rawUrl, "https://duckduckgo.com");
  1199→      url = parsed.searchParams.get("uddg") ?? parsed.href;
  1200→    } catch {
  1201→      url = rawUrl;
  1202→    }
  1203→    if (!/^https?:\/\//i.test(url)) continue;
  1204→    const domain = (() => {
  1205→      try {
  1206→        return new URL(url).hostname;
  1207→      } catch {
  1208→        return "";
  1209→      }
  1210→    })();
  1211→    if (/duckduckgo\.com|google\.com|bing\.com/i.test(domain)) continue;
  1212→    sources.push(sourceFromUrl(htmlText(match[2]), url, htmlText(match[3]).slice(0, 260)));
  1213→  }
  1214→  return sources;
  1215→}

  1232→async function fetchPageContent(url: string, maxChars = 1200): Promise<string> {
  1233→  try {
  1234→    const res = await fetchWithTimeout(url, { headers: { "User-Agent": "Mozilla/5.0 (Windows NT 10.0; Win64; x64)" } }, 15_000);
  1235→    const html = await res.text();
  1236→
  1237→    // Extraer contenido principal: intentar <article>, luego <main>, luego clases comunes
  1238→    let contentHtml = html;
  1239→    const articleMatch = html.match(/<article\b[^>]*>([\s\S]*?)<\/article>/i);
  1240→    if (articleMatch) {
  1241→      contentHtml = articleMatch[1];
  1242→    } else {
  1243→      const mainMatch = html.match(/<main\b[^>]*>([\s\S]*?)<\/main>/i);
  1244→      if (mainMatch) {
  1245→        contentHtml = mainMatch[1];
  1246→      } else {
  1247→        const classMatch = html.match(/<(?:div|section)\b[^>]*\b(?:class|id)="(?:entry-content|article-body|post-content|story-body|content-body|texto-nota|nota-content|article__content|main-content|nota-texto|content-text)[^"]*"[^>]*>([\s\S]*?)<\/\1>/i);
  1248→        if (classMatch) {
  1249→          contentHtml = classMatch[2] ?? classMatch[1];
  1250→        }
  1251→      }
  1252→    }
  1253→
  1254→    // Si no encontramos nada semántico, buscar la zona con más <p> consecutivos
  1255→    if (contentHtml === html) {
  1256→      const pMatches = html.matchAll(/<p\b[^>]*>([\s\S]*?)<\/p>/gi);
  1257→      const paragraphs: string[] = [];
  1258→      for (const m of pMatches) {
  1259→        const text = m[1].replace(/<[^>]+>/g, " ").trim();
  1260→        if (text.length > 40) paragraphs.push(text);
  1261→      }
  1262→      if (paragraphs.length > 0) {
  1263→        contentHtml = paragraphs.slice(0, 8).join(" ");
  1264→      }
  1265→    }
  1266→
  1267→    const text = contentHtml
  1268→      .replace(/<script\b[^<]*(?:(?!<\/script>)<[^<]*)*<\/script>/gi, "")
  1269→      .replace(/<style\b[^<]*(?:(?!<\/style>)<[^<]*)*<\/style>/gi, "")
  1270→      .replace(/<[^>]+>/g, " ")
  1271→      .replace(/\s+/g, " ")
  1272→      .trim();
  1273→    return text.slice(0, maxChars);
  1274→  } catch {
  1275→    return "";
  1276→  }
  1277→}

  1279→export async function runSearch(
  1280→  args: Record<string, unknown>,
  1281→  shopping = false,
  1282→  extractorCtx?: { userInput: string; chatFn: ExtractorChatFn; onDeferredChunk?: (block: UiBlock) => void },
  1283→): Promise<SearchData> {
  1284→  const query = cleanText(args.query, "noticias importantes hoy");
  1285→  const mode = shopping ? "shopping" : cleanText(args.mode, "research") as SearchData["mode"];
  1286→  const expanded = mode === "world"
  1287→    ? `${query} ultimos 30 dias tendencias`
  1288→    : mode === "news"
  1289→      ? `${query} noticias recientes`
  1290→      : shopping
  1291→        ? `${query} precio opiniones entrega`
  1292→        : query;
  1293→  const gdelt = mode === "news" || mode === "world" ? await searchGdelt(expanded).catch(() => []) : [];
  1294→  const duck = gdelt.length ? [] : await searchDuckDuckGo(expanded).catch(() => []);
  1295→  let sources = [...gdelt, ...duck].slice(0, 6);
  1296→  const comparisonItems = shopping
  1297→    ? sources.slice(0, 4).map((source, index) => ({
  1298→        title: source.title,
  1299→        vendor: source.domain,
  1300→        url: source.url,
  1301→        evidence: source.snippet,
  1302→        score: Math.max(55, 88 - index * 8),
  1303→      }))
  1304→    : undefined;
  1305→
  1306→  // Scrape content from first 3 sources for synthesis
  1307→  for (let i = 0; i < Math.min(sources.length, 3); i++) {
  1308→    sources[i].content = await fetchPageContent(sources[i].url, 1200);
  1309→  }
  1310→
  1311→  // ── Extracción de estructura validada (NO BLOQUEANTE) ──
  1312→  // runSearch devuelve inmediatamente con los sources (web_nav). La extracción
  1313→  // corre como promesa diferida que el llamador espera EN PARALELO con la
  1314→  // composición del reply. Así el extractor (~11s) queda oculto detrás del
  1315→  // Composer, no suma latencia al turno. Si encuentra datos validados, se
  1316→  // convierten en un data_card que se adjunta al resultado final.
  1317→  let deferredDataCard: Promise<UiBlock | null> | undefined;
  1318→  if (extractorCtx && !shopping && sources.length > 0) {
  1319→    const sourcesCopy = sources.map((s) => ({ ...s }));
  1320→    const userInput = extractorCtx.userInput;
  1321→    const chatFn = extractorCtx.chatFn;
  1322→    const extractStart = Date.now();
  1323→    deferredDataCard = (async (): Promise<UiBlock | null> => {
  1324→      try {
  1325→        const extracted = await extractStructuredData({ userInput, sources: sourcesCopy, chatFn });
  1326→        logger.info("runSearch", "Structure extraction (deferred)", {
  1327→          extracted: extracted ? `${extracted.items.length} items` : "none",
  1328→          durationMs: Date.now() - extractStart,
  1329→        });
  1330→        if (!extracted || extracted.items.length === 0) return null;
  1331→        return {
  1332→          type: "data_card" as const,
  1333→          title: extracted.title,
  1334→          items: extracted.items.map((it) => ({
  1335→            label: it.label,
  1336→            value: it.value,
  1337→            detail: it.detail,
  1338→            quote: it.quote,
  1339→            sourceUrl: it.sourceUrl,
  1340→            sourceDomain: it.sourceDomain,
  1341→          })),
  1342→        };
  1343→      } catch (err: any) {
  1344→        logger.warn("runSearch", "Deferred structure extraction failed (non-fatal)", { reason: err?.message });
  1345→        return null;
  1346→      }
  1347→    })();
  1348→  }
  1349→
  1350→  return {
  1351→    type: "search",
  1352→    mode,
  1353→    title: shopping ? "Comparativa" : mode === "news" ? "Noticias importantes" : mode === "world" ? "El mundo esta hablando de esto" : "Busqueda",
  1354→    summary: sources.length ? "" : "No pude conseguir fuentes útiles con los conectores abiertos. No inventes resultados.",
  1355→    sources,
  1356→    comparisonItems,
  1357→    deferredDataCard,
  1358→  };
  1359→}

  1441→function isRecordInPeriod(record: LifeRecord, period: string): boolean {
  1442→  const normalized = plainLower(period);
  1443→  if (!normalized || !record.createdAt) return true;
  1444→  const created = new Date(record.createdAt);
  1445→  if (Number.isNaN(created.getTime())) return true;
  1446→  const now = new Date();
  1447→  const ageMs = now.getTime() - created.getTime();
  1448→  if (/\bhoy|today\b/.test(normalized)) return ageMs >= 0 && ageMs <= 36 * 60 * 60 * 1000;
  1449→  if (/\bsemana|week\b/.test(normalized)) return ageMs >= 0 && ageMs <= 8 * 24 * 60 * 60 * 1000;
  1450→  if (/\bmes|month\b/.test(normalized)) return ageMs >= 0 && ageMs <= 32 * 24 * 60 * 60 * 1000;
  1451→  return true;

  1493→function queryTokens(query: string): string[] {
  1494→  return plainLower(query)
  1495→    .replace(/[^\p{L}\p{N}\s:/._-]+/gu, " ")
  1496→    .split(/\s+/)
  1497→    .map((token) => token.trim())
  1498→    .filter((token) => token.length > 1 && !["que", "con", "por", "sin", "mas", "los", "las", "del", "una", "uno", "mis", "tengo", "sobre", "guarde", "guardado", "guardaste"].includes(token));

  1501→function semanticRecordMatches(records: LifeRecord[], query: string, limit = 8): LifeRecord[] {
  1502→  const tokens = queryTokens(query);
  1503→  if (!tokens.length) return [];
  1504→  return records
  1505→    .map((record) => {
  1506→      const haystack = plainLower(recordSearchText(record));
  1507→      const score = tokens.reduce((sum, token) => sum + (haystack.includes(token) ? 1 : 0), 0);
  1508→      return { record, score };
  1509→    })
  1510→    .filter((item) => item.score > 0)
  1511→    .sort((a, b) => b.score - a.score)
  1512→    .map((item) => item.record)
  1513→    .slice(0, limit);
  1514→}

  1516→function uniqueLifeRecords(records: LifeRecord[]): LifeRecord[] {
  1517→  const seen = new Set<string>();
  1518→  return records.filter((record) => {
  1519→    const key = `${record.id}|${record.domain}|${record.kind}|${plainLower(record.title)}|${plainLower(record.value ?? "")}|${plainLower(record.url ?? "")}`;
  1520→    if (seen.has(key)) return false;
  1521→    seen.add(key);
  1522→    return true;
  1523→  });
  1524→}
