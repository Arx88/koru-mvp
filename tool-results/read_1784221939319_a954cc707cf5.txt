    19→// 🔴 P2 — Platform detection para el botón "Navegar" de route_map.
    20→// iOS detection incluye el iPad con iOS 13+ que se reporta como MacIntel.
    21→function isIOS(): boolean {
    22→  if (typeof navigator === "undefined") return false;
    23→  return (
    24→    /iPad|iPhone|iPod/.test(navigator.userAgent) ||
    25→    (navigator.platform === "MacIntel" && (navigator.maxTouchPoints ?? 0) > 1)
    26→  );
    27→}
    28→
    29→function isAndroid(): boolean {
    30→  if (typeof navigator === "undefined") return false;
    31→  return /Android/i.test(navigator.userAgent);
    32→}

    34→/**
    35→ * 🔴 P2 — Construye la URL de navegación según plataforma y datos disponibles.
    36→ * - Si lat/lng están presentes:
    37→ *   • Android: geo:${lat},${lng}?q=${destination}
    38→ *   • iOS:     maps://?daddr=${lat},${lng}
    39→ *   • Desktop: https://www.google.com/maps/dir/?api=1&destination=${dest}
    40→ * - Si NO hay lat/lng, cae a esquemas address-based:
    41→ *   • Android: geo:0,0?q=${destination}
    42→ *   • iOS:     maps://?daddr=${destination}
    43→ *   • Desktop: https://www.google.com/maps/dir/?api=1&destination=${dest}
    44→ * Devuelve null si no hay destino (ni to ni from ni lat/lng).
    45→ */
    46→function buildNavUrl(opts: {
    47→  destination: string;
    48→  lat?: number;
    49→  lng?: number;
    50→}): string | null {
    51→  const { destination, lat, lng } = opts;
    52→  const hasCoords = typeof lat === "number" && typeof lng === "number";
    53→  if (isAndroid()) {
    54→    if (hasCoords) {
    55→      return `geo:${lat},${lng}?q=${encodeURIComponent(destination)}`;
    56→    }
    57→    return `geo:0,0?q=${encodeURIComponent(destination)}`;
    58→  }
    59→  if (isIOS()) {
    60→    if (hasCoords) {
    61→      return `maps://?daddr=${lat},${lng}`;
    62→    }
    63→    return `maps://?daddr=${encodeURIComponent(destination)}`;
    64→  }
    65→  // Desktop (o fallback)
    66→  return `https://www.google.com/maps/dir/?api=1&destination=${encodeURIComponent(destination)}`;
    67→}

    69→// 🔴 v2: ComparisonBar — barra comparativa para stats de fútbol (o cualquier
    70→// matchup home/away). Dibuja dos segmentos coloreados con team colors,
    71→// mostrando visualmente quién domina esa stat.
    72→function ComparisonBar({ bar }: { bar: NonNullable<DetailRow["bar"]> }) {
    73→  const { homeValue, awayValue, isPercent, homeColor, awayColor } = bar;
    74→  const total = (homeValue ?? 0) + (awayValue ?? 0);
    75→  const homePct = total > 0 ? Math.max(2, Math.min(98, (homeValue / total) * 100)) : 50;
    76→  const awayPct = 100 - homePct;
    77→  const hc = homeColor || "#2d6a4f";
    78→  const ac = awayColor || "#8363f9";
    79→  return (
    80→    <div className="koru-comparison-bar" role="presentation">
    81→      <span className="koru-comparison-value home" style={{ color: hc }}>
    82→        {isPercent ? `${Math.round(homeValue)}%` : homeValue}
    83→      </span>
    84→      <div className="koru-comparison-track">
    85→        <div
    86→          className="koru-comparison-fill home"
    87→          style={{ width: `${homePct}%`, background: hc }}
    88→        />
    89→        <div
    90→          className="koru-comparison-fill away"
    91→          style={{ width: `${awayPct}%`, background: ac }}
    92→        />
    93→      </div>
    94→      <span className="koru-comparison-value away" style={{ color: ac }}>
    95→        {isPercent ? `${Math.round(awayValue)}%` : awayValue}
    96→      </span>
    97→    </div>
    98→  );
    99→}

   106→// 🔴 v2: Extrae el video ID de una URL de YouTube (youtu.be / watch?v= / embed / shorts).
   107→// Devuelve null si no es una URL válida de YouTube.
   108→function youtubeId(url: string): string | null {
   109→  try {
   110→    const u = new URL(url);
   111→    if (u.hostname.includes("youtu.be")) {
   112→      const id = u.pathname.slice(1).split("/")[0];
   113→      return id || null;
   114→    }
   115→    if (u.hostname.includes("youtube.com")) {
   116→      const v = u.searchParams.get("v");
   117→      if (v) return v;
   118→      const m = u.pathname.match(/\/(?:embed|shorts)\/([\w-]+)/);
   119→      if (m) return m[1];
   120→    }
   121→    return null;
   122→  } catch {
   123→    return null;
   124→  }
   125→}

   435→function FormationPitch({ pitch }: { pitch: NonNullable<Extract<DetailSection, { kind: "pitch" }>["pitch"]> }) {
   436→  const { homeFormation, awayFormation, homePlayers, awayPlayers, homeColor, awayColor, homeName, awayName } = pitch;
   437→  const hc = homeColor || "#2d6a4f";
   438→  const ac = awayColor || "#8363f9";
   439→
   440→  // Parse formation string "4-3-3" → [4, 3, 3]
   441→  function parseFormation(f: string): number[] {
   442→    return f.split("-").map(n => parseInt(n, 10) || 0).filter(n => n > 0);
   443→  }
   444→
   445→  // Posicionar jugadores en una mitad de cancha (50% vertical)
   446→  // home juega "hacia arriba" (de abajo hacia arriba), away "hacia abajo"
   447→  // Cada línea del formation se posiciona a un Y% creciente desde el arquero.
   448→  function positionPlayers(players: Array<{ number?: string; name: string; position?: string }>, formation: string, isHome: boolean): Array<{ x: number; y: number; player: typeof players[0] }> {
   449→    if (players.length === 0) return [];
   450→    const lines = parseFormation(formation);
   451→    const result: Array<{ x: number; y: number; player: typeof players[0] }> = [];
   452→    let playerIdx = 0;
   453→
   454→    // GK siempre primero, en la línea de gol
   455→    if (players[0]) {
   456→      result.push({ x: 50, y: isHome ? 95 : 5, player: players[0] });
   457→      playerIdx = 1;
   458→    }
   459→
   460→    // Cada línea del formation: distribuir N jugadores en una fila horizontal
   461→    // Y va creciendo desde la defensa (cerca del arquero) hasta el ataque (cerca del mediocampo)
   462→    const linesCount = lines.length;
   463→    lines.forEach((count, lineIdx) => {
   464→      // Y para home: de 80% (defensa) a 55% (ataque, mediocampo)
   465→      // Y para away: de 20% (defensa) a 45% (ataque, mediocampo)
   466→      const yBase = isHome ? 80 : 20;
   467→      const yEnd = isHome ? 55 : 45;
   468→      const yRange = Math.abs(yBase - yEnd);
   469→      const yStep = linesCount > 1 ? yRange / (linesCount - 1) : 0;
   470→      const y = isHome ? yBase - lineIdx * yStep : yBase + lineIdx * yStep;
   471→
   472→      // Distribuir los N jugadores horizontalmente
   473→      for (let i = 0; i < count && playerIdx < players.length; i++) {
   474→        const x = ((i + 1) / (count + 1)) * 100;
   475→        result.push({ x, y, player: players[playerIdx] });
   476→        playerIdx++;
   477→      }
   478→    });
   479→
   480→    return result;
   481→  }
   482→
   483→  const homePositions = positionPlayers(homePlayers, homeFormation, true);
   484→  const awayPositions = positionPlayers(awayPlayers, awayFormation, false);
   485→
   486→  return (
   487→    <div className="koru-pitch-container">
   488→      <div className="koru-pitch-header">
   489→        <span style={{ color: hc }}>{awayName} · {awayFormation}</span>
   490→        <span style={{ color: ac, fontSize: 11, opacity: 0.7 }}>vs</span>
   491→        <span style={{ color: hc }}>{homeName} · {homeFormation}</span>
   492→      </div>
   493→      <div className="koru-pitch">
   494→        {/* Líneas de cancha */}
   495→        <div className="koru-pitch-line halfway" />
   496→        <div className="koru-pitch-circle" />
   497→        <div className="koru-pitch-box top" />
   498→        <div className="koru-pitch-box bottom" />
   499→
   500→        {/* Players away (arriba) */}
   501→        {awayPositions.map((pos, i) => (
   502→          <div
   503→            key={`away-${i}`}
   504→            className="koru-pitch-player away"
   505→            style={{
   506→              left: `${pos.x}%`,
   507→              top: `${pos.y}%`,
   508→              background: ac,
   509→            }}
   510→            title={`${pos.player.name}${pos.player.number ? ` #${pos.player.number}` : ""}`}
   511→          >
   512→            <span className="koru-pitch-player-num">{pos.player.number ?? ""}</span>

   514→          </div>
   515→        ))}
   516→
   517→        {/* Players home (abajo) */}
   518→        {homePositions.map((pos, i) => (

   520→            key={`home-${i}`}
   521→            className="koru-pitch-player home"
   522→            style={{

   525→              background: hc,
