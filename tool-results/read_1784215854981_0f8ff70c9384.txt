    31→    31→// Load .env manually (no dotenv dependency)
    32→    32→function loadEnv() {
    33→    33→  const envPath = join(PROJECT_ROOT, ".env");
    34→    34→  if (!existsSync(envPath)) return;
    35→    35→  const content = readFileSync(envPath, "utf8");
    36→    36→  for (const line of content.split("\n")) {
    37→    37→    const trimmed = line.trim();
    38→    38→    if (!trimmed || trimmed.startsWith("#")) continue;
    39→    39→    const eqIdx = trimmed.indexOf("=");
    40→    40→    if (eqIdx === -1) continue;
    41→    41→    const key = trimmed.slice(0, eqIdx).trim();
    42→    42→    const value = trimmed.slice(eqIdx + 1).trim();
    43→    43→    if (!process.env[key]) process.env[key] = value;
    44→    44→  }
    45→    45→}

   575→   575→// ── Cache static files in memory (avoid disk I/O on every request) ──
   576→   576→const staticCache = new Map<string, { data: Buffer; contentType: string }>();
   577→   577→const STATIC_DIR = join(PROJECT_ROOT, "dist");
   578→   578→
   579→   579→function getStaticFile(filePath: string): { data: Buffer; contentType: string } | null {
   580→   580→  if (staticCache.has(filePath)) return staticCache.get(filePath)!;
   581→   581→  try {
   582→   582→    const data = readFileSync(filePath);
   583→   583→    const ext = filePath.endsWith(".html") ? "text/html"
   584→   584→      : filePath.endsWith(".js") ? "application/javascript"
   585→   585→      : filePath.endsWith(".css") ? "text/css"
   586→   586→      : filePath.endsWith(".png") ? "image/png"
   587→   587→      : filePath.endsWith(".jpg") || filePath.endsWith(".jpeg") ? "image/jpeg"
   588→   588→      : filePath.endsWith(".svg") ? "image/svg+xml"
   589→   589→      : filePath.endsWith(".json") ? "application/json"
   590→   590→      : filePath.endsWith(".woff") ? "font/woff"
   591→   591→      : filePath.endsWith(".woff2") ? "font/woff2"
   592→   592→      : filePath.endsWith(".mp4") ? "video/mp4"
   593→   593→      : filePath.endsWith(".webm") ? "video/webm"
   594→   594→      : filePath.endsWith(".webp") ? "image/webp"
   595→   595→      : filePath.endsWith(".gif") ? "image/gif"
   596→   596→      : filePath.endsWith(".ico") ? "image/x-icon"
   597→   597→      : "application/octet-stream";
   598→   598→    const entry = { data, contentType: ext };
   599→   599→    // Limit cache to 50 files to avoid memory bloat — but NOT for videos (too big)
   600→   600→    if (staticCache.size < 50 && data.length < 5_000_000) staticCache.set(filePath, entry);
   601→   601→    return entry;
   602→   602→  } catch {
   603→   603→    return null;
   604→   604→  }
   605→   605→}
