    21→    21→function isIOS(): boolean {
    22→    22→  if (typeof navigator === "undefined") return false;
    23→    23→  return (
    24→    24→    /iPad|iPhone|iPod/.test(navigator.userAgent) ||
    25→    25→    (navigator.platform === "MacIntel" && (navigator.maxTouchPoints ?? 0) > 1)
    26→    26→  );

    29→    29→function isAndroid(): boolean {

    34→    34→/**
    35→    35→ * 🔴 P2 — Construye la URL de navegación según plataforma y datos disponibles.
    36→    36→ * - Si lat/lng están presentes:
    37→    37→ *   • Android: geo:${lat},${lng}?q=${destination}
    38→    38→ *   • iOS:     maps://?daddr=${lat},${lng}
    39→    39→ *   • Desktop: https://www.google.com/maps/dir/?api=1&destination=${dest}
    40→    40→ * - Si NO hay lat/lng, cae a esquemas address-based:
    41→    41→ *   • Android: geo:0,0?q=${destination}
    42→    42→ *   • iOS:     maps://?daddr=${destination}
    43→    43→ *   • Desktop: https://www.google.com/maps/dir/?api=1&destination=${dest}
    44→    44→ * Devuelve null si no hay destino (ni to ni from ni lat/lng).
    45→    45→ */
    46→    46→function buildNavUrl(opts: {
    47→    47→  destination: string;
    48→    48→  lat?: number;
    49→    49→  lng?: number;
    50→    50→}): string | null {
    51→    51→  const { destination, lat, lng } = opts;
    52→    52→  const hasCoords = typeof lat === "number" && typeof lng === "number";
    53→    53→  if (isAndroid()) {
    54→    54→    if (hasCoords) {
    55→    55→      return `geo:${lat},${lng}?q=${encodeURIComponent(destination)}`;
    56→    56→    }
    57→    57→    return `geo:0,0?q=${encodeURIComponent(destination)}`;
    58→    58→  }
    59→    59→  if (isIOS()) {
    60→    60→    if (hasCoords) {
    61→    61→      return `maps://?daddr=${lat},${lng}`;
    62→    62→    }
    63→    63→    return `maps://?daddr=${encodeURIComponent(destination)}`;
    64→    64→  }
    65→    65→  // Desktop (o fallback)
    66→    66→  return `https://www.google.com/maps/dir/?api=1&destination=${encodeURIComponent(destination)}`;
    67→    67→}

    69→    69→// 🔴 v2: ComparisonBar — barra comparativa para stats de fútbol (o cualquier

    72→    72→function ComparisonBar({ bar }: { bar: NonNullable<DetailRow["bar"]> }) {
    73→    73→  const { homeValue, awayValue, isPercent, homeColor, awayColor } = bar;
    74→    74→  const total = (homeValue ?? 0) + (awayValue ?? 0);
    75→    75→  const homePct = total > 0 ? Math.max(2, Math.min(98, (homeValue / total) * 100)) : 50;
    76→    76→  const awayPct = 100 - homePct;
    77→    77→  const hc = homeColor || "#2d6a4f";
    78→    78→  const ac = awayColor || "#8363f9";

   106→   106→// 🔴 v2: Extrae el video ID de una URL de YouTube (youtu.be / watch?v= / embed / shorts).

   108→   108→function youtubeId(url: string): string | null {
   109→   109→  try {
   110→   110→    const u = new URL(url);
   111→   111→    if (u.hostname.includes("youtu.be")) {
   112→   112→      const id = u.pathname.slice(1).split("/")[0];
   113→   113→      return id || null;
   114→   114→    }
   115→   115→    if (u.hostname.includes("youtube.com")) {
   116→   116→      const v = u.searchParams.get("v");
   117→   117→      if (v) return v;
   118→   118→      const m = u.pathname.match(/\/(?:embed|shorts)\/([\w-]+)/);
   119→   119→      if (m) return m[1];
   120→   120→    }
   121→   121→    return null;
   122→   122→  } catch {
   123→   123→    return null;
   124→   124→  }
   125→   125→}

   439→   439→
   440→   440→  // Parse formation string "4-3-3" → [4, 3, 3]
   441→   441→  function parseFormation(f: string): number[] {
   442→   442→    return f.split("-").map(n => parseInt(n, 10) || 0).filter(n => n > 0);
   443→   443→  }
   444→   444→
   445→   445→  // Posicionar jugadores en una mitad de cancha (50% vertical)
   446→   446→  // home juega "hacia arriba" (de abajo hacia arriba), away "hacia abajo"
   447→   447→  // Cada línea del formation se posiciona a un Y% creciente desde el arquero.
   448→   448→  function positionPlayers(players: Array<{ number?: string; name: string; position?: string }>, formation: string, isHome: boolean): Array<{ x: number; y: number; player: typeof players[0] }> {
   449→   449→    if (players.length === 0) return [];
   450→   450→    const lines = parseFormation(formation);
   451→   451→    const result: Array<{ x: number; y: number; player: typeof players[0] }> = [];
   452→   452→    let playerIdx = 0;
   453→   453→
   454→   454→    // GK siempre primero, en la línea de gol
   455→   455→    if (players[0]) {
   456→   456→      result.push({ x: 50, y: isHome ? 95 : 5, player: players[0] });
   457→   457→      playerIdx = 1;
   458→   458→    }
   459→   459→
   460→   460→    // Cada línea del formation: distribuir N jugadores en una fila horizontal
   461→   461→    // Y va creciendo desde la defensa (cerca del arquero) hasta el ataque (cerca del mediocampo)
   462→   462→    const linesCount = lines.length;
   463→   463→    lines.forEach((count, lineIdx) => {
   464→   464→      // Y para home: de 80% (defensa) a 55% (ataque, mediocampo)
   465→   465→      // Y para away: de 20% (defensa) a 45% (ataque, mediocampo)
   466→   466→      const yBase = isHome ? 80 : 20;
   467→   467→      const yEnd = isHome ? 55 : 45;
   468→   468→      const yRange = Math.abs(yBase - yEnd);
   469→   469→      const yStep = linesCount > 1 ? yRange / (linesCount - 1) : 0;
   470→   470→      const y = isHome ? yBase - lineIdx * yStep : yBase + lineIdx * yStep;
   471→   471→
   472→   472→      // Distribuir los N jugadores horizontalmente
   473→   473→      for (let i = 0; i < count && playerIdx < players.length; i++) {
   474→   474→        const x = ((i + 1) / (count + 1)) * 100;
   475→   475→        result.push({ x, y, player: players[playerIdx] });
   476→   476→        playerIdx++;
   477→   477→      }
   478→   478→    });
   479→   479→
   480→   480→    return result;
   481→   481→  }
   482→   482→
   483→   483→  const homePositions = positionPlayers(homePlayers, homeFormation, true);
   484→   484→  const awayPositions = positionPlayers(awayPlayers, awayFormation, false);
   485→   485→
   486→   486→  return (
   487→   487→    <div className="koru-pitch-container">
   488→   488→      <div className="koru-pitch-header">
   489→   489→        <span style={{ color: hc }}>{awayName} · {awayFormation}</span>
   490→   490→        <span style={{ color: ac, fontSize: 11, opacity: 0.7 }}>vs</span>
   491→   491→        <span style={{ color: hc }}>{homeName} · {homeFormation}</span>
   492→   492→      </div>
   493→   493→      <div className="koru-pitch">
   494→   494→        {/* Líneas de cancha */}
   495→   495→        <div className="koru-pitch-line halfway" />
   496→   496→        <div className="koru-pitch-circle" />
   497→   497→        <div className="koru-pitch-box top" />
   498→   498→        <div className="koru-pitch-box bottom" />
   499→   499→
   500→   500→        {/* Players away (arriba) */}
   501→   501→        {awayPositions.map((pos, i) => (
   502→   502→          <div
   503→   503→            key={`away-${i}`}
   504→   504→            className="koru-pitch-player away"
   505→   505→            style={{
   506→   506→              left: `${pos.x}%`,
   507→   507→              top: `${pos.y}%`,
   508→   508→              background: ac,
   509→   509→            }}
   510→   510→            title={`${pos.player.name}${pos.player.number ? ` #${pos.player.number}` : ""}`}
   511→   511→          >
   512→   512→            <span className="koru-pitch-player-num">{pos.player.number ?? ""}</span>
   513→   513→            <span className="koru-pitch-player-name">{pos.player.name.split(" ").slice(-1)[0]}</span>
   514→   514→          </div>
   515→   515→        ))}
   516→   516→
   517→   517→        {/* Players home (abajo) */}
   518→   518→        {homePositions.map((pos, i) => (
   519→   519→          <div
   520→   520→            key={`home-${i}`}
   521→   521→            className="koru-pitch-player home"
   522→   522→            style={{
   523→   523→              left: `${pos.x}%`,
   524→   524→              top: `${pos.y}%`,
   525→   525→              background: hc,
   526→   526→            }}
   527→   527→            title={`${pos.player.name}${pos.player.number ? ` #${pos.player.number}` : ""}`}
   528→   528→          >
   529→   529→            <span className="koru-pitch-player-num">{pos.player.number ?? ""}</span>
   530→   530→            <span className="koru-pitch-player-name">{pos.player.name.split(" ").slice(-1)[0]}</span>
   531→   531→          </div>
   532→   532→        ))}
   533→   533→      </div>
   534→   534→    </div>
   535→   535→  );
   536→   536→}
