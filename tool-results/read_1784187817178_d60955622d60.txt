   254→export function KoruProvider({ children }: { children: ReactNode }) {
   255→  const [domainState, setDomainState] = useState<KoruState>(() => createInitialState());
   256→  const [onboarded, setOnboarded] = useState(() => localStorage.getItem("koru.onboarded") === "true");
   257→  const [userName, setUserName] = useState(() => localStorage.getItem("koru.username") ?? "");
   258→  const [processing, setProcessing] = useState(false);
   259→  const [activity, setActivity] = useState<AgentActivity | null>(null);
   260→  // 🔴 Memory toast: se setea cuando llegan memoryCandidates nuevos y se limpia con dismissMemoryToast
   261→  const [memoryToast, setMemoryToast] = useState<{ id: string; kind: string; text: string } | null>(null);
   262→  // 🔴 Morning brief state
   263→  const [morningBrief, setMorningBrief] = useState<any | null>(null);
   264→  // 🔴 PWA install prompt
   265→  const [installPromptEvent, setInstallPromptEvent] = useState<any>(null);
   266→  const [showInstallPrompt, setShowInstallPrompt] = useState(false);
   267→  // 🔴 Voice (TTS)
   268→  const [voiceEnabled, setVoiceEnabledState] = useState(() => localStorage.getItem("koru.voiceEnabled") === "true");
   269→  const [phase, setPhase] = useState<string | null>(null);
   270→  const [chatTurns, setChatTurns] = useState<KoruChatTurn[]>(() => readChatTurns(localStorage.getItem("koru.username") ?? ""));
   271→  // Clave versionada (v2): la clave vieja "koru.selected-model" quedaba pegada con
   272→  // modelos de prueba (ej. llama3.1:8b) y pisaba silenciosamente al proveedor
   273→  // principal (BlueSminds) en CADA turno. v2 arranca en Automático para todos;
   274→  // elegir un modelo vuelve a ser una decisión explícita del usuario.
   275→  const [selectedModel, setSelectedModel] = useState<string | null>(() => {
   276→    localStorage.removeItem("koru.selected-model");
   277→    return localStorage.getItem("koru.selected-model.v2") ?? null;
   278→  });
   279→  // 🔴 i18n — preferred language. Persisted in localStorage so the LLM and UI honor it.
   280→  const [language, setLanguageState] = useState<"es" | "en">(() => {
   281→    const stored = localStorage.getItem("koru.language");
   282→    return stored === "en" ? "en" : "es";
   283→  });
   284→  // 🔴 Offline cache — track browser online/offline status
   285→  const [online, setOnline] = useState<boolean>(() => isOnline());
   286→  // Ref to latest sendMessage — lets the offline-queue drain call the current closure.
   287→  const sendMessageRef = useRef<((text: string) => Promise<unknown>) | null>(null);
   288→  // 🔴 v2: record reabiertos desde Collections — renderizan el bloque original
   289→  const [reopenedRecord, setReopenedRecord] = useState<LifeRecord | null>(null);
   290→  const domainStateRef = useRef(domainState);
   291→  const chatTurnsRef = useRef(chatTurns);
   292→
   293→  function commitDomainState(next: KoruState | ((prev: KoruState) => KoruState)) {
   294→    setDomainState((prev) => {
   295→      const resolved = typeof next === "function" ? (next as (prev: KoruState) => KoruState)(prev) : next;
   296→      domainStateRef.current = resolved;
   297→      return resolved;
   298→    });
   299→  }
   300→
   301→  function commitChatTurns(next: KoruChatTurn[] | ((prev: KoruChatTurn[]) => KoruChatTurn[])) {
   302→    setChatTurns((prev) => {
   303→      const resolved = typeof next === "function" ? (next as (prev: KoruChatTurn[]) => KoruChatTurn[])(prev) : next;
   304→      chatTurnsRef.current = resolved;
   305→      return resolved;
   306→    });
   307→  }
   308→
   309→  useEffect(() => {
   310→    domainStateRef.current = domainState;
   311→  }, [domainState]);
   312→
   313→  // 🔴 i18n — sync UI language into domainState so the backend LLM honors it
   314→  useEffect(() => {
   315→    commitDomainState((prev) => (prev.language === language ? prev : { ...prev, language }));
   316→  }, [language]);
   317→
   318→  // 🔴 PWA install prompt listener
   319→  useEffect(() => {
   320→    const handler = (e: Event) => {
   321→      e.preventDefault();
   322→      setInstallPromptEvent(e);
   323→      // Show after 30s if not dismissed
   324→      const dismissed = localStorage.getItem("koru.installDismissed");
   325→      if (!dismissed) {
   326→        setTimeout(() => setShowInstallPrompt(true), 30_000);
   327→      }
   328→    };
   329→    window.addEventListener("beforeinstallprompt", handler);
   330→    return () => window.removeEventListener("beforeinstallprompt", handler);
   331→  }, []);
   332→
   333→  useEffect(() => {
   334→    chatTurnsRef.current = chatTurns;
   335→  }, [chatTurns]);
   336→
   337→  useEffect(() => {
   338→    try {
   339→      const params = new URLSearchParams(window.location.search);
   340→      if (params.get("koruAudit") !== "1" || params.get("reset") !== "1") return;
   341→      const sessionId = `audit_${Date.now().toString(36)}_${Math.random().toString(36).slice(2, 8)}`;
   342→      sessionStorage.setItem("koru.audit.enabled", "1");
   343→      sessionStorage.setItem("koru.audit.session", sessionId);
   344→      localStorage.removeItem(CHAT_STORAGE_KEY);
   345→      localStorage.removeItem("koru.onboarded");
   346→      localStorage.removeItem("koru.username");
   347→      const fresh = resetState();
   348→      commitDomainState(fresh);
   349→      setOnboarded(false);
   350→      setUserName("");
   351→      commitChatTurns([greetingTurn()]);
   352→      params.delete("reset");
   353→      const cleanQuery = params.toString();
   354→      const cleanUrl = `${window.location.pathname}${cleanQuery ? `?${cleanQuery}` : ""}${window.location.hash}`;
   355→      window.history.replaceState(null, "", cleanUrl);
   356→      writeAuditEvent({
   357→        type: "audit_reset",
   358→        note: "Koru browser memory, onboarding and chat reset for manual audit.",
   359→        state: auditStateSnapshot(fresh),
   360→      });
   361→    } catch {
   362→      // Audit reset is best-effort and should not block the app.
   363→    }
   364→  }, []);
   365→
   366→  useEffect(() => {
   367→    let cancelled = false;
   368→    void loadPersistedState().then(async (persisted) => {
   369→      if (!cancelled) {
   370→        const nudges = buildHeartbeatNudges(persisted);
   371→        commitDomainState(nudges.length > 0 ? applyHeartbeatNudges(persisted, nudges) : persisted);
   372→        // 🔴 Sincronizar reminders con notificaciones del navegador
   373→        const userId = persisted.userId ?? "default";
   374→        syncScheduledReminders(persisted.commitments ?? [], userId);
   375→        // Check overdue reminders on app load
   376→        checkDueReminders(userId);
   377→
   378→        // 🔴 Morning Brief: si es mañana y no se mostró hoy, generar
   379→        const now = new Date();
   380→        const hour = now.getHours();
   381→        const today = now.toISOString().slice(0, 10);
   382→        const lastBriefDate = localStorage.getItem("koru.lastBriefDate");
   383→        if (hour >= 5 && hour <= 11 && lastBriefDate !== today) {
   384→          try {
   385→            const stateForBrief = {
   386→              memories: (persisted.memories ?? []).filter((m: any) => m.status === "confirmed" || m.status === "candidate").slice(0, 10).map((m: any) => ({ kind: m.kind, text: m.text })),
   387→              commitments: (persisted.commitments ?? []).filter((c: any) => c.status === "open").slice(0, 5).map((c: any) => ({ title: c.title, dueHint: c.dueHint, dueAt: c.dueAt })),
   388→              userName: persisted.userName ?? "",
   389→              lastBriefDate,
   390→            };
   391→            const res = await fetch("/api/koru/morning-brief", {
   392→              method: "POST",
   393→              headers: { "Content-Type": "application/json" },
   394→              body: JSON.stringify({ state: stateForBrief }),
   395→            });
   396→            if (res.ok) {
   397→              const data = await res.json();
   398→              if (data.shouldShow && data.brief) {
   399→                setMorningBrief(data.brief);
   400→                localStorage.setItem("koru.lastBriefDate", data.date ?? today);
   401→              }
   402→            }
   403→          } catch {
   404→            // silent — brief is optional
   405→          }
   406→        }
   407→      }
   408→    });
   409→    return () => { cancelled = true; };
   410→  }, []);
   411→
   412→  useEffect(() => {
   413→    const runHeartbeat = () => {
   414→      // 🔴 Proactividad: check due reminders y disparar notificaciones
   415→      const userId = domainStateRef.current?.userId ?? "default";
   416→      const { fired, overdue } = checkDueReminders(userId);
   417→      if (fired.length > 0 || overdue.length > 0) {
   418→        console.log("[Koru] Reminders fired:", fired.length, "overdue:", overdue.length);
   419→      }
   420→
   421→      commitDomainState((prev) => {
   422→        const nudges = buildHeartbeatNudges(prev);
   423→        return nudges.length > 0 ? applyHeartbeatNudges(prev, nudges) : prev;
   424→      });
   425→
   426→      // 🔴 MENSAJES PROACTIVOS EN EL CHAT
   427→      // Si hay nudges nuevos, inyectarlos como mensajes de Koru en el chat.
   428→      // Esto hace que Koru "hable" proactivamente, no solo muestre items invisibles.
   429→      const state = domainStateRef.current;
   430→      if (state && !state.ephemeralMode) {
   431→        const activeNudges = (state.nudges ?? []).filter(n =>
   432→          !n.dismissed && !n.title.startsWith("[proactive_shown]")
   433→        );
   434→        if (activeNudges.length > 0) {
   435→          // Solo inyectar 1 mensaje proactivo por heartbeat tick (no spamear)
   436→          const nudge = activeNudges[0];
   437→          const proactiveTurn: KoruChatTurn = {
   438→            id: `proactive_hb_${Date.now()}`,
   439→            role: "koru",
   440→            text: nudge.title + (nudge.body ? `\n${nudge.body}` : ""),
   441→            createdAt: new Date().toISOString(),
   442→            status: "done" as const,
   443→            mascotState: "happy" as const,
   444→          };
   445→          // Marcar como mostrado para no repetir
   446→          commitDomainState((prev) => ({
   447→            ...prev,
   448→            nudges: (prev.nudges ?? []).map(n =>
   449→              n.id === nudge.id ? { ...n, title: `[proactive_shown] ${n.title}` } : n
   450→            ),
   451→          }));
   452→          commitChatTurns((prev) => [...prev, proactiveTurn].slice(-120));
   453→        }

   919→  async function submitEntry(
   920→    text: string,
   921→    transcriptSource: DailyEntry["transcriptSource"] = "typed",
   922→    history: KoruChatTurn[] = chatTurnsRef.current,
   923→  ) {
   924→    setProcessing(true);
   925→    let koruTurnId: string | null = null;
   926→    // Trackea los items creados durante el stream para preservar sus ids al
   927→    // reemplazar con el resultado final. Sin esto, el resultado final crea items
   928→    // con ids nuevos → el WebNavCardA se desmonta/remonta → pierde su estado
   929→    // (animación de progreso, visibleCount) → reinicia a "Iniciando búsqueda...".
   930→    const streamedItemsByType = new Map<string, KoruTurnItem>();
   931→    try {
   932→      const previousState = domainStateRef.current;
   933→      const domainHistory = history
   934→        .filter((turn) => turn.text.trim().length > 0)
   935→        .slice(-12)
   936→        .map((turn) => ({
   937→          role: turn.role === "koru" ? ("assistant" as const) : ("user" as const),
   938→          content: turn.text,
   939→          createdAt: turn.createdAt,
   940→        }));
   941→      const agentResult = await runBackendAgentTurn(text, previousState, domainHistory, selectedModel ?? undefined, (chunk) => {
   942→        // Fase real del pipeline: tomamos el último stateEvent para la píldora/barra.
   943→        if (chunk.stateEvents?.length) {
   944→          const last = chunk.stateEvents[chunk.stateEvents.length - 1];
   945→          setPhase(last.kind === "done" ? null : last.kind);
   946→        }
   947→        const blocksToItems = (blocks: UiBlock[]): KoruTurnItem[] =>
   948→          blocks.map((block) => ({
   949→            id: createId("item"),
   950→            kind: "action" as const,
   951→            tag: block.type,
   952→            text: (block as any).title ?? (block as any).query ?? "",
   953→            status: "working" as const,
   954→            uiBlock: block,
   955→          }));
   956→        if (!koruTurnId) {
   957→          koruTurnId = createId("turn");
   958→          const koruTurn: KoruChatTurn = {
   959→            id: koruTurnId,
   960→            role: "koru",
   961→            text: chunk.reply,
   962→            createdAt: new Date().toISOString(),
   963→            items: blocksToItems(chunk.uiBlocks),
   964→            status: "working" as const,
   965→            mascotState: chunk.mascotState ?? "working",
   966→          };
   967→          (koruTurn.items ?? []).forEach((it) => { if (it.uiBlock?.type) streamedItemsByType.set(it.uiBlock.type, it); });
   968→          commitChatTurns((prev) => [...prev, koruTurn].slice(-120));
   969→        } else {
   970→          const isDone = chunk.stateEvents?.some((e) => e.kind === "done");
   971→          commitChatTurns((prev) =>
   972→            prev.map((turn) => {
   973→              if (turn.id !== koruTurnId) return turn;
   974→              const existingItems = turn.items ?? [];
   975→              const newBlocks = chunk.uiBlocks ?? [];
   976→              // Merge por TIPO de block (no por índice). Así, si el orden cambia
   977→              // entre chunks (ej: aparece un data_card antes que el web_nav),
   978→              // cada item mantiene su identidad y estado (animación de progreso,
   979→              // contadores internos). Antes se matcheaba por idx, lo que hacía
   980→              // que un cambio de orden desmontara el componente y reiniciara la
   981→              // animación ("Iniciando búsqueda..." colgado).
   982→              const consumedExisting = new Set<string>();
   983→              const mergedItems: KoruTurnItem[] = newBlocks.map((block) => {
   984→                const existing = existingItems.find(
   985→                  (it) => it.uiBlock?.type === block.type && !consumedExisting.has(it.uiBlock!.type),
   986→                );
   987→                if (existing) {
   988→                  consumedExisting.add(existing.uiBlock!.type);
   989→                  return { ...existing, uiBlock: block, text: (block as any).title ?? (block as any).query ?? "" };
   990→                }
   991→                return {
   992→                  id: createId("item"),
   993→                  kind: "action" as const,
   994→                  tag: block.type,
   995→                  text: (block as any).title ?? (block as any).query ?? "",
   996→                  status: "working" as const,
   997→                  uiBlock: block,
   998→                };
   999→              });
  1000→              mergedItems.forEach((it) => { if (it.uiBlock?.type) streamedItemsByType.set(it.uiBlock.type, it); });
  1001→              return {
  1002→                ...turn,
  1003→                text: chunk.reply,
  1004→                items: mergedItems,
  1005→                mascotState: chunk.mascotState ?? turn.mascotState,
  1006→                status: isDone ? ("done" as const) : ("working" as const),
  1007→              };
  1008→            }),
  1009→          );
  1010→        }
  1011→      });
  1012→      const result = applyBackendTurnToState(previousState, text, transcriptSource, agentResult);
  1013→      commitDomainState(result.state);
  1014→

  1016→      const newCommitments = (result.state.commitments ?? []).filter(
  1017→        (c) => c.dueAt && !previousState.commitments?.find((pc) => pc.id === c.id)
  1018→      );

  1025→          schedulePreciseTimeout(c.id, c.dueAt!, c.title, userId);

  1029→      }
  1030→      if (koruTurnId) {
  1031→        // Preservar ids de los items del stream: si el resultado final tiene un item
  1032→        // del mismo tipo que uno ya renderizado en el stream, mantener su id (e
  1033→        // identidad de React) para no desmontar/remontar el componente y perder su

  1035→        const consumedTypes = new Set<string>();

  1037→          const type = item.uiBlock?.type ?? item.tag;
  1038→          if (type && !consumedTypes.has(type)) {

  1109→    } finally {
  1110→      setProcessing(false);
  1111→      setActivity(null);

  1297→    void (async () => {
