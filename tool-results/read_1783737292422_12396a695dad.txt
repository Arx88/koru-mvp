    83→function sourceFromRaw(item: Record<string, unknown>): WebSource | null {
    84→  const url = typeof item.url === "string"
    85→    ? item.url
    86→    : typeof item.link === "string"
    87→      ? item.link
    88→      : undefined;
    89→  const title = typeof item.title === "string"
    90→    ? item.title
    91→    : typeof item.name === "string"
    92→      ? item.name
    93→      : undefined;
    94→  if (!url || !title) return null;
    95→  return {
    96→    title,
    97→    url,
    98→    domain: domainFromUrl(url),
    99→    snippet: typeof item.snippet === "string"
   100→      ? item.snippet
   101→      : typeof item.description === "string"
   102→        ? item.description
   103→        : typeof item.content === "string"
   104→          ? item.content
   105→          : undefined,
   106→  };
   107→}

   164→async function queryOpenMeteo(queries: string[]): Promise<WebSource[]> {
   165→  const location = parseOpenMeteoLocation(queries.join(" "));
   166→  if (!location) return [];
   167→  const url = new URL("https://api.open-meteo.com/v1/forecast");
   168→  url.searchParams.set("latitude", String(location.latitude));
   169→  url.searchParams.set("longitude", String(location.longitude));
   170→  url.searchParams.set("current", "temperature_2m,precipitation,wind_speed_10m");
   171→  url.searchParams.set("daily", "temperature_2m_max,temperature_2m_min,precipitation_probability_max");
   172→  url.searchParams.set("timezone", "auto");
   173→  const response = await fetch(url);
   174→  if (!response.ok) return [];
   175→  const data = await response.json() as {
   176→    current?: { temperature_2m?: number; precipitation?: number; wind_speed_10m?: number };
   177→    daily?: { temperature_2m_max?: number[]; temperature_2m_min?: number[]; precipitation_probability_max?: number[] };
   178→  };
   179→  const current = data.current;
   180→  if (!current) return [];
   181→  const max = data.daily?.temperature_2m_max?.[0];
   182→  const min = data.daily?.temperature_2m_min?.[0];
   183→  const rain = data.daily?.precipitation_probability_max?.[0];
   184→  return [{
   185→    title: `Clima actual en ${location.label}`,
   186→    url: "https://open-meteo.com/",
   187→    domain: "open-meteo.com",
   188→    snippet: [
   189→      `${current.temperature_2m ?? "?"} C ahora`,
   190→      max !== undefined && min !== undefined ? `${min}-${max} C hoy` : undefined,
   191→      rain !== undefined ? `${rain}% probabilidad de lluvia` : undefined,
   192→      current.wind_speed_10m !== undefined ? `viento ${current.wind_speed_10m} km/h` : undefined,
   193→    ].filter(Boolean).join(" · "),
   194→  }];
   195→}

   423→async function enrichSourcesWithBrowserPages(page: {
   424→  goto: (url: string, options: Record<string, unknown>) => Promise<unknown>;
   425→  evaluate: <T, A>(pageFunction: (arg: A) => T, arg: A) => Promise<T>;
   426→}, sources: WebSource[], mode: WebSearchMode): Promise<WebSource[]> {
   427→  const enriched: WebSource[] = [];
   428→  for (const source of sources.slice(0, 5)) {
   429→    try {
   430→      await page.goto(source.url, { waitUntil: "domcontentloaded", timeout: 12_000 });
   431→      const detail = await page.evaluate((currentMode) => {
   432→        const meta =
   433→          document.querySelector<HTMLMetaElement>('meta[name="description"]')?.content ||
   434→          document.querySelector<HTMLMetaElement>('meta[property="og:description"]')?.content ||
   435→          document.querySelector<HTMLMetaElement>('meta[name="twitter:description"]')?.content ||
   436→          "";
   437→        const title =
   438→          document.querySelector<HTMLMetaElement>('meta[property="og:title"]')?.content ||
   439→          document.title ||
   440→          "";
   441→        const body = document.body?.innerText?.replace(/\s+/g, " ").slice(0, 4_000) ?? "";
   442→        const price = currentMode === "shopping"
   443→          ? body.match(/(?:€|\$)\s?\d[\d.,]*|\b\d[\d.,]*\s?(?:EUR|USD|ARS|€|\$)\b/i)?.[0]
   444→          : undefined;
   445→        return {
   446→          title: title.trim(),
   447→          snippet: [price, meta || body.slice(0, 220)].filter(Boolean).join(" · "),
   448→        };
   449→      }, mode);
   450→      enriched.push({
   451→        ...source,
   452→        title: detail.title || source.title,
   453→        snippet: detail.snippet || source.snippet,
   454→      });
   455→    } catch {
   456→      enriched.push(source);
   457→    }
   458→  }
   459→  return enriched;
   460→}

   646→function koruAiProxy(env: Record<string, string>): Plugin {
   647→  const openRouterKeys = collectOpenRouterKeys(env);
   648→  const openRouterModels = collectOpenRouterModels(env);
   649→  const nvidiaKey = env.NVIDIA_API_KEY?.trim();
   650→  const nvidiaBaseUrl = env.NVIDIA_BASE_URL?.trim() || "https://integrate.api.nvidia.com";
   651→  const nvidiaModel = env.NVIDIA_MODEL?.trim() || "minimaxai/minimax-m3";
   652→
   653→  return {
   654→    name: "koru-ai-proxy",
   655→    configureServer(server) {
   656→      server.middlewares.use("/koru-ai", async (req, res) => {
   657→        if (!nvidiaKey && !openRouterKeys.length) {
   658→          res.statusCode = 500;
   659→          res.setHeader("Content-Type", "application/json");
   660→          res.end(JSON.stringify({ error: { message: "No hay proveedor de IA configurado." } }));
   661→          return;
   662→        }
   663→
   664→        if (req.method !== "POST") {
   665→          res.statusCode = 405;
   666→          res.setHeader("Allow", "POST");
   667→          res.end();
   668→          return;
   669→        }
   670→
   671→        const chunks: Buffer[] = [];
   672→        for await (const chunk of req) chunks.push(Buffer.from(chunk));
   673→        const body = Buffer.concat(chunks);
   674→        const targetPath = req.url?.replace(/^\/?/, "") ?? "";
   675→        let lastStatus = 502;
   676→        let lastBody = JSON.stringify({ error: { message: "Ningun proveedor respondio con contenido usable." } });
   677→        let lastContentType = "application/json";
   678→
   679→        if (nvidiaKey) {
   680→          const controller = new AbortController();
   681→          const timeout = setTimeout(() => controller.abort(), 55_000);
   682→          try {
   683→            const response = await fetch(providerUrl(nvidiaBaseUrl, targetPath), {
   684→              method: "POST",
   685→              headers: {
   686→                Authorization: `Bearer ${nvidiaKey}`,
   687→                "Content-Type": req.headers["content-type"] ?? "application/json",
   688→                Accept: "application/json",
   689→              },
   690→              body: prepareJsonBody(body, { model: nvidiaModel, top_p: 0.95 }, true),
   691→              signal: controller.signal,
   692→            });
   693→            const responseBody = await response.text();
   694→            lastStatus = response.status;
   695→            lastBody = responseBody;
   696→            lastContentType = response.headers.get("content-type") ?? "application/json";
   697→
   698→            if (response.ok && bodyHasAssistantContent(responseBody)) {
   699→              res.statusCode = response.status;
   700→              res.setHeader("Content-Type", lastContentType);
   701→              res.end(responseBody);
   702→              return;
   703→            }

   707→              error: {
   708→                message: error instanceof Error ? error.message : "NVIDIA no respondio a tiempo.",
   709→                provider: "nvidia",
   710→                model: nvidiaModel,
   711→              },
   712→            });
   713→            lastContentType = "application/json";
   714→          } finally {
   715→            clearTimeout(timeout);
   716→          }
   717→        }
   718→
   719→        for (const model of openRouterModels) {
   720→          for (const key of openRouterKeys) {
   721→            const response = await fetch(providerUrl("https://openrouter.ai/api", targetPath), {
   722→              method: "POST",
   723→              headers: {
   724→                Authorization: `Bearer ${key}`,
   725→                "Content-Type": req.headers["content-type"] ?? "application/json",
   726→                "HTTP-Referer": "http://localhost:5173",
   727→                "X-OpenRouter-Title": "Koru MVP",
   728→              },
   729→              body: prepareJsonBody(body, {
   730→                model,
   731→                response_format: { type: "json_object" },
   732→                plugins: [{ id: "response-healing" }],
   733→                reasoning: { exclude: true },
   734→              }),
   735→            });
   736→            const responseBody = await response.text();
   737→            lastStatus = response.status;
   738→            lastBody = responseBody;
   739→            lastContentType = response.headers.get("content-type") ?? "application/json";
   740→
   741→            if (response.ok && bodyHasAssistantContent(responseBody)) {
   742→              res.statusCode = response.status;
   743→              res.setHeader("Content-Type", lastContentType);
   744→              res.end(responseBody);
   745→              return;
   746→            }
   747→          }
   748→        }
   749→
   750→        res.statusCode = lastStatus;
   751→        res.setHeader("Content-Type", lastContentType);
   752→        res.end(lastBody);
   753→      });
   754→    },
   755→  };
   756→}
