   262→export function KoruProvider({ children }: { children: ReactNode }) {
   263→  const [domainState, setDomainState] = useState<KoruState>(() => createInitialState());
   264→  const [onboarded, setOnboarded] = useState(() => localStorage.getItem("koru.onboarded") === "true");
   265→  const [userName, setUserName] = useState(() => localStorage.getItem("koru.username") ?? "");
   266→  const [processing, setProcessing] = useState(false);
   267→  const [activity, setActivity] = useState<AgentActivity | null>(null);
   268→  // 🔴 Memory toast: se setea cuando llegan memoryCandidates nuevos y se limpia con dismissMemoryToast
   269→  const [memoryToast, setMemoryToast] = useState<{ id: string; kind: string; text: string } | null>(null);
   270→  // 🔴 Morning brief state
   271→  const [morningBrief, setMorningBrief] = useState<any | null>(null);
   272→  // 🔴 PWA install prompt
   273→  const [installPromptEvent, setInstallPromptEvent] = useState<any>(null);
   274→  const [showInstallPrompt, setShowInstallPrompt] = useState(false);
   275→  // 🔴 Voice (TTS)
   276→  const [voiceEnabled, setVoiceEnabledState] = useState(() => localStorage.getItem("koru.voiceEnabled") === "true");
   277→  const [phase, setPhase] = useState<string | null>(null);
   278→  const [chatTurns, setChatTurns] = useState<KoruChatTurn[]>(() => readChatTurns(localStorage.getItem("koru.username") ?? ""));
   279→  // Clave versionada (v2): la clave vieja "koru.selected-model" quedaba pegada con
   280→  // modelos de prueba (ej. llama3.1:8b) y pisaba silenciosamente al proveedor
   281→  // principal (BlueSminds) en CADA turno. v2 arranca en Automático para todos;
   282→  // elegir un modelo vuelve a ser una decisión explícita del usuario.
   283→  const [selectedModel, setSelectedModel] = useState<string | null>(() => {
   284→    localStorage.removeItem("koru.selected-model");
   285→    return localStorage.getItem("koru.selected-model.v2") ?? null;
   286→  });
   287→  // 🔴 i18n — preferred language. Persisted in localStorage so the LLM and UI honor it.
   288→  const [language, setLanguageState] = useState<"es" | "en">(() => {
   289→    const stored = localStorage.getItem("koru.language");
   290→    return stored === "en" ? "en" : "es";
   291→  });
   292→  // 🔴 Offline cache — track browser online/offline status
   293→  const [online, setOnline] = useState<boolean>(() => isOnline());
   294→  // Ref to latest sendMessage — lets the offline-queue drain call the current closure.
   295→  const sendMessageRef = useRef<((text: string) => Promise<unknown>) | null>(null);
   296→  // 🔴 v2: record reabiertos desde Collections — renderizan el bloque original
   297→  const [reopenedRecord, setReopenedRecord] = useState<LifeRecord | null>(null);
   298→  const domainStateRef = useRef(domainState);
   299→  const chatTurnsRef = useRef(chatTurns);
   300→
   301→  function commitDomainState(next: KoruState | ((prev: KoruState) => KoruState)) {
   302→    setDomainState((prev) => {
   303→      const resolved = typeof next === "function" ? (next as (prev: KoruState) => KoruState)(prev) : next;
   304→      domainStateRef.current = resolved;
   305→      return resolved;
   306→    });
   307→  }
   308→
   309→  function commitChatTurns(next: KoruChatTurn[] | ((prev: KoruChatTurn[]) => KoruChatTurn[])) {
   310→    setChatTurns((prev) => {
   311→      const resolved = typeof next === "function" ? (next as (prev: KoruChatTurn[]) => KoruChatTurn[])(prev) : next;
   312→      chatTurnsRef.current = resolved;
   313→      return resolved;
   314→    });
   315→  }
   316→
   317→  useEffect(() => {
   318→    domainStateRef.current = domainState;
   319→  }, [domainState]);
   320→
   321→  // 🔴 i18n — sync UI language into domainState so the backend LLM honors it
   322→  useEffect(() => {
   323→    commitDomainState((prev) => (prev.language === language ? prev : { ...prev, language }));
   324→  }, [language]);
   325→
   326→  // 🔴 PWA install prompt listener
   327→  useEffect(() => {
   328→    const handler = (e: Event) => {
   329→      e.preventDefault();
   330→      setInstallPromptEvent(e);
   331→      // Show after 30s if not dismissed
   332→      const dismissed = localStorage.getItem("koru.installDismissed");
   333→      if (!dismissed) {
   334→        setTimeout(() => setShowInstallPrompt(true), 30_000);
   335→      }
   336→    };
   337→    window.addEventListener("beforeinstallprompt", handler);
   338→    return () => window.removeEventListener("beforeinstallprompt", handler);
   339→  }, []);
   340→
   341→  useEffect(() => {
   342→    chatTurnsRef.current = chatTurns;
   343→  }, [chatTurns]);
   344→
   345→  useEffect(() => {
   346→    try {
   347→      const params = new URLSearchParams(window.location.search);
   348→      if (params.get("koruAudit") !== "1" || params.get("reset") !== "1") return;
   349→      const sessionId = `audit_${Date.now().toString(36)}_${Math.random().toString(36).slice(2, 8)}`;
   350→      sessionStorage.setItem("koru.audit.enabled", "1");
   351→      sessionStorage.setItem("koru.audit.session", sessionId);
   352→      localStorage.removeItem(CHAT_STORAGE_KEY);
   353→      localStorage.removeItem("koru.onboarded");
   354→      localStorage.removeItem("koru.username");
   355→      const fresh = resetState();
   356→      commitDomainState(fresh);
   357→      setOnboarded(false);
   358→      setUserName("");
   359→      commitChatTurns([greetingTurn()]);
   360→      params.delete("reset");
   361→      const cleanQuery = params.toString();
   362→      const cleanUrl = `${window.location.pathname}${cleanQuery ? `?${cleanQuery}` : ""}${window.location.hash}`;
   363→      window.history.replaceState(null, "", cleanUrl);
   364→      writeAuditEvent({
   365→        type: "audit_reset",
   366→        note: "Koru browser memory, onboarding and chat reset for manual audit.",
   367→        state: auditStateSnapshot(fresh),
   368→      });
   369→    } catch {
   370→      // Audit reset is best-effort and should not block the app.
   371→    }
   372→  }, []);
   373→
   374→  useEffect(() => {
   375→    let cancelled = false;
   376→    void loadPersistedState().then(async (persisted) => {
   377→      if (!cancelled) {
   378→        const nudges = buildHeartbeatNudges(persisted);
   379→        commitDomainState(nudges.length > 0 ? applyHeartbeatNudges(persisted, nudges) : persisted);
   380→        // 🔴 Sincronizar reminders con notificaciones del navegador
   381→        const userId = persisted.userId ?? "default";
   382→        syncScheduledReminders(persisted.commitments ?? [], userId);
   383→        // Check overdue reminders on app load
   384→        checkDueReminders(userId);
   385→
   386→        // 🔴 Morning Brief: si es mañana y no se mostró hoy, generar
   387→        const now = new Date();
   388→        const hour = now.getHours();
   389→        const today = now.toISOString().slice(0, 10);
   390→        const lastBriefDate = localStorage.getItem("koru.lastBriefDate");
   391→        if (hour >= 5 && hour <= 11 && lastBriefDate !== today) {
   392→          try {
   393→            const stateForBrief = {
   394→              memories: (persisted.memories ?? []).filter((m: any) => m.status === "confirmed" || m.status === "candidate").slice(0, 10).map((m: any) => ({ kind: m.kind, text: m.text })),
   395→              commitments: (persisted.commitments ?? []).filter((c: any) => c.status === "open").slice(0, 5).map((c: any) => ({ title: c.title, dueHint: c.dueHint, dueAt: c.dueAt })),
   396→              userName: persisted.userName ?? "",
   397→              lastBriefDate,
   398→            };
   399→            const res = await fetch("/api/koru/morning-brief", {
   400→              method: "POST",
   401→              headers: { "Content-Type": "application/json" },
   402→              body: JSON.stringify({ state: stateForBrief }),
   403→            });
   404→            if (res.ok) {
   405→              const data = await res.json();
   406→              if (data.shouldShow && data.brief) {
   407→                setMorningBrief(data.brief);
   408→                localStorage.setItem("koru.lastBriefDate", data.date ?? today);
   409→                // 🔴 TIER S: persistir el brief en el store (state.lastBriefDate
   410→                // + state.lastBriefBlock) vía el reducer setLastBrief. Así
   411→                // otras pantallas pueden inspeccionar si ya se mostró hoy sin
   412→                // leer localStorage directamente.
   413→                const briefBlock: UiBlock | undefined = data.brief?.block ?? data.brief;
   414→                commitDomainState((prev) => setLastBriefReducer(prev, data.date ?? today, briefBlock));
   415→              }
   416→            }
   417→          } catch {
   418→            // silent — brief is optional
   419→          }
   420→        }
   421→      }
   422→    });
   423→    return () => { cancelled = true; };
   424→  }, []);
   425→
   426→  useEffect(() => {
   427→    const runHeartbeat = () => {
   428→      // 🔴 Proactividad: check due reminders y disparar notificaciones
   429→      const userId = domainStateRef.current?.userId ?? "default";
   430→      const { fired, overdue } = checkDueReminders(userId);
   431→      if (fired.length > 0 || overdue.length > 0) {
   432→        console.log("[Koru] Reminders fired:", fired.length, "overdue:", overdue.length);
   433→      }
   434→
   435→      commitDomainState((prev) => {
   436→        const nudges = buildHeartbeatNudges(prev);
   437→        return nudges.length > 0 ? applyHeartbeatNudges(prev, nudges) : prev;
   438→      });
   439→
   440→      // 🔴 MENSAJES PROACTIVOS EN EL CHAT
   441→      // Si hay nudges nuevos, inyectarlos como mensajes de Koru en el chat.
   442→      // Esto hace que Koru "hable" proactivamente, no solo muestre items invisibles.
   443→      const state = domainStateRef.current;
   444→      if (state && !state.ephemeralMode) {
   445→        const activeNudges = (state.nudges ?? []).filter(n =>
   446→          !n.dismissed && !n.title.startsWith("[proactive_shown]")
   447→        );
   448→        if (activeNudges.length > 0) {
   449→          // Solo inyectar 1 mensaje proactivo por heartbeat tick (no spamear)

   960→    // reemplazar con el resultado final. Sin esto, el resultado final crea items

   967→        .filter((turn) => turn.text.trim().length > 0)

   974→      const agentResult = await runBackendAgentTurn(text, previousState, domainHistory, selectedModel ?? undefined, (chunk) => {

   977→          const last = chunk.stateEvents[chunk.stateEvents.length - 1];

  1317→  function runReadonlyWebAction(original: AssistantAction): KoruTurnItem {

  1319→    // informe/investigacion real (webMode "research", o web_research sin

  1396→        });
  1397→        replaceActionInChat(patchedItem);
  1398→      } finally {
  1399→        setActivity(null);
  1400→      }
  1401→    })();
  1402→    return waitingItem;
  1403→  }

  1906→            state = { ...created, checklists: patchedChecklists };
  1907→            saveState(state);
  1908→            commitDomainState(state);
  1909→          }
  1910→          commitDomainState(toggleChecklistItemReducer(state, checklistId, itemId));
  1911→        }
  1912→      } else if (action === "archive_plan") {
  1913→        // 🔴 TIER S: archivar un Plan desde un botón inline.
  1914→        const planId = String(detail?.planId ?? "");
  1915→        if (planId) {
  1916→          commitDomainState((prev) => archivePlanReducer(prev, planId));
  1917→          setMemoryToast({ id: `action_${Date.now()}`, kind: "saved", text: "Plan archivado ✓" });
  1918→          setTimeout(() => setMemoryToast(null), 2000);
  1919→        }
  1920→      } else if (action === "log_workout") {
  1921→        // 🔴 TIER S: marcar una sesión de entrenamiento como completada.
  1922→        // detail trae planId + sessionId + durationMin + kcal (opcional).
  1923→        const planId = String(detail?.planId ?? "");
  1924→        const sessionId = String(detail?.sessionId ?? "");
  1925→        const durationMin = Number(detail?.durationMin ?? 30) || 30;
  1926→        const kcal = detail?.kcal != null ? Number(detail.kcal) : undefined;
  1927→        if (planId && sessionId) {
  1928→          commitDomainState((prev) =>
  1929→            logWorkoutReducer(prev, planId, sessionId, [], durationMin, kcal)
  1930→          );
  1931→          setMemoryToast({ id: `action_${Date.now()}`, kind: "saved", text: "Entrenamiento registrado ✓" });
  1932→          setTimeout(() => setMemoryToast(null), 2000);
  1933→        }
  1934→      } else if (action === "reserve") {
  1935→        // 🔴 v3: abrir reserveUrl del top match del restaurant_synthesis block.
  1936→        // El blockData es el UiBlock completo; matches[].reserveUrl viene de Google Places.
  1937→        const matches = blockData?.matches as Array<{ reserveUrl?: string }> | undefined;
  1938→        const reserveUrl = matches?.find((m) => typeof m?.reserveUrl === "string")?.reserveUrl;
  1939→        if (reserveUrl) {
  1940→          try {
  1941→            window.open(reserveUrl, "_blank", "noopener,noreferrer");
  1942→            setMemoryToast({ id: `action_${Date.now()}`, kind: "saved", text: "Abriendo reserva…" });
  1943→            setTimeout(() => setMemoryToast(null), 2000);
  1944→          } catch {
  1945→            setMemoryToast({ id: `action_${Date.now()}`, kind: "saved", text: "No se pudo abrir el enlace" });
  1946→            setTimeout(() => setMemoryToast(null), 2500);
  1947→          }
  1948→        }
  1949→      }
  1950→    };
  1951→    window.addEventListener("koru-card-action", onCardAction as EventListener);
  1952→    return () => window.removeEventListener("koru-card-action", onCardAction as EventListener);
  1953→  }, []);
  1954→
  1955→  return <KoruContext.Provider value={value}>{children}</KoruContext.Provider>;
  1956→}
  1957→
  1958→export function useKoru() {
  1959→  const ctx = useContext(KoruContext);
  1960→  if (!ctx) throw new Error("useKoru must be used within KoruProvider");
  1961→  return ctx;
  1962→}
