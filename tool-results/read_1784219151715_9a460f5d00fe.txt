    32→// Load .env manually (no dotenv dependency)
    33→function loadEnv() {
    34→  const envPath = join(PROJECT_ROOT, ".env");
    35→  if (!existsSync(envPath)) return;
    36→  const content = readFileSync(envPath, "utf8");
    37→  for (const line of content.split("\n")) {
    38→    const trimmed = line.trim();
    39→    if (!trimmed || trimmed.startsWith("#")) continue;
    40→    const eqIdx = trimmed.indexOf("=");
    41→    if (eqIdx === -1) continue;
    42→    const key = trimmed.slice(0, eqIdx).trim();
    43→    const value = trimmed.slice(eqIdx + 1).trim();
    44→    if (!process.env[key]) process.env[key] = value;
    45→  }
    46→}

   873→// ── Cache static files in memory (avoid disk I/O on every request) ──
   874→const staticCache = new Map<string, { data: Buffer; contentType: string }>();
   875→const STATIC_DIR = join(PROJECT_ROOT, "dist");
   876→
   877→function getStaticFile(filePath: string): { data: Buffer; contentType: string } | null {
   878→  if (staticCache.has(filePath)) return staticCache.get(filePath)!;
   879→  try {
   880→    const data = readFileSync(filePath);
   881→    const ext = filePath.endsWith(".html") ? "text/html"
   882→      : filePath.endsWith(".js") ? "application/javascript"
   883→      : filePath.endsWith(".css") ? "text/css"
   884→      : filePath.endsWith(".png") ? "image/png"
   885→      : filePath.endsWith(".jpg") || filePath.endsWith(".jpeg") ? "image/jpeg"
   886→      : filePath.endsWith(".svg") ? "image/svg+xml"
   887→      : filePath.endsWith(".json") ? "application/json"
   888→      : filePath.endsWith(".woff") ? "font/woff"
   889→      : filePath.endsWith(".woff2") ? "font/woff2"
   890→      : filePath.endsWith(".mp4") ? "video/mp4"
   891→      : filePath.endsWith(".webm") ? "video/webm"
   892→      : filePath.endsWith(".webp") ? "image/webp"
   893→      : filePath.endsWith(".gif") ? "image/gif"
   894→      : filePath.endsWith(".ico") ? "image/x-icon"
   895→      : "application/octet-stream";
   896→    const entry = { data, contentType: ext };
   897→    // Limit cache to 50 files to avoid memory bloat — but NOT for videos (too big)
   898→    if (staticCache.size < 50 && data.length < 5_000_000) staticCache.set(filePath, entry);
   899→    return entry;
   900→  } catch {
   901→    return null;
   902→  }
