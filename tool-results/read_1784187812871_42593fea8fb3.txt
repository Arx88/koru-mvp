   795вҶ’   795вҶ’  const records = b.records ?? [];
   796вҶ’   796вҶ’  const first = records[0];
   797вҶ’   797вҶ’  // La colecciГіn donde quedГі lo guardado: es el corazГіn de la promesa
   798вҶ’   798вҶ’  // "Listo, guardado en Sitios de IA" вҶ’ CTA "Ver colecciГіn" abre Mis Colecciones.
   799вҶ’   799вҶ’  // Solo usar collection si first.collection existe; NO usar b.title como fallback
   800вҶ’   800вҶ’  // porque b.title suele ser "Guardado" (genГ©rico) y produce "Guardado en Guardado".
   801вҶ’   801вҶ’  const collection = clean(first?.collection);
   802вҶ’   802вҶ’  const isLink = Boolean(first?.url);
   803вҶ’   803вҶ’  // desc: construir de forma robusta sin duplicados (case-insensitive + accent-insensitive).
   804вҶ’   804вҶ’  // Para "CafГ©" + "cafe" + "2000 ARS": solo mostrar "CafГ© В· 2000 ARS".
   805вҶ’   805вҶ’  let desc: string | undefined;
   806вҶ’   806вҶ’  if (records.length > 1) {
   807вҶ’   807вҶ’    desc = `${records.length} registros guardados`;
   808вҶ’   808вҶ’  } else if (first) {
   809вҶ’   809вҶ’    const parts: string[] = [];
   810вҶ’   810вҶ’    const seen = new Set<string>();
   811вҶ’   811вҶ’    const title = clean(first.title);
   812вҶ’   812вҶ’    const notes = clean(first.notes);
   813вҶ’   813вҶ’    const value = clean(first.value);
   814вҶ’   814вҶ’    // Normaliza: lowercase + sin acentos para comparar
   815вҶ’   815вҶ’    const norm = (s: string) => s.toLowerCase().normalize("NFD").replace(/[\u0300-\u036f]/g, "");
   816вҶ’   816вҶ’    const tryAdd = (s: string | undefined) => {
   817вҶ’   817вҶ’      if (!s) return;
   818вҶ’   818вҶ’      const key = norm(s);
   819вҶ’   819вҶ’      if (seen.has(key)) return;
   820вҶ’   820вҶ’      if (key === norm(collection ?? "")) return;
   821вҶ’   821вҶ’      seen.add(key);
   822вҶ’   822вҶ’      parts.push(s);
   823вҶ’   823вҶ’    };
   824вҶ’   824вҶ’    tryAdd(title);
   825вҶ’   825вҶ’    tryAdd(notes);
   826вҶ’   826вҶ’    tryAdd(value);
   827вҶ’   827вҶ’    desc = parts.length > 0 ? parts.join(" В· ") : undefined;
   828вҶ’   828вҶ’  }
   829вҶ’   829вҶ’  // Kicker: solo "Guardado en <colecciГіn>" si hay collection real; si no, "Guardado"
   830вҶ’   830вҶ’  const kicker = collection ? `Guardado en ${collection}` : "Guardado";
   831вҶ’   831вҶ’  // Title: si hay collection, mostrarlo. Si no, mostrar el title del primer record.
   832вҶ’   832вҶ’  const heroTitle = collection || clean(first?.title) || "Registro";
   833вҶ’   833вҶ’  return {
   834вҶ’   834вҶ’    hero: {
   835вҶ’   835вҶ’      kicker,
