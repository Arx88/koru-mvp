   275→  const [showInstallPrompt, setShowInstallPrompt] = useState(false);

   277→  const [voiceEnabled, setVoiceEnabledState] = useState(() => localStorage.getItem("koru.voiceEnabled") === "true");
   278→  const [phase, setPhase] = useState<string | null>(null);
   279→  const [chatTurns, setChatTurns] = useState<KoruChatTurn[]>(() => readChatTurns(localStorage.getItem("koru.username") ?? ""));
   280→  // Clave versionada (v2): la clave vieja "koru.selected-model" quedaba pegada con
   281→  // modelos de prueba (ej. llama3.1:8b) y pisaba silenciosamente al proveedor

   283→  // elegir un modelo vuelve a ser una decisión explícita del usuario.
   284→  const [selectedModel, setSelectedModel] = useState<string | null>(() => {
   285→    localStorage.removeItem("koru.selected-model");
   286→    return localStorage.getItem("koru.selected-model.v2") ?? null;
   287→  });
   288→  // 🔴 i18n — preferred language. Persisted in localStorage so the LLM and UI honor it.
   289→  const [language, setLanguageState] = useState<"es" | "en">(() => {
   290→    const stored = localStorage.getItem("koru.language");
   291→    return stored === "en" ? "en" : "es";
   292→  });
   293→  // 🔴 Offline cache — track browser online/offline status
   294→  const [online, setOnline] = useState<boolean>(() => isOnline());
   295→  // Ref to latest sendMessage — lets the offline-queue drain call the current closure.
   296→  const sendMessageRef = useRef<((text: string) => Promise<unknown>) | null>(null);
   297→  // 🔴 v2: record reabiertos desde Collections — renderizan el bloque original
   298→  const [reopenedRecord, setReopenedRecord] = useState<LifeRecord | null>(null);
   299→  const domainStateRef = useRef(domainState);
   300→  const chatTurnsRef = useRef(chatTurns);
   301→  // 🔴 Morning brief scheduler — deduplicación: si ya dispatcheamos el nudge

   303→  // state.lastBriefDate justo después de que ya lo seteamos), no repetir.
   304→  const lastBriefNudgeDateRef = useRef<string | null>(null);
   305→
   306→  function commitDomainState(next: KoruState | ((prev: KoruState) => KoruState)) {
   307→    setDomainState((prev) => {
   308→      const resolved = typeof next === "function" ? (next as (prev: KoruState) => KoruState)(prev) : next;
   309→      domainStateRef.current = resolved;
   310→      return resolved;
   311→    });
   312→  }
   313→
   314→  function commitChatTurns(next: KoruChatTurn[] | ((prev: KoruChatTurn[]) => KoruChatTurn[])) {
   315→    setChatTurns((prev) => {
   316→      const resolved = typeof next === "function" ? (next as (prev: KoruChatTurn[]) => KoruChatTurn[])(prev) : next;
   317→      chatTurnsRef.current = resolved;
   318→      return resolved;
   319→    });
   320→  }
   321→
   322→  useEffect(() => {
   323→    domainStateRef.current = domainState;
   324→  }, [domainState]);
   325→
   326→  // 🔴 i18n — sync UI language into domainState so the backend LLM honors it
   327→  useEffect(() => {
   328→    commitDomainState((prev) => (prev.language === language ? prev : { ...prev, language }));
   329→  }, [language]);
   330→
   331→  // 🔴 PWA install prompt listener
   332→  useEffect(() => {
   333→    const handler = (e: Event) => {
   334→      e.preventDefault();

   337→      const dismissed = localStorage.getItem("koru.installDismissed");

   339→        setTimeout(() => setShowInstallPrompt(true), 30_000);

   341→    };
   342→    window.addEventListener("beforeinstallprompt", handler);
   343→    return () => window.removeEventListener("beforeinstallprompt", handler);
   344→  }, []);
   345→
   346→  useEffect(() => {
   347→    chatTurnsRef.current = chatTurns;
   348→  }, [chatTurns]);
   349→
   350→  useEffect(() => {
   351→    try {
   352→      const params = new URLSearchParams(window.location.search);
   353→      if (params.get("koruAudit") !== "1" || params.get("reset") !== "1") return;
   354→      const sessionId = `audit_${Date.now().toString(36)}_${Math.random().toString(36).slice(2, 8)}`;
   355→      sessionStorage.setItem("koru.audit.enabled", "1");
   356→      sessionStorage.setItem("koru.audit.session", sessionId);
   357→      localStorage.removeItem(CHAT_STORAGE_KEY);
   358→      localStorage.removeItem("koru.onboarded");
   359→      localStorage.removeItem("koru.username");
   360→      const fresh = resetState();
   361→      commitDomainState(fresh);
   362→      setOnboarded(false);
   363→      setUserName("");
   364→      commitChatTurns([greetingTurn()]);
   365→      params.delete("reset");
   366→      const cleanQuery = params.toString();
   367→      const cleanUrl = `${window.location.pathname}${cleanQuery ? `?${cleanQuery}` : ""}${window.location.hash}`;
   368→      window.history.replaceState(null, "", cleanUrl);
   369→      writeAuditEvent({
   370→        type: "audit_reset",
   371→        note: "Koru browser memory, onboarding and chat reset for manual audit.",
   372→        state: auditStateSnapshot(fresh),
   373→      });
   374→    } catch {
   375→      // Audit reset is best-effort and should not block the app.
   376→    }
   377→  }, []);
   378→
   379→  useEffect(() => {
   380→    let cancelled = false;
   381→    void loadPersistedState().then(async (persisted) => {
   382→      if (!cancelled) {
   383→        const nudges = buildHeartbeatNudges(persisted);
   384→        commitDomainState(nudges.length > 0 ? applyHeartbeatNudges(persisted, nudges) : persisted);
   385→        // 🔴 Sincronizar reminders con notificaciones del navegador
   386→        const userId = persisted.userId ?? "default";
   387→        syncScheduledReminders(persisted.commitments ?? [], userId);
   388→        // Check overdue reminders on app load
   389→        checkDueReminders(userId);
   390→
   391→        // 🔴 Morning Brief: si es mañana y no se mostró hoy, generar.
   392→        // 🔴 TIER S: check usa `state.lastBriefDate` (vía persisted.lastBriefDate)
   393→        // como fuente de verdad, con fallback a localStorage para sesiones que
   394→        // todavía no tienen el campo en el store. Esto cumple el contrato del
   395→        // reducer setLastBrief: otras pantallas pueden inspeccionar si ya se
   396→        // mostró hoy leyendo state.lastBriefDate sin tocar localStorage.
   397→        const now = new Date();
   398→        const hour = now.getHours();
   399→        const today = now.toISOString().slice(0, 10);
   400→        const lastBriefDate = persisted.lastBriefDate ?? localStorage.getItem("koru.lastBriefDate");
   401→        if (hour >= 5 && hour <= 11 && lastBriefDate !== today) {
   402→          try {
   403→            const stateForBrief = {
   404→              memories: (persisted.memories ?? []).filter((m: any) => m.status === "confirmed" || m.status === "candidate").slice(0, 10).map((m: any) => ({ kind: m.kind, text: m.text })),
   405→              commitments: (persisted.commitments ?? []).filter((c: any) => c.status === "open").slice(0, 5).map((c: any) => ({ title: c.title, dueHint: c.dueHint, dueAt: c.dueAt })),
   406→              userName: persisted.userName ?? "",
   407→              lastBriefDate,
   408→            };
   409→            const res = await fetch("/api/koru/morning-brief", {
   410→              method: "POST",
   411→              headers: { "Content-Type": "application/json" },
   412→              body: JSON.stringify({ state: stateForBrief }),
   413→            });
   414→            if (res.ok) {
   415→              const data = await res.json();
   416→              if (data.shouldShow && data.brief) {
   417→                setMorningBrief(data.brief);
   418→                localStorage.setItem("koru.lastBriefDate", data.date ?? today);
   419→                // 🔴 TIER S: persistir el brief en el store (state.lastBriefDate
   420→                // + state.lastBriefBlock) vía el reducer setLastBrief. Así
   421→                // otras pantallas pueden inspeccionar si ya se mostró hoy sin
   422→                // leer localStorage directamente.
   423→                const briefBlock: UiBlock | undefined = data.brief?.block ?? data.brief;
   424→                commitDomainState((prev) => setLastBriefReducer(prev, data.date ?? today, briefBlock));
   425→              }
   426→            }
   427→          } catch {
   428→            // silent — brief is optional
   429→          }
   430→        }
   431→      }
   432→    });
   433→    return () => { cancelled = true; };
   434→  }, []);
   435→
   436→  // 🔴 Morning brief scheduler (sin LLM) — se ejecuta en mount y cada vez que
   437→  // `state.lastBriefDate` cambia. Si es de mañana (5-11h) y todavía no se
   438→  // mostró el brief hoy, dispatchea un nudge proactivo "Buenos días" y marca
   439→  // `lastBriefDate = today` para no repetir.
   440→  //
   441→  // A diferencia del useEffect de arriba (que llama a /api/koru/morning-brief

  1021→  async function submitEntry(
  1022→    text: string,
  1023→    transcriptSource: DailyEntry["transcriptSource"] = "typed",
  1024→    history: KoruChatTurn[] = chatTurnsRef.current,
  1025→  ) {
  1026→    setProcessing(true);
  1027→    let koruTurnId: string | null = null;
  1028→    // Trackea los items creados durante el stream para preservar sus ids al
  1029→    // reemplazar con el resultado final. Sin esto, el resultado final crea items
  1030→    // con ids nuevos → el WebNavCardA se desmonta/remonta → pierde su estado
  1031→    // (animación de progreso, visibleCount) → reinicia a "Iniciando búsqueda...".
  1032→    const streamedItemsByType = new Map<string, KoruTurnItem>();
  1033→    try {
  1034→      const previousState = domainStateRef.current;
  1035→      const domainHistory = history
  1036→        .filter((turn) => turn.text.trim().length > 0)
  1037→        .slice(-12)
  1038→        .map((turn) => ({
  1039→          role: turn.role === "koru" ? ("assistant" as const) : ("user" as const),
  1040→          content: turn.text,
  1041→          createdAt: turn.createdAt,
  1042→        }));
  1043→      const agentResult = await runBackendAgentTurn(text, previousState, domainHistory, selectedModel ?? undefined, (chunk) => {
  1044→        // Fase real del pipeline: tomamos el último stateEvent para la píldora/barra.
  1045→        if (chunk.stateEvents?.length) {
  1046→          const last = chunk.stateEvents[chunk.stateEvents.length - 1];
  1047→          setPhase(last.kind === "done" ? null : last.kind);
  1048→        }
  1049→        const blocksToItems = (blocks: UiBlock[]): KoruTurnItem[] =>
  1050→          blocks.map((block) => ({
  1051→            id: createId("item"),
  1052→            kind: "action" as const,
  1053→            tag: block.type,
  1054→            text: (block as any).title ?? (block as any).query ?? "",
  1055→            status: "working" as const,
  1056→            uiBlock: block,
  1057→          }));
  1058→        if (!koruTurnId) {
  1059→          koruTurnId = createId("turn");
  1060→          const koruTurn: KoruChatTurn = {
  1061→            id: koruTurnId,
  1062→            role: "koru",
  1063→            text: chunk.reply,
  1064→            createdAt: new Date().toISOString(),
  1065→            items: blocksToItems(chunk.uiBlocks),
  1066→            status: "working" as const,
  1067→            mascotState: chunk.mascotState ?? "working",
  1068→          };
  1069→          (koruTurn.items ?? []).forEach((it) => { if (it.uiBlock?.type) streamedItemsByType.set(it.uiBlock.type, it); });
  1070→          commitChatTurns((prev) => [...prev, koruTurn].slice(-120));
  1071→        } else {
  1072→          const isDone = chunk.stateEvents?.some((e) => e.kind === "done");
  1073→          commitChatTurns((prev) =>
  1074→            prev.map((turn) => {
  1075→              if (turn.id !== koruTurnId) return turn;
  1076→              const existingItems = turn.items ?? [];
  1077→              const newBlocks = chunk.uiBlocks ?? [];
  1078→              // Merge por TIPO de block (no por índice). Así, si el orden cambia
  1079→              // entre chunks (ej: aparece un data_card antes que el web_nav),
  1080→              // cada item mantiene su identidad y estado (animación de progreso,
  1081→              // contadores internos). Antes se matcheaba por idx, lo que hacía
  1082→              // que un cambio de orden desmontara el componente y reiniciara la
  1083→              // animación ("Iniciando búsqueda..." colgado).
  1084→              const consumedExisting = new Set<string>();
  1085→              const mergedItems: KoruTurnItem[] = newBlocks.map((block) => {
  1086→                const existing = existingItems.find(
  1087→                  (it) => it.uiBlock?.type === block.type && !consumedExisting.has(it.uiBlock!.type),
  1088→                );
  1089→                if (existing) {
  1090→                  consumedExisting.add(existing.uiBlock!.type);
  1091→                  return { ...existing, uiBlock: block, text: (block as any).title ?? (block as any).query ?? "" };
  1092→                }
  1093→                return {
  1094→                  id: createId("item"),
  1095→                  kind: "action" as const,
  1096→                  tag: block.type,
  1097→                  text: (block as any).title ?? (block as any).query ?? "",
  1098→                  status: "working" as const,
  1099→                  uiBlock: block,
  1100→                };
  1101→              });
  1102→              mergedItems.forEach((it) => { if (it.uiBlock?.type) streamedItemsByType.set(it.uiBlock.type, it); });
  1103→              return {
  1104→                ...turn,
  1105→                text: chunk.reply,
  1106→                items: mergedItems,
  1107→                mascotState: chunk.mascotState ?? turn.mascotState,
  1108→                status: isDone ? ("done" as const) : ("working" as const),
  1109→              };
  1110→            }),
  1111→          );
  1112→        }
  1113→      });
  1114→      const result = applyBackendTurnToState(previousState, text, transcriptSource, agentResult);
  1115→      commitDomainState(result.state);
  1116→
  1117→      // 🔴 Wire weather cache: si el backend devolvió un UiBlock de weather (tool
  1118→      // `weather_forecast` o `weather` legacy) con datos completos, actualizamos el
  1119→      // weatherCache del KoruState. El reducer existe en store.ts pero no tenía
  1120→      // caller — este es el puente entre tool result y el home widget de clima.
  1121→      const weatherBlock = (agentResult.uiBlocks ?? []).find(
  1122→        (b): b is Extract<UiBlock, { type: "weather" }> => b.type === "weather",
  1123→      );
  1124→      if (
  1125→        weatherBlock &&
  1126→        typeof weatherBlock.city === "string" &&
  1127→        weatherBlock.city.trim().length > 0 &&
  1128→        typeof weatherBlock.now === "string"
  1129→      ) {
  1130→        updateWeatherCache({
  1131→          city: weatherBlock.city,
  1132→          fetchedAt: weatherBlock.verifiedAt ?? new Date().toISOString(),
  1133→          payload: {
  1134→            now: weatherBlock.now,
  1135→            condition: weatherBlock.condition ?? "",
  1136→            hourly: weatherBlock.hourly,
  1137→            daily: weatherBlock.daily,
  1138→          },
  1139→        });
  1140→      }
  1141→
  1142→      // 🔴 Programar notificaciones para nuevos commitments con dueAt
  1143→      const newCommitments = (result.state.commitments ?? []).filter(
  1144→        (c) => c.dueAt && !previousState.commitments?.find((pc) => pc.id === c.id)
  1145→      );
  1146→      if (newCommitments.length > 0) {
  1147→        const userId = result.state.userId ?? "default";
  1148→        for (const c of newCommitments) {
  1149→          scheduleReminderNotification(c as any, userId);
  1150→          // 🔴 NUEVO: programar setTimeout preciso para que la notificación
  1151→          // dispare EXACTO cuando llega la hora, no en el próximo heartbeat tick.
  1152→          schedulePreciseTimeout(c.id, c.dueAt!, c.title, userId);
  1153→        }
  1154→        // 🔴 Request notification permission — best effort
  1155→        requestNotificationPermission().catch(() => {});
  1156→      }
  1157→      if (koruTurnId) {
  1158→        // Preservar ids de los items del stream: si el resultado final tiene un item
  1159→        // del mismo tipo que uno ya renderizado en el stream, mantener su id (e
  1160→        // identidad de React) para no desmontar/remontar el componente y perder su
  1161→        // estado interno (animación de progreso del WebNavCardA, visibleCount, etc.).
  1162→        const consumedTypes = new Set<string>();
  1163→        const stableItems = result.items.map((item) => {
  1164→          const type = item.uiBlock?.type ?? item.tag;
  1165→          if (type && !consumedTypes.has(type)) {
  1166→            const streamed = streamedItemsByType.get(type);
  1167→            if (streamed) {
  1168→              consumedTypes.add(type);
  1169→              return { ...item, id: streamed.id };
  1170→            }
  1171→          }
  1172→          return item;
  1173→        });
  1174→        commitChatTurns((prev) =>
  1175→          prev.map((turn) =>
  1176→            turn.id === koruTurnId
  1177→              ? { ...turn, text: agentResult.reply, items: stableItems, status: "done" as const, mascotState: agentResult.mascotState ?? "idle" }
  1178→              : turn,
  1179→          ),
  1180→        );
  1181→        // 🔴 Voice: si voice está activado, Koru habla la respuesta
  1182→        if (voiceEnabled && agentResult.reply) {
  1183→          speak(agentResult.reply);
  1184→        }
  1185→        // 🔴 Memory toast: si hay memoryCandidates en la respuesta, mostrar toast.
  1186→        // Buscar en memoryCandidates top-level Y dentro de toolResults (save_memory tool).
  1187→        const topCandidates = (agentResult.memoryCandidates ?? []);
  1188→        const toolCandidates = (agentResult.toolResults ?? []).flatMap((tr: any) => {
  1189→          const data = tr.data;
  1190→          if (data?.type === "memory_capture" && Array.isArray(data.memoryCandidates)) return data.memoryCandidates;
  1191→          if (data?.type === "personal_capture" && Array.isArray(data.memoryCandidates)) return data.memoryCandidates;
  1192→          return [];
  1193→        });
  1194→        const allCandidates = [...topCandidates, ...toolCandidates].filter(
  1195→          (m: any) => m && typeof m.text === "string" && m.text.length > 4,
  1196→        );
  1197→        if (allCandidates.length > 0 && !previousState.ephemeralMode && previousState.durableMemoryEnabled) {
  1198→          const first = allCandidates[0];
  1199→          // Solo mostrar si NO existe ya una memoria con el mismo texto (deduplicación visual)
  1200→          const normalizedNew = first.text.toLowerCase().replace(/[áéíóú]/g, (m: string) => ({ á: "a", é: "e", í: "i", ó: "o", ú: "u" }[m]!)).replace(/[^a-z0-9\s]/g, "").trim();
  1201→          const alreadyExists = previousState.memories.some((existing: any) => {
  1202→            const norm = existing.text.toLowerCase().replace(/[áéíóú]/g, (m: string) => ({ á: "a", é: "e", í: "i", ó: "o", ú: "u" }[m]!)).replace(/[^a-z0-9\s]/g, "").trim();
  1203→            return norm === normalizedNew;
  1204→          });
  1205→          if (!alreadyExists) {
  1206→            setMemoryToast({
  1207→              id: `toast_${Date.now()}`,
  1208→              kind: first.kind,
  1209→              text: first.text,
  1210→            });
  1211→            // Auto-dismiss después de 6 segundos
  1212→            setTimeout(() => {
  1213→              setMemoryToast((current) => (current?.text === first.text ? null : current));
  1214→            }, 6000);
  1215→          }
  1216→        }
  1217→      }
  1218→      writeAuditEvent({
  1219→        type: "turn_analyzed",
  1220→        engine: "backend_agent_loop",
  1221→        input: text,
  1222→        transcriptSource,
  1223→        historyUsed: domainHistory,
  1224→        response: agentResult.reply,
  1225→        understanding: agentResult.understanding,
  1226→        suggestedActions: agentResult.suggestedActions,
  1227→        toolResults: agentResult.toolResults,
  1228→        provider: agentResult.provider,
  1229→        model: agentResult.model,
  1230→        fallbackReason: agentResult.fallbackReason,
  1231→        items: auditTurnItems(result.items),
  1232→        state: auditStateSnapshot(result.state),
  1233→        delta: auditStateDelta(previousState, result.state),
  1234→      });
  1235→      return { response: agentResult.reply, items: result.items, state: result.state, mascotState: agentResult.mascotState, koruTurnId };
  1236→    } finally {
  1237→      setProcessing(false);
  1238→      setActivity(null);
  1239→      setPhase(null);
  1240→    }
  1241→  }

  1386→  function runReadonlyWebAction(original: AssistantAction): KoruTurnItem {
  1387→    // Dato en vivo (clima/trafico/mercado) = quick, sin panel largo. Un
  1388→    // informe/investigacion real (webMode "research", o web_research sin
  1389→    // modo puntual) = deep, se gana el panel "Trabajando...". Ver
  1390→    // flujo-informe-aoe2.html pasos 3 (informe) vs 7 (clima).
  1391→    const webMode = original.payload.webMode;
  1392→    const isQuickSignal = webMode === "weather" || webMode === "traffic" || webMode === "market";
  1393→    const isDeepResearch = webMode === "research" || (original.kind === "web_research" && !webMode);
  1394→    setActivity({
  1395→      kind: "searching",
  1396→      depth: isQuickSignal ? "quick" : isDeepResearch ? "deep" : "quick",
  1397→      label: isDeepResearch
  1398→        ? "Es un tema para investigar a fondo. Dame unos segundos."
  1399→        : original.kind === "world_signal"
  1400→          ? "Busco senales reales."
  1401→          : "Consulto fuentes reales.",
  1402→    });
  1403→    const now = new Date().toISOString();
  1404→    const approvedAction: AssistantAction = {
  1405→      ...original,
  1406→      status: "approved",
  1407→      updatedAt: now,
  1408→    };
  1409→    commitDomainState((prev) => {
  1410→      const exists = prev.actions.some((action) => action.id === original.id);
  1411→      const next = {
  1412→        ...prev,
  1413→        actions: exists
  1414→          ? prev.actions.map((action) => (action.id === original.id ? approvedAction : action))
  1415→          : [approvedAction, ...prev.actions],
  1416→        updatedAt: now,
  1417→      };
  1418→      saveState(next);
  1419→      return next;
  1420→    });
  1421→    const waitingItem = actionToTurnItem(approvedAction);
  1422→    replaceActionInChat(waitingItem);
  1423→
  1424→    void (async () => {
  1425→      try {
  1426→        const result = await runWebNavigation(original.payload);
  1427→        const webPayload = webResultToPayload(result);
  1428→        const nextUiBlock = patchUiBlockWithWebResult(approvedAction, webPayload);
  1429→        const finishedAt = new Date().toISOString();
  1430→        const patchedAction: AssistantAction = {
  1431→          ...approvedAction,
  1432→          payload: {
  1433→            ...approvedAction.payload,
  1434→            ...webPayload,
  1435→            uiBlock: nextUiBlock ?? approvedAction.payload.uiBlock,
  1436→          },
  1437→          result: result.recommendation,
  1438→          status: "executed",
  1439→          updatedAt: finishedAt,
  1440→          executedAt: finishedAt,
  1441→        };
  1442→        const patchedItem = actionToTurnItem(patchedAction);
  1443→        commitDomainState((prev) => {
  1444→          const next = {
  1445→            ...prev,
  1446→            actions: prev.actions.map((action) => (action.id === original.id ? patchedAction : action)),
  1447→            updatedAt: finishedAt,
  1448→          };
