    88→function escapeHtml(s: string): string {
    89→  return s.replace(/&/g, "&amp;").replace(/</g, "&lt;").replace(/>/g, "&gt;");
    90→}
    91→
    92→function inlineMd(s: string): string {
    93→  // s ya está escapado
    94→  s = s.replace(/\[([^\]]+)\]\(([^)]+)\)/g, '<a href="$2" target="_blank" rel="noopener noreferrer">$1</a>');
    95→  s = s.replace(/\*\*([^*]+)\*\*/g, "<strong>$1</strong>");
    96→  s = s.replace(/\*([^*]+)\*/g, "<em>$1</em>");
    97→  return s;
    98→}

   100→function renderMarkdown(text: string): string {
   101→  if (!text.trim()) return "";
   102→  const lines = text.split("\n");
   103→  const out: string[] = [];
   104→  let inList = false;
   105→  let para: string[] = [];
   106→  const flushPara = () => {
   107→    if (para.length > 0) {
   108→      out.push(`<p>${para.map(p => inlineMd(escapeHtml(p))).join("<br />")}</p>`);
   109→      para = [];
   110→    }
   111→  };
   112→  for (const line of lines) {
   113→    const listMatch = line.match(/^\s*-\s+(.*)$/);
   114→    if (listMatch) {
   115→      flushPara();
   116→      if (!inList) { out.push("<ul>"); inList = true; }
   117→      out.push(`<li>${inlineMd(escapeHtml(listMatch[1]))}</li>`);
   118→      continue;
   119→    }
   120→    if (inList) { out.push("</ul>"); inList = false; }
   121→    para.push(line);
   122→  }
   123→  flushPara();
   124→  if (inList) out.push("</ul>");
   125→  return out.join("");
   126→}

   425→        ? [1, 2, 3, 4, 5]
   426→        : rutinaCadence === "daily"
   427→          ? [0, 1, 2, 3, 4, 5, 6]
   428→          : rutinaCadence === "weekly"
   429→            ? [1]
   430→            : [0];
   431→      createRoutine(
   432→        title.trim() || "Rutina",
   433→        rutinaAnchorTime || "08:00",
   434→        [habitId],
   435→        daysOfWeek,

   437→      await persist({
   438→        title: title.trim() || "Rutina sin título",
   439→        collection: finalCollection,
   440→        notes: rutinaText,
   441→        kind,
   442→        sourceBlock: buildSavedRecordBlock({
   443→          title: title.trim() || "Rutina",
   444→          collection: finalCollection,
   445→          notes: rutinaText,
   446→          kind,
   447→        }),
   448→      });
   449→      return;
   450→    }
   451→
   452→    if (selected === "ejercicio") {
   453→      const weeks = parseInt(weeksTotal) || 0;
   454→      const sessionsText = sessions
   455→        .filter(s => s.dayLabel.trim() || s.exercises.some(e => e.name.trim()))
   456→        .map(s => {
   457→          const ex = s.exercises.filter(e => e.name.trim());
   458→          const exText = ex
   459→            .map(e => `  - ${e.name}: ${e.sets || 0}x${e.reps || 0}${e.weight ? ` @ ${e.weight}kg` : ""}`)
   460→            .join("\n");
   461→          return `${s.dayLabel || "Día"}:\n${exText}`;
   462→        })
   463→        .join("\n\n");
   464→      const ejText = `Semanas: ${weeks}\n\n${sessionsText}`;
   465→      // 🔴 GAP-3: además del LifeRecord, persistimos un ExercisePlan durable —
   466→      // habilita logWorkout, avance de sesión, completado del plan, etc.
   467→      const planSessions: Omit<ExerciseSession, "id" | "order">[] = sessions
   468→        .filter(s => s.dayLabel.trim() || s.exercises.some(e => e.name.trim()))
   469→        .map(s => ({
   470→          dayLabel: s.dayLabel || "Día",
   471→          exercises: s.exercises
   472→            .filter(e => e.name.trim())
   473→            .map(e => ({
   474→              exercise: e.name,
   475→              sets: parseInt(e.sets) || 0,
   476→              reps: parseInt(e.reps) || 0,
   477→              weight: e.weight ? parseFloat(e.weight) : undefined,
   478→            })),
   479→        }));
   480→      if (planSessions.length > 0) {
   481→        createExercisePlan(
   482→          title.trim() || "Plan de entrenamiento",
   483→          weeks,
   484→          planSessions,
   485→        );
   486→      }
   487→      await persist({
   488→        title: title.trim() || "Plan de entrenamiento",
   489→        collection: finalCollection,
   490→        notes: ejText,
   491→        kind,
   492→        sourceBlock: buildSavedRecordBlock({
   493→          title: title.trim() || "Plan de entrenamiento",
   494→          collection: finalCollection,
   495→          notes: ejText,
   496→          kind,
   497→        }),
   498→      });
   499→      return;
   500→    }
   501→
   502→    if (selected === "memoria") {
   503→      const memText = [
   504→        memoriaText.trim(),
   505→        `Tipo: ${memoriaKind}`,
   506→        `Confianza: ${Math.round(memoriaConfidence * 100)}%`,
   507→        memoriaSensitive ? "Sensibilidad: sensible" : null,
   508→      ].filter(Boolean).join("\n");
   509→      // 🔴 TIER S: si la memoria es de tipo "relationship", también persistimos
   510→      // una Person durable en el store (state.people) vía addPerson — habilita
   511→      // recordatorios de cumpleaños, follow-ups, etc. El título del form se
   512→      // interpreta como el nombre de la persona.
   513→      if (memoriaKind === "relationship" && title.trim()) {
   514→        addPerson(title.trim(), "relationship");
   515→      }

   517→        title: title.trim() || "Memoria",
   518→        collection: finalCollection,

   520→        kind,
   521→        sourceBlock: buildSavedRecordBlock({
   522→          title: title.trim() || "Memoria",
   523→          collection: finalCollection,
   524→          notes: memText,
   525→          kind,
   526→        }),
   527→      });
   528→      return;
   529→    }
   530→
   531→    if (selected === "decision") {
   532→      const facIds = factors.map(f => f.id);
   533→      const decText = [
   534→        deadline ? `Deadline: ${deadline}` : null,
   535→        "Factores:",
   536→        ...factors.filter(f => f.label.trim()).map(f => `  • ${f.label} (${f.direction === "higherIsBetter" ? "↑" : "↓"})`),
   537→        "Opciones:",
   538→        ...options.filter(o => o.label.trim()).map(o => `  • ${o.label}`),
   539→      ].filter(Boolean).join("\n");
   540→      // Embed structured payload (factor scores) as a JSON footer for later reopening.
   541→      const optionsWithScores = options.filter(o => o.label.trim()).map(o => ({
   542→        label: o.label,
   543→        factorScores: Object.fromEntries(facIds.map(id => [id, o.factorScores[id] ?? 0])),
   544→      }));
   545→      const structured = JSON.stringify({ weeks: null, factors: factors.filter(f => f.label.trim()), options: optionsWithScores, deadline: deadline || undefined });
   546→      const fullNotes = `${decText}\n\n---\n${structured}`;
   547→      // 🔴 GAP-2: además del LifeRecord, persistimos una Decision durable en
   548→      // el store y disparamos el motor (computeDecision) para poblar el
   549→      // resultado: recommendation, perOptionScore, perOptionProbability y
   550→      // confidenceInterval. El wrapper createDecision ejecuta computeDecision
   551→      // internamente y devuelve la Decision ya persistida con .result.
   552→      const decisionOptions: DecisionOption[] = options
   553→        .filter(o => o.label.trim())
   554→        .map(o => ({
   555→          id: o.id,
   556→          label: o.label,
   557→          factorScores: Object.fromEntries(facIds.map(id => [id, o.factorScores[id] ?? 0])),
   558→        }));
   559→      const decisionFactors: DecisionFactor[] = factors
   560→        .filter(f => f.label.trim())
   561→        .map(f => ({ id: f.id, label: f.label, direction: f.direction }));
   562→      // Pesos iguales (1.0) por factor — el motor normaliza internamente.
   563→      // Si no hay factores, weights queda vacío y el motor reparte probabilidad
   564→      // uniforme entre las opciones.
   565→      const weights: Record<string, number> = Object.fromEntries(
   566→        decisionFactors.map(f => [f.id, 1]),
   567→      );
   568→      if (decisionOptions.length > 0) {
   569→        const newDecision: Decision = createDecision(
   570→          title.trim() || "Decisión",
   571→          decisionOptions,
   572→          decisionFactors,
   573→          weights,
   574→        );
   575→        // 🔴 GAP-2 (contrato): invocamos computeDecision explícitamente desde
   576→        // la UI, sobre la decisión recién creada, para garantizar el poblado
   577→        // del resultado. Construimos un estado sintético porque `state` (de
   578→        // useKoru) aún no refleja la decisión nueva en este render cycle.
   579→        const syntheticState: KoruState = {
   580→          ...state,
   581→          decisions: [newDecision, ...(state.decisions ?? [])],
   582→        };
   583→        try {
   584→          computeDecision(syntheticState, newDecision.id);
   585→        } catch {
   586→          // best-effort: la decisión ya quedó persistida con .result por el wrapper.
   587→        }
   588→      }
   589→      await persist({
   590→        title: title.trim() || "Decisión",
   591→        collection: finalCollection,
   592→        notes: fullNotes,
   593→        kind,
   594→        sourceBlock: buildSavedRecordBlock({
   595→          title: title.trim() || "Decisión",
   596→          collection: finalCollection,
   597→          notes: fullNotes,
   598→          kind,
   599→        }),
   600→      });
   601→      return;
   602→    }
   603→
   604→    // nota (default)
   605→    await persist({
   606→      title: title.trim() || "Nota sin título",
   607→      collection: finalCollection,
   608→      notes: notes.trim() || undefined,
   609→      kind,
   610→    });
   611→  }

   688→  function insertWrap(wrap: string, placeholder: string) {
   689→    const ta = notesRef.current;
   690→    if (!ta) return;
   691→    const start = ta.selectionStart;
   692→    const end = ta.selectionEnd;
   693→    const sel = notes.slice(start, end) || placeholder;
   694→    const next = notes.slice(0, start) + wrap + sel + wrap + notes.slice(end);
   695→    setNotes(next);
   696→    requestAnimationFrame(() => {
   697→      ta.focus();
   698→      const pos = start + wrap.length;
   699→      ta.setSelectionRange(pos, pos + sel.length);
   700→    });
   701→  }
   702→
   703→  function insertPrefix(prefix: string) {
   704→    const ta = notesRef.current;
   705→    if (!ta) return;
   706→    const start = ta.selectionStart;
   707→    const before = notes.slice(0, start);
   708→    const lineStart = before.lastIndexOf("\n") + 1;
   709→    const next = notes.slice(0, lineStart) + prefix + notes.slice(lineStart);
   710→    setNotes(next);
   711→    requestAnimationFrame(() => {
   712→      ta.focus();
   713→      const pos = start + prefix.length;
   714→      ta.setSelectionRange(pos, pos);
   715→    });
   716→  }
   717→
   718→  function insertLink() {
   719→    const ta = notesRef.current;
   720→    if (!ta) return;
   721→    const start = ta.selectionStart;
   722→    const end = ta.selectionEnd;
   723→    const sel = notes.slice(start, end) || "texto";
   724→    const link = `[${sel}](https://)`;
   725→    const next = notes.slice(0, start) + link + notes.slice(end);
   726→    setNotes(next);
   727→    requestAnimationFrame(() => {
   728→      ta.focus();
   729→      const urlStart = start + sel.length + 3; // "[" + sel + "]("
   730→      ta.setSelectionRange(urlStart, urlStart + 8); // selecciona "https://"
   731→    });
   732→  }
