    23тЖТ    23тЖТ  const newline = text.indexOf("\n");

    25тЖТ    25тЖТ    const body = text.slice(newline + 1).trim();
    26тЖТ    26тЖТ    if (body) return { heading: text.slice(0, newline).trim(), body };

    29тЖТ    29тЖТ  if (match) return { heading: match[1].trim(), body: match[2].trim() };
