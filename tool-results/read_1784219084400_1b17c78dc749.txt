   280→  // Clave versionada (v2): la clave vieja "koru.selected-model" quedaba pegada con
   281→  // modelos de prueba (ej. llama3.1:8b) y pisaba silenciosamente al proveedor

   288→  // 🔴 i18n — preferred language. Persisted in localStorage so the LLM and UI honor it.
   289→  const [language, setLanguageState] = useState<"es" | "en">(() => {

   293→  // 🔴 Offline cache — track browser online/offline status

   297→  // 🔴 v2: record reabiertos desde Collections — renderizan el bloque original

   306→  function commitDomainState(next: KoruState | ((prev: KoruState) => KoruState)) {
   307→    setDomainState((prev) => {
   308→      const resolved = typeof next === "function" ? (next as (prev: KoruState) => KoruState)(prev) : next;

   314→  function commitChatTurns(next: KoruChatTurn[] | ((prev: KoruChatTurn[]) => KoruChatTurn[])) {
   315→    setChatTurns((prev) => {
   316→      const resolved = typeof next === "function" ? (next as (prev: KoruChatTurn[]) => KoruChatTurn[])(prev) : next;
   317→      chatTurnsRef.current = resolved;
   318→      return resolved;
   319→    });
   320→  }
   321→
   322→  useEffect(() => {
   323→    domainStateRef.current = domainState;

   326→  // 🔴 i18n — sync UI language into domainState so the backend LLM honors it
   327→  useEffect(() => {
   328→    commitDomainState((prev) => (prev.language === language ? prev : { ...prev, language }));
   329→  }, [language]);
   330→
   331→  // 🔴 PWA install prompt listener

   347→    chatTurnsRef.current = chatTurns;
   348→  }, [chatTurns]);
   349→
   350→  useEffect(() => {
   351→    try {
   352→      const params = new URLSearchParams(window.location.search);
   353→      if (params.get("koruAudit") !== "1" || params.get("reset") !== "1") return;
   354→      const sessionId = `audit_${Date.now().toString(36)}_${Math.random().toString(36).slice(2, 8)}`;
   355→      sessionStorage.setItem("koru.audit.enabled", "1");
   356→      sessionStorage.setItem("koru.audit.session", sessionId);
   357→      localStorage.removeItem(CHAT_STORAGE_KEY);
   358→      localStorage.removeItem("koru.onboarded");
   359→      localStorage.removeItem("koru.username");
   360→      const fresh = resetState();
   361→      commitDomainState(fresh);
   362→      setOnboarded(false);
   363→      setUserName("");
   364→      commitChatTurns([greetingTurn()]);
   365→      params.delete("reset");
   366→      const cleanQuery = params.toString();
   367→      const cleanUrl = `${window.location.pathname}${cleanQuery ? `?${cleanQuery}` : ""}${window.location.hash}`;
   368→      window.history.replaceState(null, "", cleanUrl);
   369→      writeAuditEvent({
   370→        type: "audit_reset",
   371→        note: "Koru browser memory, onboarding and chat reset for manual audit.",
   372→        state: auditStateSnapshot(fresh),
   373→      });
   374→    } catch {
   375→      // Audit reset is best-effort and should not block the app.
   376→    }
   377→  }, []);
   378→
   379→  useEffect(() => {
   380→    let cancelled = false;
   381→    void loadPersistedState().then(async (persisted) => {
   382→      if (!cancelled) {
   383→        const nudges = buildHeartbeatNudges(persisted);
   384→        commitDomainState(nudges.length > 0 ? applyHeartbeatNudges(persisted, nudges) : persisted);
   385→        // 🔴 Sincronizar reminders con notificaciones del navegador
   386→        const userId = persisted.userId ?? "default";
   387→        syncScheduledReminders(persisted.commitments ?? [], userId);
   388→        // Check overdue reminders on app load
   389→        checkDueReminders(userId);
   390→
   391→        // 🔴 Morning Brief: si es mañana y no se mostró hoy, generar.
   392→        // 🔴 TIER S: check usa `state.lastBriefDate` (vía persisted.lastBriefDate)
   393→        // como fuente de verdad, con fallback a localStorage para sesiones que
   394→        // todavía no tienen el campo en el store. Esto cumple el contrato del
   395→        // reducer setLastBrief: otras pantallas pueden inspeccionar si ya se
   396→        // mostró hoy leyendo state.lastBriefDate sin tocar localStorage.
   397→        const now = new Date();
   398→        const hour = now.getHours();
   399→        const today = now.toISOString().slice(0, 10);
   400→        const lastBriefDate = persisted.lastBriefDate ?? localStorage.getItem("koru.lastBriefDate");
   401→        if (hour >= 5 && hour <= 11 && lastBriefDate !== today) {
   402→          try {
   403→            // 🔴 FULL state: el endpoint necesita calendarEvents (eventos de
   404→            // hoy), commitments (deadlines debidos hoy), weatherCache, entries
   405→            // (último sentimiento) y memories (contexto de personalidad) para
   406→            // generar un brief completo. Mandamos el estado completo y dejamos
   407→            // que el endpoint filtre lo necesario.
   408→            const stateForBrief = {
   409→              userName: persisted.userName ?? "",
   410→              memories: (persisted.memories ?? [])
   411→                .filter((m: any) => m.status === "confirmed" || m.status === "candidate")
   412→                .slice(0, 10)
   413→                .map((m: any) => ({ kind: m.kind, text: m.text })),
   414→              commitments: (persisted.commitments ?? [])
   415→                .filter((c: any) => c.status === "open")
   416→                .slice(0, 20)
   417→                .map((c: any) => ({ title: c.title, dueHint: c.dueHint, dueAt: c.dueAt, status: c.status })),
   418→              calendarEvents: (persisted.calendarEvents ?? []).slice(0, 20).map((e: any) => ({
   419→                title: e.title,
   420→                startsAt: e.startsAt,
   421→                location: e.location,
   422→              })),
   423→              weatherCache: persisted.weatherCache ?? undefined,
   424→              entries: (persisted.entries ?? []).slice(-3).map((e: any) => ({
   425→                text: e.text,
   426→                sentiment: e.sentiment,
   427→                createdAt: e.createdAt,
   428→              })),
   429→              lastBriefDate,
   430→            };
   431→            const res = await fetch("/api/koru/morning-brief", {
   432→              method: "POST",
   433→              headers: { "Content-Type": "application/json" },
   434→              body: JSON.stringify({ state: stateForBrief }),
   435→            });
   436→            if (res.ok) {
   437→              const data = await res.json();
   438→              if (data.shouldShow && data.brief) {
   439→                const brief = data.brief;
   440→                // 🔴 Construir morning_brief UiBlock desde el brief del LLM.
   441→                // El LLM devuelve { greeting, items: [{icon, label, value}], reflection }.
   442→                // El UiBlock morning_brief requiere items con {icon, iconColor, label, value, variant?}.
   443→                // La reflection se appendea como un item destacado (variant: "highlight").
   444→                const uiItems: Array<{
   445→                  icon: string;
   446→                  iconColor: string;
   447→                  label: string;

  1092→  async function submitEntry(
  1093→    text: string,
  1094→    transcriptSource: DailyEntry["transcriptSource"] = "typed",
  1095→    history: KoruChatTurn[] = chatTurnsRef.current,
  1096→  ) {
  1097→    setProcessing(true);
  1098→    let koruTurnId: string | null = null;
  1099→    // Trackea los items creados durante el stream para preservar sus ids al
  1100→    // reemplazar con el resultado final. Sin esto, el resultado final crea items
  1101→    // con ids nuevos → el WebNavCardA se desmonta/remonta → pierde su estado
  1102→    // (animación de progreso, visibleCount) → reinicia a "Iniciando búsqueda...".
  1103→    const streamedItemsByType = new Map<string, KoruTurnItem>();
  1104→    try {
  1105→      const previousState = domainStateRef.current;
  1106→      const domainHistory = history
  1107→        .filter((turn) => turn.text.trim().length > 0)
  1108→        .slice(-12)
  1109→        .map((turn) => ({
  1110→          role: turn.role === "koru" ? ("assistant" as const) : ("user" as const),
  1111→          content: turn.text,
  1112→          createdAt: turn.createdAt,
  1113→        }));
  1114→      const agentResult = await runBackendAgentTurn(text, previousState, domainHistory, selectedModel ?? undefined, (chunk) => {
  1115→        // Fase real del pipeline: tomamos el último stateEvent para la píldora/barra.
  1116→        if (chunk.stateEvents?.length) {
  1117→          const last = chunk.stateEvents[chunk.stateEvents.length - 1];
  1118→          setPhase(last.kind === "done" ? null : last.kind);
  1119→        }
  1120→        const blocksToItems = (blocks: UiBlock[]): KoruTurnItem[] =>
  1121→          blocks.map((block) => ({
  1122→            id: createId("item"),
  1123→            kind: "action" as const,
  1124→            tag: block.type,
  1125→            text: (block as any).title ?? (block as any).query ?? "",
  1126→            status: "working" as const,
  1127→            uiBlock: block,
  1128→          }));
  1129→        if (!koruTurnId) {
  1130→          koruTurnId = createId("turn");
  1131→          const koruTurn: KoruChatTurn = {
  1132→            id: koruTurnId,
  1133→            role: "koru",
  1134→            text: chunk.reply,
  1135→            createdAt: new Date().toISOString(),
  1136→            items: blocksToItems(chunk.uiBlocks),
  1137→            status: "working" as const,
  1138→            mascotState: chunk.mascotState ?? "working",
  1139→          };
  1140→          (koruTurn.items ?? []).forEach((it) => { if (it.uiBlock?.type) streamedItemsByType.set(it.uiBlock.type, it); });
  1141→          commitChatTurns((prev) => [...prev, koruTurn].slice(-120));
  1142→        } else {
  1143→          const isDone = chunk.stateEvents?.some((e) => e.kind === "done");
  1144→          commitChatTurns((prev) =>
  1145→            prev.map((turn) => {
  1146→              if (turn.id !== koruTurnId) return turn;
  1147→              const existingItems = turn.items ?? [];
  1148→              const newBlocks = chunk.uiBlocks ?? [];
  1149→              // Merge por TIPO de block (no por índice). Así, si el orden cambia
  1150→              // entre chunks (ej: aparece un data_card antes que el web_nav),
  1151→              // cada item mantiene su identidad y estado (animación de progreso,
  1152→              // contadores internos). Antes se matcheaba por idx, lo que hacía
  1153→              // que un cambio de orden desmontara el componente y reiniciara la
  1154→              // animación ("Iniciando búsqueda..." colgado).
  1155→              const consumedExisting = new Set<string>();
  1156→              const mergedItems: KoruTurnItem[] = newBlocks.map((block) => {
  1157→                const existing = existingItems.find(
  1158→                  (it) => it.uiBlock?.type === block.type && !consumedExisting.has(it.uiBlock!.type),
  1159→                );
  1160→                if (existing) {
  1161→                  consumedExisting.add(existing.uiBlock!.type);
  1162→                  return { ...existing, uiBlock: block, text: (block as any).title ?? (block as any).query ?? "" };
  1163→                }
  1164→                return {
  1165→                  id: createId("item"),
  1166→                  kind: "action" as const,
  1167→                  tag: block.type,
  1168→                  text: (block as any).title ?? (block as any).query ?? "",
  1169→                  status: "working" as const,
  1170→                  uiBlock: block,
  1171→                };
  1172→              });
  1173→              mergedItems.forEach((it) => { if (it.uiBlock?.type) streamedItemsByType.set(it.uiBlock.type, it); });
  1174→              return {
  1175→                ...turn,
  1176→                text: chunk.reply,
  1177→                items: mergedItems,
  1178→                mascotState: chunk.mascotState ?? turn.mascotState,
  1179→                status: isDone ? ("done" as const) : ("working" as const),
  1180→              };
  1181→            }),
  1182→          );
  1183→        }
  1184→      });
  1185→      const result = applyBackendTurnToState(previousState, text, transcriptSource, agentResult);
  1186→      commitDomainState(result.state);
  1187→
  1188→      // 🔴 Wire weather cache: si el backend devolvió un UiBlock de weather (tool
  1189→      // `weather_forecast` o `weather` legacy) con datos completos, actualizamos el
  1190→      // weatherCache del KoruState. El reducer existe en store.ts pero no tenía
  1191→      // caller — este es el puente entre tool result y el home widget de clima.
  1192→      const weatherBlock = (agentResult.uiBlocks ?? []).find(
  1193→        (b): b is Extract<UiBlock, { type: "weather" }> => b.type === "weather",
  1194→      );
  1195→      if (
  1196→        weatherBlock &&
  1197→        typeof weatherBlock.city === "string" &&
  1198→        weatherBlock.city.trim().length > 0 &&
  1199→        typeof weatherBlock.now === "string"
  1200→      ) {
  1201→        updateWeatherCache({
  1202→          city: weatherBlock.city,
  1203→          fetchedAt: weatherBlock.verifiedAt ?? new Date().toISOString(),
  1204→          payload: {
  1205→            now: weatherBlock.now,
  1206→            condition: weatherBlock.condition ?? "",
  1207→            hourly: weatherBlock.hourly,
  1208→            daily: weatherBlock.daily,
  1209→          },
  1210→        });
  1211→      }
  1212→
  1213→      // 🔴 Programar notificaciones para nuevos commitments con dueAt
  1214→      const newCommitments = (result.state.commitments ?? []).filter(
  1215→        (c) => c.dueAt && !previousState.commitments?.find((pc) => pc.id === c.id)
  1216→      );
  1217→      if (newCommitments.length > 0) {
  1218→        const userId = result.state.userId ?? "default";
  1219→        for (const c of newCommitments) {
  1220→          scheduleReminderNotification(c as any, userId);
  1221→          // 🔴 NUEVO: programar setTimeout preciso para que la notificación
  1222→          // dispare EXACTO cuando llega la hora, no en el próximo heartbeat tick.
  1223→          schedulePreciseTimeout(c.id, c.dueAt!, c.title, userId);
  1224→        }
  1225→        // 🔴 Request notification permission — best effort
  1226→        requestNotificationPermission().catch(() => {});
  1227→      }
  1228→      if (koruTurnId) {
  1229→        // Preservar ids de los items del stream: si el resultado final tiene un item
  1230→        // del mismo tipo que uno ya renderizado en el stream, mantener su id (e
  1231→        // identidad de React) para no desmontar/remontar el componente y perder su
  1232→        // estado interno (animación de progreso del WebNavCardA, visibleCount, etc.).
  1233→        const consumedTypes = new Set<string>();
  1234→        const stableItems = result.items.map((item) => {
  1235→          const type = item.uiBlock?.type ?? item.tag;
  1236→          if (type && !consumedTypes.has(type)) {
  1237→            const streamed = streamedItemsByType.get(type);
  1238→            if (streamed) {
  1239→              consumedTypes.add(type);
  1240→              return { ...item, id: streamed.id };
  1241→            }
  1242→          }
  1243→          return item;
  1244→        });
  1245→        commitChatTurns((prev) =>
  1246→          prev.map((turn) =>
  1247→            turn.id === koruTurnId
  1248→              ? { ...turn, text: agentResult.reply, items: stableItems, status: "done" as const, mascotState: agentResult.mascotState ?? "idle" }
  1249→              : turn,
  1250→          ),
  1251→        );
  1252→        // 🔴 Voice: si voice está activado, Koru habla la respuesta
  1253→        if (voiceEnabled && agentResult.reply) {
  1254→          speak(agentResult.reply);
  1255→        }
  1256→        // 🔴 Memory toast: si hay memoryCandidates en la respuesta, mostrar toast.
  1257→        // Buscar en memoryCandidates top-level Y dentro de toolResults (save_memory tool).
  1258→        const topCandidates = (agentResult.memoryCandidates ?? []);
  1259→        const toolCandidates = (agentResult.toolResults ?? []).flatMap((tr: any) => {
  1260→          const data = tr.data;
  1261→          if (data?.type === "memory_capture" && Array.isArray(data.memoryCandidates)) return data.memoryCandidates;
  1262→          if (data?.type === "personal_capture" && Array.isArray(data.memoryCandidates)) return data.memoryCandidates;
  1263→          return [];
  1264→        });
  1265→        const allCandidates = [...topCandidates, ...toolCandidates].filter(
  1266→          (m: any) => m && typeof m.text === "string" && m.text.length > 4,
  1267→        );
  1268→        if (allCandidates.length > 0 && !previousState.ephemeralMode && previousState.durableMemoryEnabled) {
  1269→          const first = allCandidates[0];
  1270→          // Solo mostrar si NO existe ya una memoria con el mismo texto (deduplicación visual)
  1271→          const normalizedNew = first.text.toLowerCase().replace(/[áéíóú]/g, (m: string) => ({ á: "a", é: "e", í: "i", ó: "o", ú: "u" }[m]!)).replace(/[^a-z0-9\s]/g, "").trim();
  1272→          const alreadyExists = previousState.memories.some((existing: any) => {
  1273→            const norm = existing.text.toLowerCase().replace(/[áéíóú]/g, (m: string) => ({ á: "a", é: "e", í: "i", ó: "o", ú: "u" }[m]!)).replace(/[^a-z0-9\s]/g, "").trim();
  1274→            return norm === normalizedNew;
  1275→          });
  1276→          if (!alreadyExists) {
  1277→            setMemoryToast({
  1278→              id: `toast_${Date.now()}`,
  1279→              kind: first.kind,
  1280→              text: first.text,
  1281→            });
  1282→            // Auto-dismiss después de 6 segundos
  1283→            setTimeout(() => {
  1284→              setMemoryToast((current) => (current?.text === first.text ? null : current));
  1285→            }, 6000);
  1286→          }
  1287→        }
  1288→      }
  1289→      writeAuditEvent({
  1290→        type: "turn_analyzed",
  1291→        engine: "backend_agent_loop",
  1292→        input: text,
  1293→        transcriptSource,
  1294→        historyUsed: domainHistory,
  1295→        response: agentResult.reply,
  1296→        understanding: agentResult.understanding,
  1297→        suggestedActions: agentResult.suggestedActions,
  1298→        toolResults: agentResult.toolResults,
  1299→        provider: agentResult.provider,
  1300→        model: agentResult.model,
  1301→        fallbackReason: agentResult.fallbackReason,
  1302→        items: auditTurnItems(result.items),
  1303→        state: auditStateSnapshot(result.state),
  1304→        delta: auditStateDelta(previousState, result.state),
  1305→      });
  1306→      return { response: agentResult.reply, items: result.items, state: result.state, mascotState: agentResult.mascotState, koruTurnId };
  1307→    } finally {
  1308→      setProcessing(false);
  1309→      setActivity(null);
  1310→      setPhase(null);
  1311→    }
  1312→  }

  1314→  async function sendMessage(text: string, transcriptSource: DailyEntry["transcriptSource"] = "typed") {
  1315→    const cleanText = text.trim();
  1316→    if (!cleanText) return null;
  1317→    const previousState = domainStateRef.current;
  1318→    const userTurn: KoruChatTurn = {
  1319→      id: createId("turn"),
  1320→      role: "user",
  1321→      text: cleanText,
  1322→      createdAt: new Date().toISOString(),
  1323→      status: "done",
  1324→    };
  1325→    const historyBeforeUser = chatTurnsRef.current;
  1326→    commitChatTurns((prev) => [...prev, userTurn].slice(-120));
  1327→    writeAuditEvent({
  1328→      type: "user_message",
  1329→      turn: userTurn,
  1330→      transcriptSource,
  1331→      historyBeforeUser: historyBeforeUser.map((turn) => ({
  1332→        id: turn.id,
  1333→        role: turn.role,
  1334→        text: turn.text,
  1335→        createdAt: turn.createdAt,
  1336→        status: turn.status,
  1337→        items: auditTurnItems(turn.items),
  1338→      })),
  1339→      stateBefore: auditStateSnapshot(previousState),
  1340→    });
  1341→    setActivity(inferActivity(cleanText));
  1342→    try {
  1343→      const result = await submitEntry(cleanText, transcriptSource, historyBeforeUser);
  1344→      let koruTurn: KoruChatTurn | undefined;
  1345→      if (!result.koruTurnId) {
  1346→        koruTurn = {
  1347→          id: createId("turn"),
  1348→          role: "koru",
  1349→          text: result.response,
  1350→          createdAt: new Date().toISOString(),
  1351→          items: result.items,
  1352→          status: "done",
  1353→          mascotState: result.mascotState ?? "idle",
  1354→        };
  1355→        commitChatTurns((prev) => [...prev, koruTurn!].slice(-120));
  1356→        writeAuditEvent({
  1357→          type: "koru_message",
  1358→          turn: {
  1359→            id: koruTurn.id,
  1360→            role: koruTurn.role,
  1361→            text: koruTurn.text,
  1362→            createdAt: koruTurn.createdAt,
  1363→            status: koruTurn.status,
  1364→            items: auditTurnItems(koruTurn.items),
  1365→          },
  1366→          stateAfter: auditStateSnapshot(result.state),
  1367→        });
  1368→      } else {
  1369→        commitChatTurns((prev) =>
  1370→          prev.map((turn) => {
  1371→            if (turn.id !== result.koruTurnId) return turn;
  1372→            const existingItems = turn.items ?? [];
  1373→            const mergedItems: KoruTurnItem[] = result.items.map((newItem, idx) => {
  1374→              const existing = existingItems[idx];
  1375→              if (existing && existing.uiBlock?.type === newItem.uiBlock?.type) {
  1376→                return { ...newItem, id: existing.id };
  1377→              }
  1378→              return newItem;
  1379→            });
  1380→            return { ...turn, items: mergedItems, status: "done" as const, mascotState: result.mascotState ?? "idle" };
  1381→          }),
  1382→        );
  1383→        koruTurn = chatTurnsRef.current.find((t) => t.id === result.koruTurnId);
  1384→        writeAuditEvent({
  1385→          type: "koru_message",
  1386→          turn: {
  1387→            id: result.koruTurnId,
  1388→            role: "koru",
  1389→            text: result.response,
  1390→            createdAt: new Date().toISOString(),
  1391→            status: "done",
  1392→            items: auditTurnItems(result.items),
  1393→          },
  1394→          stateAfter: auditStateSnapshot(result.state),
  1395→        });
  1396→      }
  1397→      const autoWebAction = result.state.actions
  1398→        .filter((action) => !previousState.actions.some((existing) => existing.id === action.id))
  1399→        .find(shouldAutoRunAction);
  1400→      if (autoWebAction) {
  1401→        window.setTimeout(() => runReadonlyWebAction(autoWebAction), 0);
  1402→      }
  1403→      return koruTurn ?? null;
  1404→    } catch (error) {
  1405→      const message = error instanceof Error ? error.message : "Fallo desconocido del agente.";
  1406→      const errorTurn: KoruChatTurn = {
  1407→        id: createId("turn"),
  1408→        role: "koru",
  1409→        text: `No pude contactar bien al agente ahora. No voy a inventarte una respuesta. Detalle: ${message}`,
  1410→        createdAt: new Date().toISOString(),
  1411→        status: "error",
  1412→      };
  1413→      commitChatTurns((prev) => [...prev, errorTurn].slice(-120));
  1414→      writeAuditEvent({
  1415→        type: "turn_error",
  1416→        input: cleanText,
  1417→        error: message,
  1418→        turn: errorTurn,
  1419→        state: auditStateSnapshot(domainStateRef.current),
  1420→      });
  1421→      setActivity(null);
  1422→      return errorTurn;
  1423→    }
  1424→  }

  1459→    // informe/investigacion real (webMode "research", o web_research sin

  1495→    void (async () => {

  1518→            updatedAt: finishedAt,
