    26→function splitKoruText(text: string): { heading: string | null; body: string } {
    27→  const newline = text.indexOf("\n");
    28→  if (newline > 0 && newline <= 48) {
    29→    const body = text.slice(newline + 1).trim();
    30→    if (body) return { heading: text.slice(0, newline).trim(), body };
    31→  }
    32→  const match = text.match(/^(.{2,44}?[!¡?¿])\s+([\s\S]+)$/);
    33→  if (match) return { heading: match[1].trim(), body: match[2].trim() };
    34→  return { heading: null, body: text };
    35→}

   261→function WorkingPanel({ phase, kind, deliverable, onCancel }: { phase: string | null; kind?: AgentActivityKind; deliverable?: WorkingDeliverable | null; onCancel?: () => void }) {
   262→  const idx = phase ? (PHASE_ORDER as readonly string[]).indexOf(phase) : -1;
   263→  const doneIdx = PHASE_ORDER.length - 1;
   264→  const pct = deliverable?.progress != null
   265→    ? Math.min(100, Math.max(0, Math.round(deliverable.progress)))
   266→    : idx >= 0 ? Math.round(((idx + 1) / doneIdx) * 100) : null;
   267→
   268→  // 🔴 FIX UX: usar chips dinámicos según el tipo de tarea
   269→  const taskChips = getTaskPhases(deliverable?.kicker, kind);
   270→
   271→  // 🔴 Voz mágica — el título del deliverable no dice "Trabajando en..." (frío),
   272→  // sino que abraza con una frase cálida + nombre del deliverable.
   273→  const kicker = deliverable?.kicker?.toLowerCase() ?? "";
   274→  const friendlyTitle = deliverable
   275→    ? `Sumergiéndome en ${kicker.startsWith("tu") ? kicker : `tu ${kicker}`}…`
   276→    : null;
   277→
   278→  const copy = deliverable
   279→    ? {
   280→        title: friendlyTitle ?? "Trabajando…",
   281→        subtitle: deliverable.phaseLabel ?? "Me llevo esto en serio 🌿",
   282→      }
   283→    : (kind && WORKING_COPY[kind]) ?? { title: "Trabajando…", subtitle: "Me llevo esto en serio 🌿" };
   284→
   285→  const phaseIdx = phase ? (PHASE_ORDER as readonly string[]).indexOf(phase) : -1;
   286→  const chips = taskChips.map((chip, i) => {
   287→    if (phaseIdx < 0) return { ...chip, status: "pending" as const };
   288→    if (i < phaseIdx) return { ...chip, status: "done" as const };
   289→    if (i === phaseIdx) return { ...chip, status: "active" as const };
   290→    return { ...chip, status: "pending" as const };
   291→  });
   292→
   293→  // 🔴 Motto mágico rotativo — se elige uno al azar al render para que se
   294→  // sienta vivo, no como una frase repetida mecánicamente.
   295→  const motto = MAGIC_MOTTOS[Math.floor(Math.random() * MAGIC_MOTTOS.length)];
   296→
   297→  return (

   299→      <img src={getTaskIllustration(deliverable?.kicker, kind)} alt="" className="koru-working-illustration" />

   301→        <h2>{copy.title}</h2>
   302→        <p>{copy.subtitle}</p>
   303→      </div>
   304→      <div className="koru-working-progress">
   305→        <div
   306→          className="koru-progress"
   307→          role="progressbar"

   310→          aria-valuemax={100}
   311→          aria-valuenow={pct ?? undefined}
   312→        >
   313→          {pct === null ? (
   314→            <span className="koru-progress-bar is-indeterminate" />
   315→          ) : (
   316→            <span className="koru-progress-bar" style={{ width: `${pct}%` }} />
   317→          )}
   318→        </div>
   319→        {pct !== null && <span className="koru-working-pct">{pct}%</span>}
   320→      </div>
   321→      {/* Chips de fases — réplica exacta del demo pantalla 2 */}
   322→      <div className="koru-working-chips">
   323→        {chips.map((chip, i) => (
   324→          <span key={i} className={`koru-working-chip is-${chip.status}`}>
   325→            {chip.status === "done" ? (
   326→              <span className="material-symbols-outlined koru-working-chip-icon is-done">check_circle</span>

   331→            )}
   332→            {chip.label}
   333→          </span>
   334→        ))}
   335→      </div>
   336→      <p className="koru-working-motto">
   337→        <svg fill="none" height="20" viewBox="0 0 24 24" width="20" xmlns="http://www.w3.org/2000/svg">

   340→        {motto}
   341→        <svg fill="currentColor" height="18" viewBox="0 0 24 24" width="18" xmlns="http://www.w3.org/2000/svg">
   342→          <path d="M12 21.35L10.55 20.03C5.4 15.36 2 12.28 2 8.5C2 5.42 4.42 3 7.5 3C9.24 3 10.91 3.81 12 5.09C13.09 3.81 14.76 3 16.5 3C19.58 3 22 5.42 22 8.5C22 12.28 18.6 15.36 13.45 20.04L12 21.35Z" />
   343→        </svg>
