   471→function plainLower(value: string): string {
   472→  return value.normalize("NFD").replace(/[\u0300-\u036f]/g, "").toLowerCase();
   473→}

   475→function timeFromText(value: string): string | undefined {
   476→  const match = /\b(?:a\s+las|las)\s+(\d{1,2})(?::(\d{2}))?\b/i.exec(plainLower(value));
   477→  if (!match) return undefined;
   478→  const hour = Math.max(0, Math.min(23, Number(match[1])));
   479→  const minute = match[2] ? Math.max(0, Math.min(59, Number(match[2]))) : 0;
   480→  return `${hour.toString().padStart(2, "0")}:${minute.toString().padStart(2, "0")}`;
   481→}

   483→function createId(prefix: string): string {
   484→  if (typeof crypto !== "undefined" && "randomUUID" in crypto) {
   485→    return `${prefix}_${crypto.randomUUID().replace(/-/g, "").slice(0, 12)}`;
   486→  }
   487→  return `${prefix}_${Math.random().toString(36).slice(2, 12)}`;
   488→}

   495→  const clean = String(name ?? "").trim().toUpperCase();
   496→  if (!clean) return "???";
   497→  // Si ya es sigla (PSG, MLS, etc), devolverla
   498→  if (/^[A-Z]{2,4}$/.test(clean) && clean.length <= 4) return clean;
   499→  const words = clean.split(/\s+/).filter(w => w.length > 0);
   500→  if (words.length === 1) return words[0].slice(0, 3);
   501→  // Tomar primera letra de las primeras 3 palabras, saltando artículos
   502→  const filtered = words.filter(w => !["DE", "DEL", "LA", "LAS", "EL", "LOS", "Y", "FC", "CF"].includes(w));
   503→  return (filtered.slice(0, 3).map(w => w[0]).join("") || words[0].slice(0, 3)).padEnd(3, "X");
   504→}

   507→ * Formatea una fecha ISO a "DD/MM YYYY" o "Hoy" / "Ayer" según corresponda.
   508→ */
   509→function formatMatchDate(iso: string): string {
   510→  try {
   511→    const d = new Date(iso);
   512→    if (isNaN(d.getTime())) return "";
   513→    const now = new Date();
   514→    const todayStart = new Date(now.getFullYear(), now.getMonth(), now.getDate());
   515→    const dStart = new Date(d.getFullYear(), d.getMonth(), d.getDate());
   516→    const diffDays = Math.round((todayStart.getTime() - dStart.getTime()) / (24 * 60 * 60 * 1000));
   517→    if (diffDays === 0) return "Hoy";
   518→    if (diffDays === 1) return "Ayer";
   519→    if (diffDays === -1) return "Mañana";
   520→    const day = String(d.getDate()).padStart(2, "0");
   521→    const month = String(d.getMonth() + 1).padStart(2, "0");
   522→    return `${day}/${month}`;
   523→  } catch { return ""; }
   524→}

   534→function extractJsonBlock(text: string): string {
   535→  const start = text.indexOf("{");
   536→  if (start === -1) return text;
   537→  let depth = 0;
   538→  let inString = false;
   539→  let escapeNext = false;
   540→  for (let i = start; i < text.length; i++) {
   541→    const c = text[i];
   542→    if (escapeNext) { escapeNext = false; continue; }
   543→    if (c === "\\") { escapeNext = true; continue; }
   544→    if (c === '"' && !inString) { inString = true; continue; }
   545→    if (c === '"' && inString) { inString = false; continue; }
   546→    if (inString) continue;
   547→    if (c === "{") depth++;
   548→    if (c === "}") { depth--; if (depth === 0) return text.slice(start, i + 1); }
   549→  }
   550→  return text;
   551→}

   553→/**
   554→ * 🔴 FIX P0 — stripReasoning: elimina razonamiento interno del LLM antes de
   555→ * exponerlo como reply al usuario.
   556→ *
   557→ * Causa raíz del Bug 1: Nemotron Ultra (y otros modelos con modo "thinking")
   558→ * a veces emiten su CoT en el campo `content` en lugar de `reasoning_content`.
   559→ * Esto se ve como: "The user is asking about Argentina's football match...".
   560→ *
   561→ * Estrategias de strippado (en orden):
   562→ *  1. Tags XML-style explícitos: <think>...</think>, <reasoning>...</reasoning>,
   563→ *     <reflection>...</reflection>, <output>...</output> (conservar solo el contenido de <output>).
   564→ *  2. CoT en inglés antes del JSON: si el texto contiene `{"reply"` más adelante,
   565→ *     tirar todo lo anterior al primer `{`.
   566→ *  3. AGRESIVO: si el texto EMPIEZA con un patrón de thinking en inglés
   567→ *     ("The user is asking...", "I need to...", "Let me...", "I should..."),
   568→ *     y NO contiene JSON reply, devolver "" — es thinking puro.
   569→ *     Sin threshold de length: cualquier thinking al inicio = strip.
   570→ *  4. AGRESIVO: si el texto contiene 2+ indicadores de thinking
   571→ *     ("I need to", "Let me", "I should", "I will", "The user..."),
   572→ *     y NO contiene JSON reply, devolver "".
   573→ *
   574→ * Tests verificados:
   575→ *  - "The user is asking about Argentina's match yesterday..." (148 chars) → ""
   576→ *  - "The user is asking about Argentina's football match... Let me use match_live..." (280 chars) → ""
   577→ *  - "¡Hola Juan! ¿Cómo va todo?" → preservado
   578→ *  - "En Madrid hace 29°C ahora..." → preservado
   579→ *  - "Listo, guardado en gastos." → preservado
   580→ */
   581→function stripReasoning(text: string): string {
   582→  if (!text) return "";
   583→  let out = text;
   584→  // 1. Tags XML-style
   585→  out = out
   586→    .replace(/<think>[\s\S]*?<\/think>/gi, "")
   587→    .replace(/<reasoning>[\s\S]*?<\/reasoning>/gi, "")
   588→    .replace(/<reflection>[\s\S]*?<\/reflection>/gi, "")
   589→    .replace(/<reflection>[\s\S]*$/gi, "") // reflection sin cerrar (truncado)
   590→    .replace(/<\/?think>/gi, "")
   591→    .replace(/<\/?reasoning>/gi, "");
   592→  // Si hay tag <output>, conservar SOLO su contenido
   593→  const outputMatch = out.match(/<output>([\s\S]*?)<\/output>/i);
   594→  if (outputMatch) {
   595→    out = outputMatch[1];
   596→  }
   597→  // 2. Si hay `{"reply"` o `{"reply":` más adelante, tirar todo lo anterior (CoT en inglés)
   598→  const jsonStart = out.search(/\{\s*["']reply["']\s*:/);
   599→  if (jsonStart > 0) {
   600→    out = out.slice(jsonStart);
   601→  }
   602→  // Si ya encontramos JSON reply, no seguir strippando
   603→  const hasJsonReply = /\{\s*["']reply["']\s*:/.test(out);
   604→  if (hasJsonReply) {
   605→    return out;
   606→  }
   607→  // 3. AGRESIVO: si empieza con patrón de thinking en inglés, strip total
   608→  //    Patrones ampliados para cubrir más variantes del CoT de Nemotron.
   609→  const thinkingStartPatterns = [
   610→    /^(the user|the user is|the user wants|the user is asking|i should|i need to|let me|let's think|i'll|i will|i am going to|first,?\s*i|now i|the question|looking at|analyzing|to answer this|based on the|so,?\s*i|this is a|this is an|let's consider|step by step|i have to|i must|i'm going to|the request|the input|the message|i want to|i can|i could|i'm thinking|okay,?\s*(so|i|let|the)|alright,?\s*(so|i|let|the))\b/i,
   611→  ];
   612→  const trimmed = out.trim();
   613→  if (trimmed.length > 20 && thinkingStartPatterns.some(re => re.test(trimmed))) {
   614→    return "";
   615→  }
   616→  // 4. AGRESIVO: si contiene múltiples indicadores de thinking, es thinking
   617→  //    aunque no empiece exactamente con un patrón.
   618→  const thinkingIndicators = (out.match(/\b(i need to|let me|i should|i will|i'll|i am going to|i'm going to|i have to|i must|the user|i want to|i can|step by step|let's think|i think|i believe|first i|then i|next i|finally i)\b/gi) || []).length;
   619→  if (thinkingIndicators >= 2 && trimmed.length > 30) {
   620→    return "";
   621→  }
   622→  return out;
   623→}

   625→function safeJsonObjectFromContent(raw: string): Record<string, unknown> {
   626→  // 🔴 FIX P0: stripar reasoning ANTES de intentar parsear JSON
   627→  const cleaned = stripReasoning(raw);
   628→  const direct = safeJsonParse(cleaned);
   629→  if (direct.reply !== undefined || direct.uiBlocks !== undefined) return direct;
   630→  const extracted = safeJsonParse(extractJsonBlock(cleaned));

   632→  // Dirty extraction: fields individually with regex
   633→  const reply = extractStringField(cleaned, "reply");
   634→  const mascotState = extractStringField(cleaned, "mascotState") || extractStringField(cleaned, "mascot_state");
   635→  if (reply && reply.length > 3) {
   636→    return { reply, mascotState: mascotState || "idle", uiBlocks: [] };
   637→  }
   638→  return {};

   641→function extractStringField(raw: string, field: string): string | undefined {
   642→  const idx = raw.toLowerCase().indexOf(`"${field.toLowerCase()}"`);
   643→  if (idx === -1) return undefined;
   644→  let start = raw.indexOf('"', idx + field.length + 2);
   645→  if (start === -1) return undefined;
   646→  start++;
   647→  let i = start;
   648→  let escaped = false;
   649→  while (i < raw.length) {
   650→    const c = raw[i];
   651→    if (escaped) { escaped = false; i++; continue; }
   652→    if (c === '\\') { escaped = true; i++; continue; }
   653→    if (c === '"') break;
   654→    i++;
   655→  }
   656→  return raw.slice(start, i);
   657→}

   705→  config: ProviderConfig,
   706→  messages: ChatMessage[],
   707→  timeoutMs: number,
   708→  toolsEnabled = true,

   710→  const accessToken = config.minimaxAccessToken;

   712→  logger.info("callMinimax", "Requesting MiniMax", { model: "MiniMax-M2.7", msgCount: messages.length, toolsEnabled });
   713→  const minimaxMessages = messages.map((m) => {
   714→    if (m.role === "tool") {
   715→      let content = m.content ?? "";
   716→      try {
   717→        const data = JSON.parse(content);
   718→        if (data.type === "search" && Array.isArray(data.sources)) {
   719→          const formatted = data.sources
   720→            .map((s: any, i: number) => {
   721→              const text = s.content || s.snippet || "";
   722→              return `${i + 1}. ${s.title} (${s.domain})\n${text}`;
   723→            })
   724→            .filter((s: string) => s.trim().length > 3)
   725→            .join("\n\n");
   726→          content = formatted || `Búsqueda: ${data.title || ""}`;
   727→        } else if (data.type === "weather") {

   729→        }
   730→      } catch {
   731→        // mantener contenido original si no es JSON
   732→      }
   733→      return { role: "user" as const, content: `Resultado de herramienta (${m.tool_call_id ?? "unknown"}):\n${content}` };
   734→    }
   735→    if (m.role === "assistant" && m.tool_calls) {
   736→      return {
   737→        role: "assistant" as const,
   738→        content: m.content ?? `Voy a usar herramientas: ${m.tool_calls.map((t) => t.function.name).join(", ")}`,
   739→      };
   740→    }
   741→    return { role: m.role, content: m.content ?? "" };
   742→  });

   744→    method: "POST",
   745→    headers: {
   746→      Authorization: `Bearer ${accessToken}`,
   747→      "Content-Type": "application/json",
   748→      Accept: "application/json",
   749→    },

   751→      model: "MiniMax-M2.7",
   752→      messages: minimaxMessages,

   754→      temperature: 0.25,
   755→      top_p: 0.95,
   756→      max_tokens: 8192,
   757→      stream: false,
   758→      reasoning_split: true,
   759→    }),
   760→  }, timeoutMs);
   761→  const data = await response.json().catch(() => ({}));
   762→  const choice = asRecord(asArray(asRecord(data).choices)[0]);
   763→  const assistantMsg = asRecord(choice.message);
   764→  const content = asString(assistantMsg.content) ?? "";
   765→  const toolCalls = asArray(assistantMsg.tool_calls);
   766→  const reasoningContent = asString(assistantMsg.reasoning_content) ?? "";
   767→  logger.info("callMinimax", `Response HTTP ${response.status}`, { contentPreview: content.slice(0, 500), reasoningPreview: reasoningContent.slice(0, 200), hasTools: toolCalls.length > 0, usage: dump(data.usage, 300) });
   768→  if (!response.ok || !hasUsableAssistantMessage(data)) {

   770→    throw new Error(`MiniMax returned ${response.status}`);
   771→  }
   772→  return {
   773→    provider: "minimax",
   774→    model: asString(asRecord(data).model) ?? "MiniMax-M2.7",
   775→    message: assistantMsg as ProviderMessage,
   776→  };

   965→  for (const key of keys) {
   966→    try {
   967→      return await callBlueSmindsCandidate(key, model, messages, timeoutMs, toolsEnabled);
   968→    } catch (err) {
   969→      lastError = err instanceof Error ? err : new Error(String(err));
   970→      logger.warn("callBlueSminds", `BlueSminds ${model} failed with a key, trying next`, { reason: lastError.message });
   971→    }
   972→  }
   973→  throw lastError ?? new Error(`BlueSminds ${model} failed with all available keys.`);
   974→}

   989→export function inferProviderFromModel(model: string | undefined): "minimax" | "nvidia" | "openrouter" | "bluesminds" | undefined {
   990→  if (!model) return undefined;
   991→  if (model === "MiniMax-M2.7") return "minimax";
   992→  if (model === "mimo-v2.5") return "bluesminds";
   993→  if (model.startsWith("hf.co/")) return "nvidia"; // HuggingFace models served by Ollama
   994→  if (!model.includes("/")) return "nvidia"; // Ollama tags without namespace, e.g. qwen3.6:27b, llama3.1:8b
   995→  if (model.startsWith("nvidia/") && !model.includes(":")) return "nvidia"; // NVIDIA API models
   996→  if (model.startsWith("stepfun-ai/")) return "nvidia"; // Flash model served by NVIDIA API
   997→  if (model.includes("/") && model.includes(":")) return "openrouter"; // OpenRouter free models (e.g. openai/gpt-oss-120b:free)
   998→  return undefined;
   999→}

  1009→/**
  1010→ * Fase 4.1 — Router de modelos NVIDIA.
  1011→ *
  1012→ * Selecciona el modelo óptimo según la complejidad del input del usuario:
  1013→ * - Trivial (hola, gracias, adios): modelo flash (~1-2s)
  1014→ * - Normal (clima, gasto, lista): modelo mediano (~5-8s)
  1015→ * - Complejo (informe, deep research): modelo ultra (~15s)
  1016→ *
  1017→ * Esto reduce la latencia percibida para interacciones simples sin
  1018→ * sacrificar calidad en tareas complejas.

  1020→function selectModelForInput(
  1021→  input: string,
  1022→  config: ProviderConfig,
  1023→  isTrivial: boolean,
  1024→  isDeliverable: boolean,
  1025→): string | undefined {
  1026→  // Si no hay fast/medium model configurado, usar el default
  1027→  if (!config.nvidiaFastModel && !config.nvidiaMediumModel) return undefined;
  1028→
  1029→  // Trivial → flash (sin tools, respuesta rápida)
  1030→  if (isTrivial && config.nvidiaFastModel) {
  1031→    logger.info("modelRouter", "Using fast model", { model: config.nvidiaFastModel, reason: "trivial input" });
  1032→    return config.nvidiaFastModel;
  1033→  }
  1034→
  1035→  // Deliverable (informe/investigación) → ultra (máxima calidad)
  1036→  if (isDeliverable) {
  1037→    logger.info("modelRouter", "Using ultra model", { model: config.nvidiaModel, reason: "deliverable" });
  1038→    return config.nvidiaModel; // nvidiaModel = ultra por default
  1039→  }
  1040→
  1041→  // Normal → medium si está disponible, sino ultra
  1042→  if (config.nvidiaMediumModel) {
  1043→    logger.info("modelRouter", "Using medium model", { model: config.nvidiaMediumModel, reason: "normal input" });
  1044→    return config.nvidiaMediumModel;
  1045→  }
  1046→
  1047→  return undefined;
  1048→}

  1050→export async function callProvider(
  1051→  config: ProviderConfig,
  1052→  messages: ChatMessage[],
  1053→  timeoutMs: number,
  1054→  toolsEnabled = true,
  1055→  preferredProvider?: "minimax" | "nvidia" | "openrouter" | "bluesminds",
  1056→  availableTools?: ToolDefinition[],
  1057→  modelOverride?: string,
  1058→): Promise<ProviderResult & { fallbackReason?: string }> {
  1059→  const isOllama = isOllamaUrl(config.nvidiaBaseUrl);
  1060→  const nvidiaAvailable = Boolean(config.nvidiaApiKey) || isOllama;
  1061→  const bluesmindsKeys = config.bluesmindsKeys ?? [];
  1062→  const bluesmindsModel = config.bluesmindsModel ?? "mimo-v2.5";
  1063→  const bluesmindsAvailable = bluesmindsKeys.length > 0 && bluesmindsModel.length > 0;
  1064→
  1065→  // SALTO DIRECTO si el usuario eligió un provider específico
  1066→  if (preferredProvider === "bluesminds" && bluesmindsAvailable) {
  1067→    try {
  1068→      return await callBlueSminds(config, messages, Math.min(60_000, timeoutMs), toolsEnabled);
  1069→    } catch (err: any) {
  1070→      if (isRateLimitError(err)) throw err;
  1071→      logger.warn("callProvider", "Preferred BlueSminds failed, falling through", { reason: err?.message });
  1072→    }
  1073→  }
  1074→
  1075→  if (preferredProvider === "minimax" && config.minimaxAccessToken) {
  1076→    try {
  1077→      const result = await callMinimax(config, messages, timeoutMs, toolsEnabled);
  1078→      if (providerResultIsValid(result)) {
  1079→        logger.info("callProvider", "Preferred MiniMax succeeded");
  1080→        return result;
  1081→      }
  1082→      logger.warn("callProvider", "Preferred MiniMax responded but invalid, falling through");
  1083→    } catch (err: any) {
  1084→      logger.warn("callProvider", "Preferred MiniMax failed, falling through", { reason: err?.message });
  1085→    }
  1086→  }
  1087→
  1088→  if (preferredProvider === "nvidia" && nvidiaAvailable) {
  1089→    try {
  1090→      const result = await callNvidia(config, messages, nvidiaTimeoutMs(config, isOllama, timeoutMs), toolsEnabled, availableTools, modelOverride);
  1091→      if (providerResultIsValid(result)) return result;
  1092→      logger.warn("callProvider", "Preferred NVIDIA responded but invalid, falling back");
  1093→    } catch (err: any) {
  1094→      if (isRateLimitError(err)) throw err;
  1095→      logger.warn("callProvider", "Preferred NVIDIA failed, falling back", { reason: err?.message });
  1096→    }
  1097→  }
  1098→
  1099→  if (preferredProvider === "openrouter" && config.openRouterKeys.length) {
  1100→    return callOpenRouter(config, messages, Math.min(115_000, timeoutMs), toolsEnabled);
  1101→  }
  1102→
  1103→  // FLUJO NORMAL (sin preferencia o preferencia fallida)
  1104→  // BlueSminds es el proveedor principal si está configurado.
  1105→  if (bluesmindsAvailable && (!preferredProvider || preferredProvider === "bluesminds")) {
  1106→    try {
  1107→      const result = await callBlueSminds(config, messages, Math.min(60_000, timeoutMs), toolsEnabled);
  1108→      if (providerResultIsValid(result)) {
  1109→        logger.info("callProvider", "BlueSminds succeeded");
  1110→        return result;
  1111→      }
  1112→      logger.warn("callProvider", "BlueSminds responded but invalid, falling through");
  1113→    } catch (err: any) {
  1114→      if (isRateLimitError(err)) throw err;
  1115→      logger.warn("callProvider", "BlueSminds failed, falling through", { reason: err?.message });
  1116→    }
  1117→  }
  1118→
  1119→  // Si el usuario eligió un provider específico que no es MiniMax, no lo intentamos.
  1120→  if (config.minimaxAccessToken && (!preferredProvider || preferredProvider === "minimax")) {
  1121→    try {
  1122→      const result = await callMinimax(config, messages, timeoutMs, toolsEnabled);
  1123→      if (providerResultIsValid(result)) {
  1124→        logger.info("callProvider", "MiniMax succeeded");
  1125→        return result;
  1126→      }
  1127→      logger.warn("callProvider", "MiniMax responded but invalid, falling through");
  1128→    } catch (err: any) {
  1129→      logger.warn("callProvider", "MiniMax failed, falling through", { reason: err?.message });
  1130→    }
  1131→  }
  1132→
  1133→  if (!nvidiaAvailable) {
  1134→    return callOpenRouter(config, messages, Math.min(115_000, timeoutMs), toolsEnabled);
  1135→  }
  1136→
  1137→  // Si el usuario eligió OpenRouter, saltamos NVIDIA en el flujo normal
  1138→  if (!preferredProvider || preferredProvider !== "openrouter") {
  1139→    try {
  1140→      const result = await callNvidia(config, messages, nvidiaTimeoutMs(config, isOllama, timeoutMs), toolsEnabled, availableTools, modelOverride);
  1141→      if (providerResultIsValid(result)) return result;
  1142→      logger.warn("callProvider", "NVIDIA responded but invalid, falling back");
  1143→    } catch (err: any) {
  1144→      if (isRateLimitError(err)) throw err;
  1145→      logger.warn("callProvider", "NVIDIA failed, falling back to OpenRouter", { reason: err?.message });
  1146→    }
  1147→  }
  1148→
  1149→  if (config.openRouterKeys.length) {
  1150→    return callOpenRouter(config, messages, Math.min(115_000, timeoutMs), toolsEnabled);
  1151→  }
  1152→
  1153→  throw new Error("Ningún proveedor de IA respondió. Verificá la conexión o las credenciales.");
  1154→}

  1392→  }
  1393→  return undefined;

  1396→function isValidContentImage(src: string): boolean {
  1397→  if (!src) return false;
  1398→  const lower = src.toLowerCase();
  1399→  if (lower.startsWith("data:")) return false;
  1400→  if (/\b(logo|icon|avatar|sprite|placeholder|blank|spacer|pixel|favicon|tracking|beacon|1x1)\b/i.test(lower)) return false;
  1401→  if (lower.endsWith(".svg") || lower.endsWith(".gif")) return false;
  1402→  if (/(google-analytics|doubleclick|facebook\.com\/tr|pixel\.)/i.test(lower)) return false;
  1403→  if (/[?&](w|h|width|height)=(\d{1,2})\b/i.test(lower)) {
  1404→    const sizeMatch = lower.match(/[?&](w|h|width|height)=(\d{1,2})\b/i);
  1405→    if (sizeMatch && parseInt(sizeMatch[2]) < 100) return false;
  1406→  }
  1407→  return true;
  1408→}

  1410→function resolveImageUrl(src: string, baseUrl: string): string {
  1411→  try {
  1412→    if (/^https?:\/\//i.test(src)) return src;
  1413→    if (src.startsWith("//")) {
  1414→      const proto = baseUrl.match(/^(https?)/i)?.[1] ?? "https";
  1415→      return `${proto}:${src}`;
  1416→    }
  1417→    if (src.startsWith("/")) {
  1418→      const origin = baseUrl.match(/^(https?:\/\/[^/]+)/i)?.[1];
  1419→      if (origin) return `${origin}${src}`;
  1420→    }
  1421→    const base = baseUrl.replace(/[^/]*$/, "");
  1422→    return `${base}${src}`;
  1423→  } catch {
  1424→    return src;
  1425→  }
  1426→}

  1428→async function fetchPageContent(url: string, maxChars = 1200): Promise<{ text: string; imageUrl?: string }> {
  1429→  try {
  1430→    const res = await fetchWithTimeout(url, { headers: { "User-Agent": "Mozilla/5.0 (Windows NT 10.0; Win64; x64)" } }, 15_000);
  1431→    const html = await res.text();
  1432→
  1433→    // 🔴 FIX P2.2: extraer imagen principal
  1434→    const imageUrl = extractMainImage(html, url);
  1435→
  1436→    // Extraer contenido principal: intentar <article>, luego <main>, luego clases comunes
  1437→    let contentHtml = html;
  1438→    const articleMatch = html.match(/<article\b[^>]*>([\s\S]*?)<\/article>/i);
  1439→    if (articleMatch) {
  1440→      contentHtml = articleMatch[1];
  1441→    } else {
  1442→      const mainMatch = html.match(/<main\b[^>]*>([\s\S]*?)<\/main>/i);
  1443→      if (mainMatch) {
  1444→        contentHtml = mainMatch[1];
  1445→      } else {
  1446→        const classMatch = html.match(/<(?:div|section)\b[^>]*\b(?:class|id)="(?:entry-content|article-body|post-content|story-body|content-body|texto-nota|nota-content|article__content|main-content|nota-texto|content-text)[^"]*"[^>]*>([\s\S]*?)<\/\1>/i);
  1447→        if (classMatch) {
  1448→          contentHtml = classMatch[2] ?? classMatch[1];
  1449→        }
  1450→      }
  1451→    }
  1452→
  1453→    // Si no encontramos nada semántico, buscar la zona con más <p> consecutivos
  1454→    if (contentHtml === html) {
  1455→      const pMatches = html.matchAll(/<p\b[^>]*>([\s\S]*?)<\/p>/gi);
  1456→      const paragraphs: string[] = [];
  1457→      for (const m of pMatches) {
  1458→        const text = m[1].replace(/<[^>]+>/g, " ").trim();
  1459→        if (text.length > 40) paragraphs.push(text);
  1460→      }
  1461→      if (paragraphs.length > 0) {
  1462→        contentHtml = paragraphs.slice(0, 8).join(" ");
  1463→      }
  1464→    }
  1465→
  1466→    const text = contentHtml
  1467→      .replace(/<script\b[^<]*(?:(?!<\/script>)<[^<]*)*<\/script>/gi, "")
  1468→      .replace(/<style\b[^<]*(?:(?!<\/style>)<[^<]*)*<\/style>/gi, "")
  1469→      .replace(/<[^>]+>/g, " ")
  1470→      .replace(/\s+/g, " ")
  1471→      .trim();
  1472→    return { text: text.slice(0, maxChars), imageUrl };
  1473→  } catch {
  1474→    return { text: "" };
  1475→  }
  1476→}

  1478→export async function runSearch(

  1645→function isRecordInPeriod(record: LifeRecord, period: string): boolean {
  1646→  const normalized = plainLower(period);
  1647→  if (!normalized || !record.createdAt) return true;
  1648→  const created = new Date(record.createdAt);
  1649→  if (Number.isNaN(created.getTime())) return true;
  1650→  const now = new Date();
  1651→  const ageMs = now.getTime() - created.getTime();
  1652→  if (/\bhoy|today\b/.test(normalized)) return ageMs >= 0 && ageMs <= 36 * 60 * 60 * 1000;
  1653→  if (/\bsemana|week\b/.test(normalized)) return ageMs >= 0 && ageMs <= 8 * 24 * 60 * 60 * 1000;
  1654→  if (/\bmes|month\b/.test(normalized)) return ageMs >= 0 && ageMs <= 32 * 24 * 60 * 60 * 1000;
  1655→  return true;
  1656→}

  1697→function queryTokens(query: string): string[] {
  1698→  return plainLower(query)
  1699→    .replace(/[^\p{L}\p{N}\s:/._-]+/gu, " ")
  1700→    .split(/\s+/)

  1705→function semanticRecordMatches(records: LifeRecord[], query: string, limit = 8): LifeRecord[] {
  1706→  const tokens = queryTokens(query);
  1707→  if (!tokens.length) return [];
  1708→  return records
  1709→    .map((record) => {
  1710→      const haystack = plainLower(recordSearchText(record));
  1711→      const score = tokens.reduce((sum, token) => sum + (haystack.includes(token) ? 1 : 0), 0);
  1712→      return { record, score };
  1713→    })
  1714→    .filter((item) => item.score > 0)
  1715→    .sort((a, b) => b.score - a.score)
  1716→    .map((item) => item.record)
  1717→    .slice(0, limit);
  1718→}
