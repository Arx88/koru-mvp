   264→  const [onboarded, setOnboarded] = useState(() => localStorage.getItem("koru.onboarded") === "true");

   274→  const [showInstallPrompt, setShowInstallPrompt] = useState(false);

   277→  const [phase, setPhase] = useState<string | null>(null);
   278→  const [chatTurns, setChatTurns] = useState<KoruChatTurn[]>(() => readChatTurns(localStorage.getItem("koru.username") ?? ""));
   279→  // Clave versionada (v2): la clave vieja "koru.selected-model" quedaba pegada con
   280→  // modelos de prueba (ej. llama3.1:8b) y pisaba silenciosamente al proveedor

   282→  // elegir un modelo vuelve a ser una decisión explícita del usuario.
   283→  const [selectedModel, setSelectedModel] = useState<string | null>(() => {
   284→    localStorage.removeItem("koru.selected-model");
   285→    return localStorage.getItem("koru.selected-model.v2") ?? null;
   286→  });
   287→  // 🔴 i18n — preferred language. Persisted in localStorage so the LLM and UI honor it.
   288→  const [language, setLanguageState] = useState<"es" | "en">(() => {
   289→    const stored = localStorage.getItem("koru.language");
   290→    return stored === "en" ? "en" : "es";
   291→  });
   292→  // 🔴 Offline cache — track browser online/offline status
   293→  const [online, setOnline] = useState<boolean>(() => isOnline());
   294→  // Ref to latest sendMessage — lets the offline-queue drain call the current closure.
   295→  const sendMessageRef = useRef<((text: string) => Promise<unknown>) | null>(null);
   296→  // 🔴 v2: record reabiertos desde Collections — renderizan el bloque original
   297→  const [reopenedRecord, setReopenedRecord] = useState<LifeRecord | null>(null);
   298→  const domainStateRef = useRef(domainState);
   299→  const chatTurnsRef = useRef(chatTurns);
   300→
   301→  function commitDomainState(next: KoruState | ((prev: KoruState) => KoruState)) {
   302→    setDomainState((prev) => {
   303→      const resolved = typeof next === "function" ? (next as (prev: KoruState) => KoruState)(prev) : next;
   304→      domainStateRef.current = resolved;
   305→      return resolved;
   306→    });
   307→  }
   308→
   309→  function commitChatTurns(next: KoruChatTurn[] | ((prev: KoruChatTurn[]) => KoruChatTurn[])) {
   310→    setChatTurns((prev) => {
   311→      const resolved = typeof next === "function" ? (next as (prev: KoruChatTurn[]) => KoruChatTurn[])(prev) : next;
   312→      chatTurnsRef.current = resolved;
   313→      return resolved;
   314→    });
   315→  }
   316→
   317→  useEffect(() => {
   318→    domainStateRef.current = domainState;
   319→  }, [domainState]);
   320→
   321→  // 🔴 i18n — sync UI language into domainState so the backend LLM honors it
   322→  useEffect(() => {
   323→    commitDomainState((prev) => (prev.language === language ? prev : { ...prev, language }));
   324→  }, [language]);
   325→
   326→  // 🔴 PWA install prompt listener
   327→  useEffect(() => {
   328→    const handler = (e: Event) => {
   329→      e.preventDefault();
   330→      setInstallPromptEvent(e);
   331→      // Show after 30s if not dismissed
   332→      const dismissed = localStorage.getItem("koru.installDismissed");
   333→      if (!dismissed) {
   334→        setTimeout(() => setShowInstallPrompt(true), 30_000);

   336→    };

   338→    return () => window.removeEventListener("beforeinstallprompt", handler);
   339→  }, []);
   340→
   341→  useEffect(() => {
   342→    chatTurnsRef.current = chatTurns;
   343→  }, [chatTurns]);
   344→
   345→  useEffect(() => {
   346→    try {
   347→      const params = new URLSearchParams(window.location.search);
   348→      if (params.get("koruAudit") !== "1" || params.get("reset") !== "1") return;
   349→      const sessionId = `audit_${Date.now().toString(36)}_${Math.random().toString(36).slice(2, 8)}`;
   350→      sessionStorage.setItem("koru.audit.enabled", "1");
   351→      sessionStorage.setItem("koru.audit.session", sessionId);
   352→      localStorage.removeItem(CHAT_STORAGE_KEY);
   353→      localStorage.removeItem("koru.onboarded");
   354→      localStorage.removeItem("koru.username");
   355→      const fresh = resetState();
   356→      commitDomainState(fresh);
   357→      setOnboarded(false);
   358→      setUserName("");
   359→      commitChatTurns([greetingTurn()]);
   360→      params.delete("reset");
   361→      const cleanQuery = params.toString();
   362→      const cleanUrl = `${window.location.pathname}${cleanQuery ? `?${cleanQuery}` : ""}${window.location.hash}`;
   363→      window.history.replaceState(null, "", cleanUrl);
   364→      writeAuditEvent({
   365→        type: "audit_reset",
   366→        note: "Koru browser memory, onboarding and chat reset for manual audit.",
   367→        state: auditStateSnapshot(fresh),
   368→      });
   369→    } catch {
   370→      // Audit reset is best-effort and should not block the app.
   371→    }
   372→  }, []);
   373→
   374→  useEffect(() => {
   375→    let cancelled = false;
   376→    void loadPersistedState().then(async (persisted) => {
   377→      if (!cancelled) {
   378→        const nudges = buildHeartbeatNudges(persisted);
   379→        commitDomainState(nudges.length > 0 ? applyHeartbeatNudges(persisted, nudges) : persisted);
   380→        // 🔴 Sincronizar reminders con notificaciones del navegador
   381→        const userId = persisted.userId ?? "default";
   382→        syncScheduledReminders(persisted.commitments ?? [], userId);
   383→        // Check overdue reminders on app load
   384→        checkDueReminders(userId);
   385→
   386→        // 🔴 Morning Brief: si es mañana y no se mostró hoy, generar
   387→        const now = new Date();
   388→        const hour = now.getHours();
   389→        const today = now.toISOString().slice(0, 10);
   390→        const lastBriefDate = localStorage.getItem("koru.lastBriefDate");
   391→        if (hour >= 5 && hour <= 11 && lastBriefDate !== today) {
   392→          try {
   393→            const stateForBrief = {
   394→              memories: (persisted.memories ?? []).filter((m: any) => m.status === "confirmed" || m.status === "candidate").slice(0, 10).map((m: any) => ({ kind: m.kind, text: m.text })),
   395→              commitments: (persisted.commitments ?? []).filter((c: any) => c.status === "open").slice(0, 5).map((c: any) => ({ title: c.title, dueHint: c.dueHint, dueAt: c.dueAt })),
   396→              userName: persisted.userName ?? "",
   397→              lastBriefDate,
   398→            };
   399→            const res = await fetch("/api/koru/morning-brief", {
   400→              method: "POST",
   401→              headers: { "Content-Type": "application/json" },
   402→              body: JSON.stringify({ state: stateForBrief }),
   403→            });
   404→            if (res.ok) {
   405→              const data = await res.json();
   406→              if (data.shouldShow && data.brief) {
   407→                setMorningBrief(data.brief);
   408→                localStorage.setItem("koru.lastBriefDate", data.date ?? today);
   409→              }
   410→            }
   411→          } catch {
   412→            // silent — brief is optional
   413→          }
   414→        }
   415→      }
   416→    });
   417→    return () => { cancelled = true; };
   418→  }, []);
   419→
   420→  useEffect(() => {
   421→    const runHeartbeat = () => {
   422→      // 🔴 Proactividad: check due reminders y disparar notificaciones
   423→      const userId = domainStateRef.current?.userId ?? "default";
   424→      const { fired, overdue } = checkDueReminders(userId);
   425→      if (fired.length > 0 || overdue.length > 0) {
   426→        console.log("[Koru] Reminders fired:", fired.length, "overdue:", overdue.length);
   427→      }
   428→
   429→      commitDomainState((prev) => {
   430→        const nudges = buildHeartbeatNudges(prev);
   431→        return nudges.length > 0 ? applyHeartbeatNudges(prev, nudges) : prev;
   432→      });
   433→
   434→      // 🔴 MENSAJES PROACTIVOS EN EL CHAT
   435→      // Si hay nudges nuevos, inyectarlos como mensajes de Koru en el chat.
   436→      // Esto hace que Koru "hable" proactivamente, no solo muestre items invisibles.
   437→      const state = domainStateRef.current;
   438→      if (state && !state.ephemeralMode) {
   439→        const activeNudges = (state.nudges ?? []).filter(n =>
   440→          !n.dismissed && !n.title.startsWith("[proactive_shown]")
   441→        );
   442→        if (activeNudges.length > 0) {
   443→          // Solo inyectar 1 mensaje proactivo por heartbeat tick (no spamear)
   444→          const nudge = activeNudges[0];
   445→          const proactiveTurn: KoruChatTurn = {
   446→            id: `proactive_hb_${Date.now()}`,
   447→            role: "koru",
   448→            text: nudge.title + (nudge.body ? `\n${nudge.body}` : ""),
   449→            createdAt: new Date().toISOString(),
   450→            status: "done" as const,
   451→            mascotState: "happy" as const,
   452→          };
   453→          // Marcar como mostrado para no repetir

   970→        if (chunk.stateEvents?.length) {
   971→          const last = chunk.stateEvents[chunk.stateEvents.length - 1];
   972→          setPhase(last.kind === "done" ? null : last.kind);

   974→        const blocksToItems = (blocks: UiBlock[]): KoruTurnItem[] =>
   975→          blocks.map((block) => ({
   976→            id: createId("item"),

   983→        if (!koruTurnId) {

   991→            status: "working" as const,
   992→            mascotState: chunk.mascotState ?? "working",
   993→          };
   994→          (koruTurn.items ?? []).forEach((it) => { if (it.uiBlock?.type) streamedItemsByType.set(it.uiBlock.type, it); });
   995→          commitChatTurns((prev) => [...prev, koruTurn].slice(-120));
   996→        } else {
   997→          const isDone = chunk.stateEvents?.some((e) => e.kind === "done");
   998→          commitChatTurns((prev) =>
   999→            prev.map((turn) => {
  1000→              if (turn.id !== koruTurnId) return turn;
  1001→              const existingItems = turn.items ?? [];
  1002→              const newBlocks = chunk.uiBlocks ?? [];
  1003→              // Merge por TIPO de block (no por índice). Así, si el orden cambia
  1004→              // entre chunks (ej: aparece un data_card antes que el web_nav),
  1005→              // cada item mantiene su identidad y estado (animación de progreso,
  1006→              // contadores internos). Antes se matcheaba por idx, lo que hacía
  1007→              // que un cambio de orden desmontara el componente y reiniciara la
  1008→              // animación ("Iniciando búsqueda..." colgado).
  1009→              const consumedExisting = new Set<string>();
  1010→              const mergedItems: KoruTurnItem[] = newBlocks.map((block) => {
  1011→                const existing = existingItems.find(
  1012→                  (it) => it.uiBlock?.type === block.type && !consumedExisting.has(it.uiBlock!.type),
  1013→                );
  1014→                if (existing) {
  1015→                  consumedExisting.add(existing.uiBlock!.type);
  1016→                  return { ...existing, uiBlock: block, text: (block as any).title ?? (block as any).query ?? "" };
  1017→                }

  1019→                  id: createId("item"),
  1020→                  kind: "action" as const,
  1021→                  tag: block.type,
  1022→                  text: (block as any).title ?? (block as any).query ?? "",
  1023→                  status: "working" as const,
  1024→                  uiBlock: block,
  1025→                };
  1026→              });
  1027→              mergedItems.forEach((it) => { if (it.uiBlock?.type) streamedItemsByType.set(it.uiBlock.type, it); });
  1028→              return {
  1029→                ...turn,
  1030→                text: chunk.reply,
  1031→                items: mergedItems,
  1032→                mascotState: chunk.mascotState ?? turn.mascotState,
  1033→                status: isDone ? ("done" as const) : ("working" as const),
  1034→              };
  1035→            }),
  1036→          );
  1037→        }
  1038→      });
  1039→      const result = applyBackendTurnToState(previousState, text, transcriptSource, agentResult);
  1040→      commitDomainState(result.state);
  1041→

  1091→          if (data?.type === "personal_capture" && Array.isArray(data.memoryCandidates)) return data.memoryCandidates;
  1092→          return [];

  1113→              setMemoryToast((current) => (current?.text === first.text ? null : current));
  1114→            }, 6000);

  1116→        }
  1117→      }
  1118→      writeAuditEvent({
  1119→        type: "turn_analyzed",
  1120→        engine: "backend_agent_loop",
  1121→        input: text,
  1122→        transcriptSource,
  1123→        historyUsed: domainHistory,
  1124→        response: agentResult.reply,

  1131→        items: auditTurnItems(result.items),

  1136→    } finally {
  1137→      setProcessing(false);
  1138→      setActivity(null);
  1139→      setPhase(null);

  1286→  function runReadonlyWebAction(original: AssistantAction): KoruTurnItem {
  1287→    // Dato en vivo (clima/trafico/mercado) = quick, sin panel largo. Un
  1288→    // informe/investigacion real (webMode "research", o web_research sin
  1289→    // modo puntual) = deep, se gana el panel "Trabajando...". Ver
  1290→    // flujo-informe-aoe2.html pasos 3 (informe) vs 7 (clima).
  1291→    const webMode = original.payload.webMode;
  1292→    const isQuickSignal = webMode === "weather" || webMode === "traffic" || webMode === "market";
  1293→    const isDeepResearch = webMode === "research" || (original.kind === "web_research" && !webMode);
  1294→    setActivity({
  1295→      kind: "searching",
  1296→      depth: isQuickSignal ? "quick" : isDeepResearch ? "deep" : "quick",
  1297→      label: isDeepResearch
  1298→        ? "Es un tema para investigar a fondo. Dame unos segundos."
  1299→        : original.kind === "world_signal"
  1300→          ? "Busco senales reales."
  1301→          : "Consulto fuentes reales.",
  1302→    });
  1303→    const now = new Date().toISOString();
  1304→    const approvedAction: AssistantAction = {
  1305→      ...original,
  1306→      status: "approved",
  1307→      updatedAt: now,
  1308→    };
  1309→    commitDomainState((prev) => {
  1310→      const exists = prev.actions.some((action) => action.id === original.id);
  1311→      const next = {
  1312→        ...prev,
  1313→        actions: exists
  1314→          ? prev.actions.map((action) => (action.id === original.id ? approvedAction : action))
  1315→          : [approvedAction, ...prev.actions],
  1316→        updatedAt: now,
  1317→      };
  1318→      saveState(next);
  1319→      return next;
  1320→    });
  1321→    const waitingItem = actionToTurnItem(approvedAction);
  1322→    replaceActionInChat(waitingItem);
  1323→
  1324→    void (async () => {
  1325→      try {
  1326→        const result = await runWebNavigation(original.payload);
  1327→        const webPayload = webResultToPayload(result);
  1328→        const nextUiBlock = patchUiBlockWithWebResult(approvedAction, webPayload);
  1329→        const finishedAt = new Date().toISOString();
  1330→        const patchedAction: AssistantAction = {
  1331→          ...approvedAction,
  1332→          payload: {
  1333→            ...approvedAction.payload,
  1334→            ...webPayload,
  1335→            uiBlock: nextUiBlock ?? approvedAction.payload.uiBlock,
  1336→          },
  1337→          result: result.recommendation,
  1338→          status: "executed",
  1339→          updatedAt: finishedAt,
  1340→          executedAt: finishedAt,
  1341→        };
  1342→        const patchedItem = actionToTurnItem(patchedAction);
  1343→        commitDomainState((prev) => {
  1344→          const next = {
  1345→            ...prev,
  1346→            actions: prev.actions.map((action) => (action.id === original.id ? patchedAction : action)),
  1347→            updatedAt: finishedAt,
  1348→          };
  1349→          saveState(next);
  1350→          writeAuditEvent({
  1351→            type: "readonly_web_action_finished",
  1352→            action: {
  1353→              id: patchedAction.id,
  1354→              kind: patchedAction.kind,
  1355→              title: patchedAction.title,
  1356→              status: patchedAction.status,
  1357→              result: patchedAction.result,
  1358→              payload: patchedAction.payload,
  1359→            },
  1360→            item: auditTurnItems([patchedItem])[0],
  1361→            state: auditStateSnapshot(next),
  1362→            delta: auditStateDelta(prev, next),
  1363→          });
  1364→          return next;
  1365→        });
  1366→        replaceActionInChat(patchedItem);
  1367→      } finally {
  1368→        setActivity(null);
  1369→      }
  1370→    })();
  1371→    return waitingItem;
  1372→  }

  1374→  function reviewAction(id: string, approve: boolean): KoruTurnItem | null {
  1375→    const currentState = domainStateRef.current;
  1376→    const original = currentState.actions.find((action) => action.id === id);
  1377→    if (approve && (original?.kind === "web_research" || original?.kind === "world_signal")) {
  1378→      return runReadonlyWebAction(original);

  1380→
  1381→    const nextState = approve ? approveAndExecuteAction(currentState, id) : rejectAction(currentState, id);
  1382→    const reviewed = nextState.actions.find((action) => action.id === id);

  1384→    commitDomainState(nextState);

  1390→      approved: approve,
  1391→      item: auditTurnItems([item])[0],
  1392→      state: auditStateSnapshot(nextState),
  1393→      delta: auditStateDelta(currentState, nextState),
  1394→    });
  1395→    appendKoruConfirmation(approve ? actionConfirmationText(item) : "Lo suelto. No hice nada con eso.");
  1396→
  1397→    return item;
  1398→  }

  1462→    // 🔴 GAP-2: el motor de decisión (computeDecision) se invoca acá,
  1463→    // dentro del wrapper, para que la decisión persistida quede con su
  1464→    // `result` ya poblado (recommendation + confidenceInterval + scores).
  1465→    // El reducer original (createDecisionReducer) sólo persiste la decisión
  1466→    // "en crudo"; nosotros construimos el objeto, lo computamos, y luego
  1467→    // commiteamos el estado resultante. Devolvemos la Decision al caller
  1468→    // (CreateScreen) para que pueda invocar computeDecision explícitamente
  1469→    // sobre la decisión recién creada — cumpliendo el contrato GAP-2.
  1470→    const now = new Date().toISOString();
  1471→    const decision: Decision = {
  1472→      id: createId("decision"),
  1473→      question,
  1474→      options,
  1475→      factors,
  1476→      weights,
  1477→      linkedMemoryIds: [],
  1478→      createdAt: now,
  1479→    };
  1480→    const prev = domainStateRef.current;
  1481→    const nextState: KoruState = {
  1482→      ...prev,
  1483→      decisions: [decision, ...(prev.decisions ?? [])],
  1484→      updatedAt: now,
  1485→    };
  1486→    // El motor muta `decision.result` in-place.
  1487→    try {
  1488→      computeDecision(nextState, decision.id);
  1489→    } catch {
  1490→      // best-effort: si el motor falla (ej. opciones vacías), la decisión
  1491→      // se persiste igual sin result — el usuario puede re-computar luego.
  1492→    }
  1493→    saveState(nextState);
  1494→    commitDomainState(nextState);
  1495→    return decision;
  1496→  }
