    88→function escapeHtml(s: string): string {
    89→  return s.replace(/&/g, "&amp;").replace(/</g, "&lt;").replace(/>/g, "&gt;");
    90→}
    91→
    92→function inlineMd(s: string): string {
    93→  // s ya está escapado
    94→  s = s.replace(/\[([^\]]+)\]\(([^)]+)\)/g, '<a href="$2" target="_blank" rel="noopener noreferrer">$1</a>');
    95→  s = s.replace(/\*\*([^*]+)\*\*/g, "<strong>$1</strong>");
    96→  s = s.replace(/\*([^*]+)\*/g, "<em>$1</em>");
    97→  return s;
    98→}

   100→function renderMarkdown(text: string): string {
   101→  if (!text.trim()) return "";
   102→  const lines = text.split("\n");
   103→  const out: string[] = [];
   104→  let inList = false;
   105→  let para: string[] = [];
   106→  const flushPara = () => {
   107→    if (para.length > 0) {
   108→      out.push(`<p>${para.map(p => inlineMd(escapeHtml(p))).join("<br />")}</p>`);
   109→      para = [];
   110→    }
   111→  };
   112→  for (const line of lines) {
   113→    const listMatch = line.match(/^\s*-\s+(.*)$/);
   114→    if (listMatch) {
   115→      flushPara();
   116→      if (!inList) { out.push("<ul>"); inList = true; }
   117→      out.push(`<li>${inlineMd(escapeHtml(listMatch[1]))}</li>`);
   118→      continue;
   119→    }
   120→    if (inList) { out.push("</ul>"); inList = false; }
   121→    para.push(line);
   122→  }
   123→  flushPara();
   124→  if (inList) out.push("</ul>");
   125→  return out.join("");
   126→}

   408→        title: title.trim() || "Rutina sin título",
   409→        collection: finalCollection,
   410→        notes: rutinaText,
   411→        kind,
   412→        sourceBlock: buildSavedRecordBlock({
   413→          title: title.trim() || "Rutina",
   414→          collection: finalCollection,
   415→          notes: rutinaText,
   416→          kind,
   417→        }),
   418→      });
   419→      return;
   420→    }
   421→
   422→    if (selected === "ejercicio") {
   423→      const weeks = parseInt(weeksTotal) || 0;
   424→      const sessionsText = sessions
   425→        .filter(s => s.dayLabel.trim() || s.exercises.some(e => e.name.trim()))
   426→        .map(s => {
   427→          const ex = s.exercises.filter(e => e.name.trim());
   428→          const exText = ex
   429→            .map(e => `  - ${e.name}: ${e.sets || 0}x${e.reps || 0}${e.weight ? ` @ ${e.weight}kg` : ""}`)
   430→            .join("\n");
   431→          return `${s.dayLabel || "Día"}:\n${exText}`;
   432→        })

   442→            .filter(e => e.name.trim())
   443→            .map(e => ({
   444→              exercise: e.name,
   445→              sets: parseInt(e.sets) || 0,
   446→              reps: parseInt(e.reps) || 0,

   458→        title: title.trim() || "Plan de entrenamiento",
   459→        collection: finalCollection,
   460→        notes: ejText,
   461→        kind,
   462→        sourceBlock: buildSavedRecordBlock({
   463→          title: title.trim() || "Plan de entrenamiento",
   464→          collection: finalCollection,
   465→          notes: ejText,
   466→          kind,
   467→        }),
   468→      });
   469→      return;
   470→    }
   471→
   472→    if (selected === "memoria") {
   473→      const memText = [
   474→        memoriaText.trim(),
   475→        `Tipo: ${memoriaKind}`,
   476→        `Confianza: ${Math.round(memoriaConfidence * 100)}%`,
   477→        memoriaSensitive ? "Sensibilidad: sensible" : null,
   478→      ].filter(Boolean).join("\n");
   479→      await persist({
   480→        title: title.trim() || "Memoria",
   481→        collection: finalCollection,
   482→        notes: memText,
   483→        kind,
   484→        sourceBlock: buildSavedRecordBlock({
   485→          title: title.trim() || "Memoria",
   486→          collection: finalCollection,
   487→          notes: memText,
   488→          kind,
   489→        }),
   490→      });
   491→      return;
   492→    }
   493→
   494→    if (selected === "decision") {
   495→      const facIds = factors.map(f => f.id);
   496→      const decText = [
   497→        deadline ? `Deadline: ${deadline}` : null,
   498→        "Factores:",
   499→        ...factors.filter(f => f.label.trim()).map(f => `  • ${f.label} (${f.direction === "higherIsBetter" ? "↑" : "↓"})`),
   500→        "Opciones:",
   501→        ...options.filter(o => o.label.trim()).map(o => `  • ${o.label}`),
   502→      ].filter(Boolean).join("\n");
   503→      // Embed structured payload (factor scores) as a JSON footer for later reopening.
   504→      const optionsWithScores = options.filter(o => o.label.trim()).map(o => ({
   505→        label: o.label,
   506→        factorScores: Object.fromEntries(facIds.map(id => [id, o.factorScores[id] ?? 0])),
   507→      }));
   508→      const structured = JSON.stringify({ weeks: null, factors: factors.filter(f => f.label.trim()), options: optionsWithScores, deadline: deadline || undefined });
   509→      const fullNotes = `${decText}\n\n---\n${structured}`;
   510→      // 🔴 GAP-2: además del LifeRecord, persistimos una Decision durable en
   511→      // el store y disparamos el motor (computeDecision) para poblar el
   512→      // resultado: recommendation, perOptionScore, perOptionProbability y
   513→      // confidenceInterval. El wrapper createDecision ejecuta computeDecision
   514→      // internamente y devuelve la Decision ya persistida con .result.
   515→      const decisionOptions: DecisionOption[] = options
   516→        .filter(o => o.label.trim())
   517→        .map(o => ({
   518→          id: o.id,
   519→          label: o.label,
   520→          factorScores: Object.fromEntries(facIds.map(id => [id, o.factorScores[id] ?? 0])),
   521→        }));
   522→      const decisionFactors: DecisionFactor[] = factors
   523→        .filter(f => f.label.trim())
   524→        .map(f => ({ id: f.id, label: f.label, direction: f.direction }));
   525→      // Pesos iguales (1.0) por factor — el motor normaliza internamente.
   526→      // Si no hay factores, weights queda vacío y el motor reparte probabilidad
   527→      // uniforme entre las opciones.
   528→      const weights: Record<string, number> = Object.fromEntries(
   529→        decisionFactors.map(f => [f.id, 1]),
   530→      );
   531→      if (decisionOptions.length > 0) {
   532→        const newDecision: Decision = createDecision(
   533→          title.trim() || "Decisión",
   534→          decisionOptions,
   535→          decisionFactors,
   536→          weights,
   537→        );
   538→        // 🔴 GAP-2 (contrato): invocamos computeDecision explícitamente desde
   539→        // la UI, sobre la decisión recién creada, para garantizar el poblado
   540→        // del resultado. Construimos un estado sintético porque `state` (de
   541→        // useKoru) aún no refleja la decisión nueva en este render cycle.
   542→        const syntheticState: KoruState = {
   543→          ...state,
   544→          decisions: [newDecision, ...(state.decisions ?? [])],
   545→        };
   546→        try {
   547→          computeDecision(syntheticState, newDecision.id);
   548→        } catch {
   549→          // best-effort: la decisión ya quedó persistida con .result por el wrapper.
   550→        }
   551→      }
   552→      await persist({
   553→        title: title.trim() || "Decisión",
   554→        collection: finalCollection,
   555→        notes: fullNotes,
   556→        kind,
   557→        sourceBlock: buildSavedRecordBlock({
   558→          title: title.trim() || "Decisión",
   559→          collection: finalCollection,
   560→          notes: fullNotes,
   561→          kind,
   562→        }),
   563→      });
   564→      return;
   565→    }
   566→
   567→    // nota (default)
   568→    await persist({
   569→      title: title.trim() || "Nota sin título",
   570→      collection: finalCollection,
   571→      notes: notes.trim() || undefined,
   572→      kind,
   573→    });
   574→  }

   651→  function insertWrap(wrap: string, placeholder: string) {
   652→    const ta = notesRef.current;
   653→    if (!ta) return;
   654→    const start = ta.selectionStart;
   655→    const end = ta.selectionEnd;
   656→    const sel = notes.slice(start, end) || placeholder;
   657→    const next = notes.slice(0, start) + wrap + sel + wrap + notes.slice(end);
   658→    setNotes(next);
   659→    requestAnimationFrame(() => {
   660→      ta.focus();
   661→      const pos = start + wrap.length;
   662→      ta.setSelectionRange(pos, pos + sel.length);
   663→    });
   664→  }

   666→  function insertPrefix(prefix: string) {
   667→    const ta = notesRef.current;
   668→    if (!ta) return;
   669→    const start = ta.selectionStart;
   670→    const before = notes.slice(0, start);
   671→    const lineStart = before.lastIndexOf("\n") + 1;
   672→    const next = notes.slice(0, lineStart) + prefix + notes.slice(lineStart);
   673→    setNotes(next);
   674→    requestAnimationFrame(() => {
   675→      ta.focus();
   676→      const pos = start + prefix.length;
   677→      ta.setSelectionRange(pos, pos);
   678→    });
   679→  }
   680→
   681→  function insertLink() {
   682→    const ta = notesRef.current;
   683→    if (!ta) return;
   684→    const start = ta.selectionStart;
   685→    const end = ta.selectionEnd;
   686→    const sel = notes.slice(start, end) || "texto";
   687→    const link = `[${sel}](https://)`;
   688→    const next = notes.slice(0, start) + link + notes.slice(end);
   689→    setNotes(next);
   690→    requestAnimationFrame(() => {
   691→      ta.focus();
   692→      const urlStart = start + sel.length + 3; // "[" + sel + "]("
   693→      ta.setSelectionRange(urlStart, urlStart + 8); // selecciona "https://"
   694→    });
   695→  }
