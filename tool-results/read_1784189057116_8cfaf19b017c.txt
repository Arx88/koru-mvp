    55→/**
    56→ * Detecta si el query menciona una selección nacional y devuelve el canonical name.
    57→ * Esto permite que "como salio España" → match exacto con "Spain" en ESPN.
    58→ */
    59→function detectNationalTeam(queryLower: string): string | null {
    60→  for (const team of NATIONAL_TEAM_SYNONYMS) {
    61→    for (const alias of team.aliases) {
    62→      // Match exacto de palabra (para no confundir "España" con "español")
    63→      const re = new RegExp(`\\b${alias.replace(/[.*+?^${}()|[\\]\\\\]/g, "\\\\$&")}\\b`, "i");
    64→      if (re.test(queryLower)) return team.canonical;
    65→    }
    66→  }
    67→  return null;
    68→}

    70→/**
    71→ * Calcula fechas a buscar basado en el query del usuario.
    72→ * "ayer" → fecha de ayer.
    73→ * "hoy" / "en vivo" → fecha de hoy.
    74→ * "mañana" → fecha de mañana.
    75→ * Sin indicador → busca hoy + ayer (3 días de ventana) para captar últimos resultados.
    76→ */
    77→function detectDatesToQuery(queryLower: string): string[] {
    78→  const dates: string[] = [];
    79→  const now = new Date();
    80→  const fmt = (d: Date) => d.toISOString().slice(0, 10).replace(/-/g, "");
    81→
    82→  if (/\bayer\b|\bd'?ayer\b|last match|último partido|ultimo partido/.test(queryLower)) {
    83→    const y = new Date(now.getTime() - 24 * 60 * 60 * 1000);
    84→    dates.push(fmt(y));
    85→    // También 2 días atrás por si el partido fue de madrugada
    86→    const y2 = new Date(now.getTime() - 48 * 60 * 60 * 1000);
    87→    dates.push(fmt(y2));
    88→  } else if (/\bhoy\b|\btoday\b|\ben vivo\b|\blive\b/.test(queryLower)) {
    89→    dates.push(fmt(now));
    90→  } else if (/\bmañana\b|\bmanana\b|\btomorrow\b/.test(queryLower)) {
    91→    const t = new Date(now.getTime() + 24 * 60 * 60 * 1000);
    92→    dates.push(fmt(t));
    93→  } else {
    94→    // Sin indicador: ventana de 3 días (ayer + hoy + anteayer) para captar último resultado
    95→    dates.push(fmt(now));
    96→    const y = new Date(now.getTime() - 24 * 60 * 60 * 1000);
    97→    dates.push(fmt(y));
    98→    const y2 = new Date(now.getTime() - 48 * 60 * 60 * 1000);
    99→    dates.push(fmt(y2));
   100→  }
   101→  return [...new Set(dates)];
   102→}

   164→async function searchEspnScoreboards(query: string): Promise<Array<{ event: EspnEvent; leagueId: string; leagueName: string }>> {
   165→  const queryLower = query.toLowerCase();
   166→  const results: Array<{ event: EspnEvent; leagueId: string; leagueName: string }> = [];
   167→  const datesToQuery = detectDatesToQuery(queryLower);
   168→  const nationalTeam = detectNationalTeam(queryLower);
   169→
   170→  // Si detectamos selección nacional, agregamos el canonical name al query
   171→  // para que el filtro por nombre funcione (España → "Spain" en ESPN)
   172→  const matchTerms = [queryLower];
   173→  if (nationalTeam) matchTerms.push(nationalTeam.toLowerCase());
   174→
   175→  // Buscar en paralelo en todas las ligas × todas las fechas relevantes
   176→  const promises: Promise<void>[] = [];
   177→  for (const league of ESPN_LEAGUES) {
   178→    for (const date of datesToQuery) {
   179→      promises.push((async () => {
   180→        try {
   181→          const url = `${ESPN_BASE}/${league.id}/scoreboard?dates=${date}`;
   182→          const res = await fetch(url, { signal: AbortSignal.timeout(6000) });
   183→          if (!res.ok) return;
   184→          const data = await res.json() as { events?: EspnEvent[] };
   185→          const events = data.events ?? [];
   186→          const matching = events.filter(e => {
   187→            const eventName = (e.name ?? "").toLowerCase();
   188→            const comps = e.competitions ?? [];
   189→            const teams = comps.flatMap(c => (c.competitors ?? []).map(comp => comp.team?.displayName?.toLowerCase() ?? ""));
   190→            return matchTerms.some(term =>
   191→              eventName.includes(term) || teams.some(t => t.includes(term))
   192→            );
   193→          });
   194→          for (const m of matching) {
   195→            results.push({ event: m, leagueId: league.id, leagueName: league.name });
   196→          }
   197→        } catch { /* league/date timeout — skip */ }
   198→      })());
   199→    }
   200→  }
   201→  await Promise.all(promises);
   202→
   203→  // Dedupe por id (mismo evento puede aparecer en múltiples fechas)
   204→  const seen = new Set<string>();
   205→  const deduped = results.filter(({ event: e }) => {
   206→    const id = e.id ?? `${e.name ?? ""}-${e.date ?? ""}`;
   207→    if (seen.has(id)) return false;
   208→    seen.add(id);
   209→    return true;
   210→  });
   211→  return deduped;
   212→}

   214→function normalizeEspnEvent(e: EspnEvent, leagueName?: string) {
   215→  const comps = e.competitions ?? [];
   216→  const comp = comps[0];
   217→  const competitors = comp?.competitors ?? [];
   218→  const home = competitors.find(c => c.homeAway === "home");
   219→  const away = competitors.find(c => c.homeAway === "away");
   220→  const status = e.status?.type?.description ?? e.status?.type?.detail ?? "?";
   221→  const venue = comp?.venue;
   222→  return {
   223→    id: e.id,
   224→    eventId: e.id,
   225→    match: `${home?.team?.displayName ?? "?"} vs ${away?.team?.displayName ?? "?"}`,
   226→    homeTeam: home?.team?.displayName,
   227→    awayTeam: away?.team?.displayName,
   228→    homeShortName: home?.team?.shortDisplayName,
   229→    awayShortName: away?.team?.shortDisplayName,
   230→    homeLogo: home?.team?.logo,
   231→    awayLogo: away?.team?.logo,
   232→    homeColor: home?.team?.color ? `#${home.team.color}` : undefined,
   233→    awayColor: away?.team?.color ? `#${away.team.color}` : undefined,
   234→    homeAbbrev: home?.team?.abbreviation,
   235→    awayAbbrev: away?.team?.abbreviation,
   236→    homeScore: home?.score != null ? Number(home.score) : undefined,
   237→    awayScore: away?.score != null ? Number(away.score) : undefined,
   238→    status,
   239→    state: e.status?.type?.state, // "pre" | "in" | "post"
   240→    date: comp?.date ?? e.date,

   242→    league: leagueName,
   243→    venue: venue?.fullName,
   244→    venueCity: venue?.address?.city,
   245→    venueCountry: venue?.address?.country,

   263→// 🔴 Extrae goles, tarjetas y sustituciones de keyEvents, agrupados por tipo.
   264→// ESPN pone el texto del evento en `text`, con el nombre del jugador entre paréntesis.
   265→// Ej: "Goal! Spain 1, England 0. Nico Williams (Spain) left footed shot..."
   266→function parseKeyEvents(summary: EspnSummary | null) {
   267→  const events = summary?.keyEvents ?? [];
   268→  const goals: Array<{ minute: string; team?: string; scorer?: string; text?: string }> = [];
   269→  const yellowCards: Array<{ minute: string; team?: string; player?: string }> = [];
   270→  const redCards: Array<{ minute: string; team?: string; player?: string }> = [];
   271→  const substitutions: Array<{ minute: string; team?: string; playerIn?: string; playerOut?: string }> = [];
   272→
   273→  for (const ev of events) {
   274→    const minute = ev.clock?.displayValue ?? "";
   275→    const team = ev.team?.displayName;
   276→    const text = ev.text ?? "";
   277→    const athlete = ev.athlete?.displayName;
   278→    const typeText = ev.type?.text ?? "";
   279→
   280→    if (typeText === "Goal") {
   281→      // Texto: "Goal! Spain 1, England 0. Nico Williams (Spain) left footed shot..."
   282→      // El scorer suele estar justo después de "Goal! TeamA X, TeamB Y."
   283→      let scorer = athlete;
   284→      if (!scorer) {
   285→        // Buscar patrón "FirstName LastName (Team)" en el texto
   286→        const m = text.match(/(?:Goal!.*?\.\s+)?([A-ZÁÉÍÓÚÑ][a-záéíóúñ]+(?:\s+[A-ZÁÉÍÓÚÑ][a-záéíóúñ]+)*)\s*\(/);
   287→        if (m) scorer = m[1];
   288→      }
   289→      goals.push({ minute, team, scorer, text });
   290→    } else if (typeText === "Yellow Card") {
   291→      let player = athlete;
   292→      if (!player) {
   293→        const m = text.match(/^([A-ZÁÉÍÓÚÑ][a-záéíóúñ]+(?:\s+[A-ZÁÉÍÓÚÑ][a-záéíóúñ]+)*)\s*\(/);
   294→        if (m) player = m[1];
   295→      }
   296→      yellowCards.push({ minute, team, player });
   297→    } else if (typeText === "Red Card") {
   298→      let player = athlete;
   299→      if (!player) {
   300→        const m = text.match(/^([A-ZÁÉÍÓÚÑ][a-záéíóúñ]+(?:\s+[A-ZÁÉÍÓÚÑ][a-záéíóúñ]+)*)\s*\(/);
   301→        if (m) player = m[1];
   302→      }
   303→      redCards.push({ minute, team, player });
   304→    } else if (typeText === "Substitution") {
   305→      // Texto: "Substitution, Spain. Martín Zubimendi replaces Rodri because of an injury."
   306→      const m = text.match(/([A-ZÁÉÍÓÚÑa-záéíóúñ][\wáéíóúñ]+)\s+replaces\s+([A-ZÁÉÍÓÚÑa-záéíóúñ][\wáéíóúñ]+)/);
   307→      substitutions.push({
   308→        minute,
   309→        team,
   310→        playerIn: m?.[1],
   311→        playerOut: m?.[2],
   312→      });
   313→    }
   314→  }
   315→
   316→  return { goals, yellowCards, redCards, substitutions };
   317→}

   319→// 🔴 Extrae alineaciones (formation + titulares + suplentes) por equipo.
   320→function parseRosters(summary: EspnSummary | null) {
   321→  const rosters = summary?.rosters ?? [];
   322→  const byTeam: Record<string, { formation?: string; starters: Array<{ number?: string; name: string; position?: string }>; subs: Array<{ number?: string; name: string; position?: string }> }> = {};
   323→
   324→  for (const r of rosters) {
   325→    const teamName = r.team?.displayName ?? "?";
   326→    const formation = r.formation;
   327→    const starters: Array<{ number?: string; name: string; position?: string }> = [];
   328→    const subs: Array<{ number?: string; name: string; position?: string }> = [];
   329→    for (const p of r.roster ?? []) {
   330→      const entry = {
   331→        number: p.jerseyNumber != null ? String(p.jerseyNumber) : undefined,
   332→        name: p.athlete?.displayName ?? "?",
   333→        position: p.position?.abbreviation,
   334→      };
   335→      if (p.starter) starters.push(entry);
   336→      else subs.push(entry);
   337→    }
   338→    byTeam[teamName] = { formation, starters, subs };
   339→  }
   340→  return byTeam;
   341→}

   343→// 🔴 Extrae estadísticas detalladas (posesión, tiros, faltas, córners, etc.)
   344→// y las normaliza a pares home/away con valores numéricos.
   345→function parseBoxscore(summary: EspnSummary | null, homeTeam?: string, awayTeam?: string) {
   346→  const teams = summary?.boxscore?.teams ?? [];
   347→  if (teams.length < 2) return [];
   348→  const homeStats = teams.find(t => t.team?.displayName === homeTeam) ?? teams[0];
   349→  const awayStats = teams.find(t => t.team?.displayName === awayTeam) ?? teams[1];
   350→  if (!homeStats?.statistics || !awayStats?.statistics) return [];
   351→
   352→  // Map de stats de cada team por nombre
   353→  const homeMap = new Map<string, string>();
   354→  for (const s of homeStats.statistics) {
   355→    if (s.name) homeMap.set(s.name, s.displayValue ?? "");
   356→  }
   357→  const awayMap = new Map<string, string>();
   358→  for (const s of awayStats.statistics) {
   359→    if (s.name) awayMap.set(s.name, s.displayValue ?? "");
   360→  }
   361→
   362→  // Stats que nos interesan con su label en español. Para % usamos leftPercent/rightPercent.
   363→  // Para conteos (tiros, faltas, córners) usamos los valores absolutos.
   364→  const STAT_DEFS: Array<{ name: string; label: string; isPercent: boolean }> = [
   365→    { name: "possessionPct", label: "Posesión", isPercent: true },
   366→    { name: "totalShots", label: "Tiros", isPercent: false },
   367→    { name: "shotsOnTarget", label: "Tiros al arco", isPercent: false },
   368→    { name: "foulsCommitted", label: "Faltas", isPercent: false },
   369→    { name: "cornerKicks", label: "Córners", isPercent: false },
   370→    { name: "offsides", label: "Offsides", isPercent: false },
   371→    { name: "saves", label: "Atajadas", isPercent: false },
   372→    { name: "yellowCards", label: "Amarillas", isPercent: false },
   373→    { name: "redCards", label: "Rojas", isPercent: false },
   374→    { name: "accuratePasses", label: "Pases buenos", isPercent: false },
   375→    { name: "totalPasses", label: "Pases totales", isPercent: false },
   376→    { name: "passPct", label: "Precisión pases", isPercent: true },
   377→  ];
   378→
   379→  const result: Array<{ label: string; home: number; away: number; isPercent: boolean }> = [];
   380→  for (const def of STAT_DEFS) {
   381→    const hv = homeMap.get(def.name);
   382→    const av = awayMap.get(def.name);
   383→    if (hv == null && av == null) continue;
   384→    const hNum = parseFloat(hv ?? "0") || 0;
   385→    const aNum = parseFloat(av ?? "0") || 0;
   386→    result.push({ label: def.label, home: hNum, away: aNum, isPercent: def.isPercent });
   387→  }
   388→  return result;
   389→}

   406→function normalizeEvent(e: TsdbEvent) {
   407→  const homeScore = e.intHomeScore != null && e.intHomeScore !== "" ? Number(e.intHomeScore) : undefined;
   408→  const awayScore = e.intAwayScore != null && e.intAwayScore !== "" ? Number(e.intAwayScore) : undefined;
   409→  const live = homeScore !== undefined && awayScore !== undefined && /progress|q[1-5]|ht|1h|2h|live|in play/i.test(e.strStatus ?? "");
   410→  return {
   411→    id: e.idEvent,
   412→    match: `${e.strHomeTeam ?? "?"} vs ${e.strAwayTeam ?? "?"}`,
   413→    homeTeam: e.strHomeTeam,
   414→    awayTeam: e.strAwayTeam,
   415→    homeScore,
   416→    awayScore,
   417→    status: e.strStatus ?? (live ? "en juego" : homeScore !== undefined ? "finalizado" : "programado"),
   418→    league: e.strLeague,
   419→    sport: e.strSport,
   420→    date: e.dateEvent,
   421→    time: e.strTime,
   422→    live,
   423→  };
   424→}

   530→          source: "Wikipedia",

   538→      status: "no_data",
   539→      query,
