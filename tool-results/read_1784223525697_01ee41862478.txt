    20→// 🔴 P2 — Platform detection para el botón "Navegar" de route_map.
    21→// iOS detection incluye el iPad con iOS 13+ que se reporta como MacIntel.
    22→function isIOS(): boolean {
    23→  if (typeof navigator === "undefined") return false;
    24→  return (
    25→    /iPad|iPhone|iPod/.test(navigator.userAgent) ||
    26→    (navigator.platform === "MacIntel" && (navigator.maxTouchPoints ?? 0) > 1)
    27→  );
    28→}
    29→
    30→function isAndroid(): boolean {
    31→  if (typeof navigator === "undefined") return false;
    32→  return /Android/i.test(navigator.userAgent);
    33→}

    35→/**
    36→ * 🔴 P2 — Construye la URL de navegación según plataforma y datos disponibles.
    37→ * - Si lat/lng están presentes:
    38→ *   • Android: geo:${lat},${lng}?q=${destination}
    39→ *   • iOS:     maps://?daddr=${lat},${lng}
    40→ *   • Desktop: https://www.google.com/maps/dir/?api=1&destination=${dest}
    41→ * - Si NO hay lat/lng, cae a esquemas address-based:
    42→ *   • Android: geo:0,0?q=${destination}
    43→ *   • iOS:     maps://?daddr=${destination}
    44→ *   • Desktop: https://www.google.com/maps/dir/?api=1&destination=${dest}
    45→ * Devuelve null si no hay destino (ni to ni from ni lat/lng).
    46→ */
    47→function buildNavUrl(opts: {
    48→  destination: string;
    49→  lat?: number;
    50→  lng?: number;
    51→}): string | null {
    52→  const { destination, lat, lng } = opts;
    53→  const hasCoords = typeof lat === "number" && typeof lng === "number";
    54→  if (isAndroid()) {
    55→    if (hasCoords) {
    56→      return `geo:${lat},${lng}?q=${encodeURIComponent(destination)}`;
    57→    }
    58→    return `geo:0,0?q=${encodeURIComponent(destination)}`;
    59→  }
    60→  if (isIOS()) {
    61→    if (hasCoords) {
    62→      return `maps://?daddr=${lat},${lng}`;
    63→    }
    64→    return `maps://?daddr=${encodeURIComponent(destination)}`;
    65→  }
    66→  // Desktop (o fallback)
    67→  return `https://www.google.com/maps/dir/?api=1&destination=${encodeURIComponent(destination)}`;
    68→}

    70→// 🔴 v2: ComparisonBar — barra comparativa para stats de fútbol (o cualquier
    71→// matchup home/away). Dibuja dos segmentos coloreados con team colors,
    72→// mostrando visualmente quién domina esa stat.
    73→function ComparisonBar({ bar }: { bar: NonNullable<DetailRow["bar"]> }) {
    74→  const { homeValue, awayValue, isPercent, homeColor, awayColor } = bar;
    75→  const total = (homeValue ?? 0) + (awayValue ?? 0);
    76→  const homePct = total > 0 ? Math.max(2, Math.min(98, (homeValue / total) * 100)) : 50;
    77→  const awayPct = 100 - homePct;
    78→  const hc = homeColor || "#2d6a4f";
    79→  const ac = awayColor || "#8363f9";
    80→  return (
    81→    <div className="koru-comparison-bar" role="presentation">
    82→      <span className="koru-comparison-value home" style={{ color: hc }}>
    83→        {isPercent ? `${Math.round(homeValue)}%` : homeValue}
    84→      </span>
    85→      <div className="koru-comparison-track">
    86→        <div
    87→          className="koru-comparison-fill home"
    88→          style={{ width: `${homePct}%`, background: hc }}
    89→        />
    90→        <div
    91→          className="koru-comparison-fill away"
    92→          style={{ width: `${awayPct}%`, background: ac }}
    93→        />
    94→      </div>
    95→      <span className="koru-comparison-value away" style={{ color: ac }}>
    96→        {isPercent ? `${Math.round(awayValue)}%` : awayValue}
    97→      </span>
    98→    </div>
    99→  );
   100→}

   107→// 🔴 v2: Extrae el video ID de una URL de YouTube (youtu.be / watch?v= / embed / shorts).
   108→// Devuelve null si no es una URL válida de YouTube.
   109→function youtubeId(url: string): string | null {
   110→  try {
   111→    const u = new URL(url);
   112→    if (u.hostname.includes("youtu.be")) {
   113→      const id = u.pathname.slice(1).split("/")[0];
   114→      return id || null;
   115→    }
   116→    if (u.hostname.includes("youtube.com")) {
   117→      const v = u.searchParams.get("v");
   118→      if (v) return v;
   119→      const m = u.pathname.match(/\/(?:embed|shorts)\/([\w-]+)/);
   120→      if (m) return m[1];
   121→    }
   122→    return null;
   123→  } catch {
   124→    return null;
   125→  }
   126→}

   467→function FormationPitch({ pitch }: { pitch: NonNullable<Extract<DetailSection, { kind: "pitch" }>["pitch"]> }) {
   468→  const { homeFormation, awayFormation, homePlayers, awayPlayers, homeColor, awayColor, homeName, awayName } = pitch;
   469→  const hc = homeColor || "#2d6a4f";
   470→  const ac = awayColor || "#8363f9";
   471→
   472→  // Parse formation string "4-3-3" → [4, 3, 3]
   473→  function parseFormation(f: string): number[] {
   474→    return f.split("-").map(n => parseInt(n, 10) || 0).filter(n => n > 0);
   475→  }
   476→
   477→  // Posicionar jugadores en una mitad de cancha (50% vertical)
   478→  // home juega "hacia arriba" (de abajo hacia arriba), away "hacia abajo"
   479→  // Cada línea del formation se posiciona a un Y% creciente desde el arquero.
   480→  function positionPlayers(players: Array<{ number?: string; name: string; position?: string }>, formation: string, isHome: boolean): Array<{ x: number; y: number; player: typeof players[0] }> {
   481→    if (players.length === 0) return [];
   482→    const lines = parseFormation(formation);
   483→    const result: Array<{ x: number; y: number; player: typeof players[0] }> = [];
   484→    let playerIdx = 0;
   485→
   486→    // GK siempre primero, en la línea de gol
   487→    if (players[0]) {
   488→      result.push({ x: 50, y: isHome ? 95 : 5, player: players[0] });
   489→      playerIdx = 1;
   490→    }
   491→
   492→    // Cada línea del formation: distribuir N jugadores en una fila horizontal
   493→    // Y va creciendo desde la defensa (cerca del arquero) hasta el ataque (cerca del mediocampo)
   494→    const linesCount = lines.length;
   495→    lines.forEach((count, lineIdx) => {
   496→      // Y para home: de 80% (defensa) a 55% (ataque, mediocampo)
   497→      // Y para away: de 20% (defensa) a 45% (ataque, mediocampo)
   498→      const yBase = isHome ? 80 : 20;
   499→      const yEnd = isHome ? 55 : 45;
   500→      const yRange = Math.abs(yBase - yEnd);
   501→      const yStep = linesCount > 1 ? yRange / (linesCount - 1) : 0;
   502→      const y = isHome ? yBase - lineIdx * yStep : yBase + lineIdx * yStep;
   503→
   504→      // Distribuir los N jugadores horizontalmente
   505→      for (let i = 0; i < count && playerIdx < players.length; i++) {
   506→        const x = ((i + 1) / (count + 1)) * 100;
   507→        result.push({ x, y, player: players[playerIdx] });
   508→        playerIdx++;
   509→      }
   510→    });
   511→
   512→    return result;
   513→  }
   514→
   515→  const homePositions = positionPlayers(homePlayers, homeFormation, true);
   516→  const awayPositions = positionPlayers(awayPlayers, awayFormation, false);
   517→
   518→  return (
