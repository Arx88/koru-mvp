   383→  function completeOnboarding(name: string, facts: string[] = []) {
   384→    const cleanName = name.trim() || "amigo";
   385→    const previousState = domainStateRef.current;
   386→    setUserName(cleanName);
   387→    setOnboarded(true);
   388→    localStorage.setItem("koru.onboarded", "true");
   389→    localStorage.setItem("koru.username", cleanName);
   390→    commitChatTurns((prev) => {
   391→      if (prev.length === 0) return [greetingTurn(cleanName)];
   392→      if (prev.length === 1 && prev[0].role === "koru") {
   393→        return [{ ...prev[0], text: `Hola, ${cleanName}. Cuéntame cómo estás.` }];
   394→      }
   395→      return prev;
   396→    });
   397→    commitDomainState((prev) => {
   398→      const withName = { ...prev, userName: cleanName };
   399→      const next = facts.length ? addOnboardingMemories(withName, facts) : withName;
   400→      if (!facts.length) saveState(next);
   401→      writeAuditEvent({
   402→        type: "onboarding_completed",
   403→        name: cleanName,
   404→        facts,
   405→        state: auditStateSnapshot(next),
   406→        delta: auditStateDelta(previousState, next),
   407→      });
   408→      return next;
   409→    });
   410→  }

   584→  async function submitEntry(
   585→    text: string,
   586→    transcriptSource: DailyEntry["transcriptSource"] = "typed",
   587→    history: KoruChatTurn[] = chatTurnsRef.current,
   588→  ) {
   589→    setProcessing(true);
   590→    let koruTurnId: string | null = null;
   591→    // Trackea los items creados durante el stream para preservar sus ids al
   592→    // reemplazar con el resultado final. Sin esto, el resultado final crea items
   593→    // con ids nuevos → el WebNavCardA se desmonta/remonta → pierde su estado
   594→    // (animación de progreso, visibleCount) → reinicia a "Iniciando búsqueda...".
   595→    const streamedItemsByType = new Map<string, KoruTurnItem>();
   596→    try {
   597→      const previousState = domainStateRef.current;
   598→      const domainHistory = history
   599→        .filter((turn) => turn.text.trim().length > 0)
   600→        .slice(-12)
   601→        .map((turn) => ({
   602→          role: turn.role === "koru" ? ("assistant" as const) : ("user" as const),
   603→          content: turn.text,
   604→          createdAt: turn.createdAt,
   605→        }));
   606→      const agentResult = await runBackendAgentTurn(text, previousState, domainHistory, selectedModel ?? undefined, (chunk) => {
   607→        // Fase real del pipeline: tomamos el último stateEvent para la píldora/barra.
   608→        if (chunk.stateEvents?.length) {
   609→          const last = chunk.stateEvents[chunk.stateEvents.length - 1];
   610→          setPhase(last.kind === "done" ? null : last.kind);
   611→        }
   612→        const blocksToItems = (blocks: UiBlock[]): KoruTurnItem[] =>
   613→          blocks.map((block) => ({
   614→            id: createId("item"),
   615→            kind: "action" as const,
   616→            tag: block.type,
   617→            text: (block as any).title ?? (block as any).query ?? "",
   618→            status: "working" as const,
   619→            uiBlock: block,
   620→          }));
   621→        if (!koruTurnId) {
   622→          koruTurnId = createId("turn");
   623→          const koruTurn: KoruChatTurn = {
   624→            id: koruTurnId,
   625→            role: "koru",
   626→            text: chunk.reply,
   627→            createdAt: new Date().toISOString(),
   628→            items: blocksToItems(chunk.uiBlocks),
   629→            status: "working" as const,
   630→            mascotState: chunk.mascotState ?? "working",
   631→          };
   632→          (koruTurn.items ?? []).forEach((it) => { if (it.uiBlock?.type) streamedItemsByType.set(it.uiBlock.type, it); });
   633→          commitChatTurns((prev) => [...prev, koruTurn].slice(-120));
   634→        } else {
   635→          const isDone = chunk.stateEvents?.some((e) => e.kind === "done");
   636→          commitChatTurns((prev) =>
   637→            prev.map((turn) => {
   638→              if (turn.id !== koruTurnId) return turn;
   639→              const existingItems = turn.items ?? [];
   640→              const newBlocks = chunk.uiBlocks ?? [];
   641→              // Merge por TIPO de block (no por índice). Así, si el orden cambia
   642→              // entre chunks (ej: aparece un data_card antes que el web_nav),
   643→              // cada item mantiene su identidad y estado (animación de progreso,
   644→              // contadores internos). Antes se matcheaba por idx, lo que hacía
   645→              // que un cambio de orden desmontara el componente y reiniciara la
   646→              // animación ("Iniciando búsqueda..." colgado).
   647→              const consumedExisting = new Set<string>();
   648→              const mergedItems: KoruTurnItem[] = newBlocks.map((block) => {
   649→                const existing = existingItems.find(
   650→                  (it) => it.uiBlock?.type === block.type && !consumedExisting.has(it.uiBlock!.type),
   651→                );
   652→                if (existing) {
   653→                  consumedExisting.add(existing.uiBlock!.type);
   654→                  return { ...existing, uiBlock: block, text: (block as any).title ?? (block as any).query ?? "" };
   655→                }
   656→                return {
   657→                  id: createId("item"),
   658→                  kind: "action" as const,
   659→                  tag: block.type,
   660→                  text: (block as any).title ?? (block as any).query ?? "",
   661→                  status: "working" as const,
   662→                  uiBlock: block,
   663→                };
   664→              });
   665→              mergedItems.forEach((it) => { if (it.uiBlock?.type) streamedItemsByType.set(it.uiBlock.type, it); });
   666→              return {
   667→                ...turn,
   668→                text: chunk.reply,
   669→                items: mergedItems,
   670→                mascotState: chunk.mascotState ?? turn.mascotState,
   671→                status: isDone ? ("done" as const) : ("working" as const),
   672→              };
   673→            }),
   674→          );
   675→        }
   676→      });
   677→      const result = applyBackendTurnToState(previousState, text, transcriptSource, agentResult);
   678→      commitDomainState(result.state);
   679→      if (koruTurnId) {
   680→        // Preservar ids de los items del stream: si el resultado final tiene un item
   681→        // del mismo tipo que uno ya renderizado en el stream, mantener su id (e
   682→        // identidad de React) para no desmontar/remontar el componente y perder su
   683→        // estado interno (animación de progreso del WebNavCardA, visibleCount, etc.).
   684→        const consumedTypes = new Set<string>();
   685→        const stableItems = result.items.map((item) => {
   686→          const type = item.uiBlock?.type ?? item.tag;
   687→          if (type && !consumedTypes.has(type)) {
   688→            const streamed = streamedItemsByType.get(type);
   689→            if (streamed) {
   690→              consumedTypes.add(type);
   691→              return { ...item, id: streamed.id };
   692→            }
   693→          }
   694→          return item;
   695→        });
   696→        commitChatTurns((prev) =>
   697→          prev.map((turn) =>
   698→            turn.id === koruTurnId
   699→              ? { ...turn, text: agentResult.reply, items: stableItems, status: "done" as const, mascotState: agentResult.mascotState ?? "idle" }
   700→              : turn,
   701→          ),
   702→        );
   703→        // 🔴 Memory toast: si hay memoryCandidates en la respuesta, mostrar toast.
   704→        // Buscar en memoryCandidates top-level Y dentro de toolResults (save_memory tool).
   705→        const topCandidates = (agentResult.memoryCandidates ?? []);
   706→        const toolCandidates = (agentResult.toolResults ?? []).flatMap((tr: any) => {
   707→          const data = tr.data;
   708→          if (data?.type === "memory_capture" && Array.isArray(data.memoryCandidates)) return data.memoryCandidates;
   709→          if (data?.type === "personal_capture" && Array.isArray(data.memoryCandidates)) return data.memoryCandidates;
   710→          return [];
   711→        });
   712→        const allCandidates = [...topCandidates, ...toolCandidates].filter(
   713→          (m: any) => m && typeof m.text === "string" && m.text.length > 4,
   714→        );
   715→        if (allCandidates.length > 0 && !previousState.ephemeralMode && previousState.durableMemoryEnabled) {
   716→          const first = allCandidates[0];
   717→          // Solo mostrar si NO existe ya una memoria con el mismo texto (deduplicación visual)
   718→          const normalizedNew = first.text.toLowerCase().replace(/[áéíóú]/g, (m: string) => ({ á: "a", é: "e", í: "i", ó: "o", ú: "u" }[m]!)).replace(/[^a-z0-9\s]/g, "").trim();
   719→          const alreadyExists = previousState.memories.some((existing: any) => {
   720→            const norm = existing.text.toLowerCase().replace(/[áéíóú]/g, (m: string) => ({ á: "a", é: "e", í: "i", ó: "o", ú: "u" }[m]!)).replace(/[^a-z0-9\s]/g, "").trim();
   721→            return norm === normalizedNew;
   722→          });
   723→          if (!alreadyExists) {
   724→            setMemoryToast({
   725→              id: `toast_${Date.now()}`,
   726→              kind: first.kind,
   727→              text: first.text,
   728→            });
   729→            // Auto-dismiss después de 6 segundos
   730→            setTimeout(() => {
   731→              setMemoryToast((current) => (current?.text === first.text ? null : current));
   732→            }, 6000);
   733→          }
   734→        }
   735→      }
   736→      writeAuditEvent({
   737→        type: "turn_analyzed",
   738→        engine: "backend_agent_loop",
   739→        input: text,
   740→        transcriptSource,
   741→        historyUsed: domainHistory,
   742→        response: agentResult.reply,
   743→        understanding: agentResult.understanding,
   744→        suggestedActions: agentResult.suggestedActions,
   745→        toolResults: agentResult.toolResults,
   746→        provider: agentResult.provider,
   747→        model: agentResult.model,
   748→        fallbackReason: agentResult.fallbackReason,
   749→        items: auditTurnItems(result.items),
   750→        state: auditStateSnapshot(result.state),
   751→        delta: auditStateDelta(previousState, result.state),
   752→      });
   753→      return { response: agentResult.reply, items: result.items, state: result.state, mascotState: agentResult.mascotState, koruTurnId };
   754→    } finally {
   755→      setProcessing(false);
   756→      setActivity(null);
   757→      setPhase(null);
   758→    }
   759→  }

   761→  async function sendMessage(text: string, transcriptSource: DailyEntry["transcriptSource"] = "typed") {
   762→    const cleanText = text.trim();
   763→    if (!cleanText) return null;
   764→    const previousState = domainStateRef.current;
   765→    const userTurn: KoruChatTurn = {
   766→      id: createId("turn"),
   767→      role: "user",
   768→      text: cleanText,
   769→      createdAt: new Date().toISOString(),
   770→      status: "done",
   771→    };
   772→    const historyBeforeUser = chatTurnsRef.current;
   773→    commitChatTurns((prev) => [...prev, userTurn].slice(-120));
   774→    writeAuditEvent({
   775→      type: "user_message",
   776→      turn: userTurn,
   777→      transcriptSource,
   778→      historyBeforeUser: historyBeforeUser.map((turn) => ({
   779→        id: turn.id,
   780→        role: turn.role,
   781→        text: turn.text,
   782→        createdAt: turn.createdAt,
   783→        status: turn.status,
   784→        items: auditTurnItems(turn.items),
   785→      })),
   786→      stateBefore: auditStateSnapshot(previousState),
   787→    });
   788→    setActivity(inferActivity(cleanText));
   789→    try {
   790→      const result = await submitEntry(cleanText, transcriptSource, historyBeforeUser);
   791→      let koruTurn: KoruChatTurn | undefined;
   792→      if (!result.koruTurnId) {
   793→        koruTurn = {
   794→          id: createId("turn"),
   795→          role: "koru",
   796→          text: result.response,
   797→          createdAt: new Date().toISOString(),
   798→          items: result.items,
   799→          status: "done",
   800→          mascotState: result.mascotState ?? "idle",
   801→        };
   802→        commitChatTurns((prev) => [...prev, koruTurn!].slice(-120));
   803→        writeAuditEvent({
   804→          type: "koru_message",
   805→          turn: {
   806→            id: koruTurn.id,
   807→            role: koruTurn.role,
   808→            text: koruTurn.text,
   809→            createdAt: koruTurn.createdAt,
   810→            status: koruTurn.status,
   811→            items: auditTurnItems(koruTurn.items),
   812→          },
   813→          stateAfter: auditStateSnapshot(result.state),
   814→        });
   815→      } else {
   816→        commitChatTurns((prev) =>
   817→          prev.map((turn) => {
   818→            if (turn.id !== result.koruTurnId) return turn;
   819→            const existingItems = turn.items ?? [];
   820→            const mergedItems: KoruTurnItem[] = result.items.map((newItem, idx) => {
   821→              const existing = existingItems[idx];
   822→              if (existing && existing.uiBlock?.type === newItem.uiBlock?.type) {
   823→                return { ...newItem, id: existing.id };
   824→              }
   825→              return newItem;
   826→            });
   827→            return { ...turn, items: mergedItems, status: "done" as const, mascotState: result.mascotState ?? "idle" };
   828→          }),
   829→        );
   830→        koruTurn = chatTurnsRef.current.find((t) => t.id === result.koruTurnId);
   831→        writeAuditEvent({
   832→          type: "koru_message",
   833→          turn: {
   834→            id: result.koruTurnId,
   835→            role: "koru",
   836→            text: result.response,
   837→            createdAt: new Date().toISOString(),
   838→            status: "done",
   839→            items: auditTurnItems(result.items),
   840→          },
   841→          stateAfter: auditStateSnapshot(result.state),
   842→        });
   843→      }
   844→      const autoWebAction = result.state.actions
   845→        .filter((action) => !previousState.actions.some((existing) => existing.id === action.id))
   846→        .find(shouldAutoRunAction);
   847→      if (autoWebAction) {
   848→        window.setTimeout(() => runReadonlyWebAction(autoWebAction), 0);
   849→      }
   850→      return koruTurn ?? null;
   851→    } catch (error) {
   852→      const message = error instanceof Error ? error.message : "Fallo desconocido del agente.";
   853→      const errorTurn: KoruChatTurn = {
   854→        id: createId("turn"),
   855→        role: "koru",
   856→        text: `No pude contactar bien al agente ahora. No voy a inventarte una respuesta. Detalle: ${message}`,
   857→        createdAt: new Date().toISOString(),
   858→        status: "error",
   859→      };
   860→      commitChatTurns((prev) => [...prev, errorTurn].slice(-120));
   861→      writeAuditEvent({
   862→        type: "turn_error",
   863→        input: cleanText,
   864→        error: message,
   865→        turn: errorTurn,
   866→        state: auditStateSnapshot(domainStateRef.current),
   867→      });
   868→      setActivity(null);
   869→      return errorTurn;
   870→    }
   871→  }

   898→  function runReadonlyWebAction(original: AssistantAction): KoruTurnItem {
   899→    // Dato en vivo (clima/trafico/mercado) = quick, sin panel largo. Un
   900→    // informe/investigacion real (webMode "research", o web_research sin
   901→    // modo puntual) = deep, se gana el panel "Trabajando...". Ver
   902→    // flujo-informe-aoe2.html pasos 3 (informe) vs 7 (clima).
   903→    const webMode = original.payload.webMode;
   904→    const isQuickSignal = webMode === "weather" || webMode === "traffic" || webMode === "market";
   905→    const isDeepResearch = webMode === "research" || (original.kind === "web_research" && !webMode);
   906→    setActivity({
   907→      kind: "searching",
   908→      depth: isQuickSignal ? "quick" : isDeepResearch ? "deep" : "quick",
   909→      label: isDeepResearch
   910→        ? "Es un tema para investigar a fondo. Dame unos segundos."
   911→        : original.kind === "world_signal"
   912→          ? "Busco senales reales."
   913→          : "Consulto fuentes reales.",
   914→    });
   915→    const now = new Date().toISOString();
   916→    const approvedAction: AssistantAction = {
   917→      ...original,
   918→      status: "approved",
   919→      updatedAt: now,
   920→    };
   921→    commitDomainState((prev) => {
   922→      const exists = prev.actions.some((action) => action.id === original.id);
   923→      const next = {
   924→        ...prev,
   925→        actions: exists
   926→          ? prev.actions.map((action) => (action.id === original.id ? approvedAction : action))
   927→          : [approvedAction, ...prev.actions],
   928→        updatedAt: now,
   929→      };
   930→      saveState(next);
   931→      return next;
   932→    });
   933→    const waitingItem = actionToTurnItem(approvedAction);
   934→    replaceActionInChat(waitingItem);
   935→
   936→    void (async () => {
   937→      try {
   938→        const result = await runWebNavigation(original.payload);
   939→        const webPayload = webResultToPayload(result);
   940→        const nextUiBlock = patchUiBlockWithWebResult(approvedAction, webPayload);
   941→        const finishedAt = new Date().toISOString();
   942→        const patchedAction: AssistantAction = {
   943→          ...approvedAction,
   944→          payload: {
   945→            ...approvedAction.payload,
   946→            ...webPayload,
   947→            uiBlock: nextUiBlock ?? approvedAction.payload.uiBlock,
   948→          },
   949→          result: result.recommendation,
   950→          status: "executed",
   951→          updatedAt: finishedAt,
   952→          executedAt: finishedAt,
   953→        };
   954→        const patchedItem = actionToTurnItem(patchedAction);
   955→        commitDomainState((prev) => {
   956→          const next = {
   957→            ...prev,
   958→            actions: prev.actions.map((action) => (action.id === original.id ? patchedAction : action)),
   959→            updatedAt: finishedAt,
   960→          };
   961→          saveState(next);
   962→          writeAuditEvent({
   963→            type: "readonly_web_action_finished",
   964→            action: {
   965→              id: patchedAction.id,
   966→              kind: patchedAction.kind,
   967→              title: patchedAction.title,
   968→              status: patchedAction.status,
   969→              result: patchedAction.result,
   970→              payload: patchedAction.payload,
   971→            },
   972→            item: auditTurnItems([patchedItem])[0],
   973→            state: auditStateSnapshot(next),
   974→            delta: auditStateDelta(prev, next),
   975→          });
   976→          return next;
   977→        });
   978→        replaceActionInChat(patchedItem);
   979→      } finally {
   980→        setActivity(null);
   981→      }
   982→    })();
   983→    return waitingItem;
   984→  }

   986→  function reviewAction(id: string, approve: boolean): KoruTurnItem | null {
   987→    const currentState = domainStateRef.current;
   988→    const original = currentState.actions.find((action) => action.id === id);
   989→    if (approve && (original?.kind === "web_research" || original?.kind === "world_signal")) {
   990→      return runReadonlyWebAction(original);
   991→    }
   992→
   993→    const nextState = approve ? approveAndExecuteAction(currentState, id) : rejectAction(currentState, id);
   994→    const reviewed = nextState.actions.find((action) => action.id === id);
   995→    if (!reviewed) return null;
   996→    commitDomainState(nextState);
   997→    const item = actionToTurnItem(reviewed);
   998→    replaceActionInChat(item);
   999→    writeAuditEvent({
  1000→      type: "action_reviewed",
  1001→      actionId: id,
  1002→      approved: approve,
  1003→      item: auditTurnItems([item])[0],
  1004→      state: auditStateSnapshot(nextState),
  1005→      delta: auditStateDelta(currentState, nextState),
  1006→    });
  1007→    appendKoruConfirmation(approve ? actionConfirmationText(item) : "Lo suelto. No hice nada con eso.");
  1008→
  1009→    return item;
  1010→  }
