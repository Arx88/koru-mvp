   447→            } catch {
   448→              // If still failing, leave it queued for the next retry.
   449→              break;
   450→            }
   451→          }
   452→        }).catch(() => { /* best-effort */ });
   453→      }
   454→    });
   455→    return unsubscribe;
   456→  }, []);
   457→
   458→  // ── Escuchar mensajes proactivos del engine ──
   459→  useEffect(() => {

   465→    };

   467→    return () => window.removeEventListener("koru:proactive", onProactive);
   468→  }, []);
   469→
   470→  const energy = domainState.trustedEnergy;
   471→  const roots = domainState.memories.filter((m) => m.status === "confirmed").length;
   472→  const stage = domainStageToNew(domainState.stage);
   473→  const ephemeral = domainState.ephemeralMode;
   474→
   475→  const priorities: Priority[] = useMemo(() => {
   476→    const actionItems = domainState.actions

   856→  async function submitEntry(
   857→    text: string,
   858→    transcriptSource: DailyEntry["transcriptSource"] = "typed",
   859→    history: KoruChatTurn[] = chatTurnsRef.current,
   860→  ) {
   861→    setProcessing(true);
   862→    let koruTurnId: string | null = null;
   863→    // Trackea los items creados durante el stream para preservar sus ids al
   864→    // reemplazar con el resultado final. Sin esto, el resultado final crea items
   865→    // con ids nuevos → el WebNavCardA se desmonta/remonta → pierde su estado
   866→    // (animación de progreso, visibleCount) → reinicia a "Iniciando búsqueda...".
   867→    const streamedItemsByType = new Map<string, KoruTurnItem>();
   868→    try {
   869→      const previousState = domainStateRef.current;
   870→      const domainHistory = history
   871→        .filter((turn) => turn.text.trim().length > 0)
   872→        .slice(-12)
   873→        .map((turn) => ({
   874→          role: turn.role === "koru" ? ("assistant" as const) : ("user" as const),
   875→          content: turn.text,
   876→          createdAt: turn.createdAt,
   877→        }));
   878→      const agentResult = await runBackendAgentTurn(text, previousState, domainHistory, selectedModel ?? undefined, (chunk) => {
   879→        // Fase real del pipeline: tomamos el último stateEvent para la píldora/barra.
   880→        if (chunk.stateEvents?.length) {
   881→          const last = chunk.stateEvents[chunk.stateEvents.length - 1];
   882→          setPhase(last.kind === "done" ? null : last.kind);
   883→        }
   884→        const blocksToItems = (blocks: UiBlock[]): KoruTurnItem[] =>
   885→          blocks.map((block) => ({
   886→            id: createId("item"),
   887→            kind: "action" as const,
   888→            tag: block.type,
   889→            text: (block as any).title ?? (block as any).query ?? "",
   890→            status: "working" as const,
   891→            uiBlock: block,
   892→          }));
   893→        if (!koruTurnId) {
   894→          koruTurnId = createId("turn");
   895→          const koruTurn: KoruChatTurn = {
   896→            id: koruTurnId,
   897→            role: "koru",
   898→            text: chunk.reply,
   899→            createdAt: new Date().toISOString(),
   900→            items: blocksToItems(chunk.uiBlocks),
   901→            status: "working" as const,
   902→            mascotState: chunk.mascotState ?? "working",
   903→          };
   904→          (koruTurn.items ?? []).forEach((it) => { if (it.uiBlock?.type) streamedItemsByType.set(it.uiBlock.type, it); });
   905→          commitChatTurns((prev) => [...prev, koruTurn].slice(-120));
   906→        } else {
   907→          const isDone = chunk.stateEvents?.some((e) => e.kind === "done");
   908→          commitChatTurns((prev) =>
   909→            prev.map((turn) => {
   910→              if (turn.id !== koruTurnId) return turn;
   911→              const existingItems = turn.items ?? [];
   912→              const newBlocks = chunk.uiBlocks ?? [];
   913→              // Merge por TIPO de block (no por índice). Así, si el orden cambia
   914→              // entre chunks (ej: aparece un data_card antes que el web_nav),
   915→              // cada item mantiene su identidad y estado (animación de progreso,
   916→              // contadores internos). Antes se matcheaba por idx, lo que hacía
   917→              // que un cambio de orden desmontara el componente y reiniciara la
   918→              // animación ("Iniciando búsqueda..." colgado).
   919→              const consumedExisting = new Set<string>();
   920→              const mergedItems: KoruTurnItem[] = newBlocks.map((block) => {
   921→                const existing = existingItems.find(
   922→                  (it) => it.uiBlock?.type === block.type && !consumedExisting.has(it.uiBlock!.type),
   923→                );
   924→                if (existing) {
   925→                  consumedExisting.add(existing.uiBlock!.type);
   926→                  return { ...existing, uiBlock: block, text: (block as any).title ?? (block as any).query ?? "" };
   927→                }
   928→                return {
   929→                  id: createId("item"),
   930→                  kind: "action" as const,
   931→                  tag: block.type,
   932→                  text: (block as any).title ?? (block as any).query ?? "",
   933→                  status: "working" as const,
   934→                  uiBlock: block,
   935→                };
   936→              });
   937→              mergedItems.forEach((it) => { if (it.uiBlock?.type) streamedItemsByType.set(it.uiBlock.type, it); });
   938→              return {
   939→                ...turn,
   940→                text: chunk.reply,
   941→                items: mergedItems,
   942→                mascotState: chunk.mascotState ?? turn.mascotState,
   943→                status: isDone ? ("done" as const) : ("working" as const),
   944→              };
   945→            }),
   946→          );
   947→        }
   948→      });
   949→      const result = applyBackendTurnToState(previousState, text, transcriptSource, agentResult);
   950→      commitDomainState(result.state);
   951→
   952→      // 🔴 Programar notificaciones para nuevos commitments con dueAt
   953→      const newCommitments = (result.state.commitments ?? []).filter(
   954→        (c) => c.dueAt && !previousState.commitments?.find((pc) => pc.id === c.id)
   955→      );
   956→      if (newCommitments.length > 0) {
   957→        const userId = result.state.userId ?? "default";
   958→        for (const c of newCommitments) {
   959→          scheduleReminderNotification(c as any, userId);
   960→          // 🔴 NUEVO: programar setTimeout preciso para que la notificación
   961→          // dispare EXACTO cuando llega la hora, no en el próximo heartbeat tick.
   962→          schedulePreciseTimeout(c.id, c.dueAt!, c.title, userId);
   963→        }
   964→        // 🔴 Request notification permission — best effort
   965→        requestNotificationPermission().catch(() => {});
   966→      }
   967→      if (koruTurnId) {
   968→        // Preservar ids de los items del stream: si el resultado final tiene un item
   969→        // del mismo tipo que uno ya renderizado en el stream, mantener su id (e
   970→        // identidad de React) para no desmontar/remontar el componente y perder su
   971→        // estado interno (animación de progreso del WebNavCardA, visibleCount, etc.).
   972→        const consumedTypes = new Set<string>();
   973→        const stableItems = result.items.map((item) => {
   974→          const type = item.uiBlock?.type ?? item.tag;
   975→          if (type && !consumedTypes.has(type)) {
   976→            const streamed = streamedItemsByType.get(type);
   977→            if (streamed) {
   978→              consumedTypes.add(type);
   979→              return { ...item, id: streamed.id };
   980→            }
   981→          }
   982→          return item;
   983→        });
   984→        commitChatTurns((prev) =>
   985→          prev.map((turn) =>
   986→            turn.id === koruTurnId
   987→              ? { ...turn, text: agentResult.reply, items: stableItems, status: "done" as const, mascotState: agentResult.mascotState ?? "idle" }
   988→              : turn,
   989→          ),
   990→        );
   991→        // 🔴 Voice: si voice está activado, Koru habla la respuesta
   992→        if (voiceEnabled && agentResult.reply) {
   993→          speak(agentResult.reply);
   994→        }
   995→        // 🔴 Memory toast: si hay memoryCandidates en la respuesta, mostrar toast.

  1196→  function runReadonlyWebAction(original: AssistantAction): KoruTurnItem {
  1197→    // Dato en vivo (clima/trafico/mercado) = quick, sin panel largo. Un
  1198→    // informe/investigacion real (webMode "research", o web_research sin
  1199→    // modo puntual) = deep, se gana el panel "Trabajando...". Ver
  1200→    // flujo-informe-aoe2.html pasos 3 (informe) vs 7 (clima).
  1201→    const webMode = original.payload.webMode;
  1202→    const isQuickSignal = webMode === "weather" || webMode === "traffic" || webMode === "market";
  1203→    const isDeepResearch = webMode === "research" || (original.kind === "web_research" && !webMode);
  1204→    setActivity({
  1205→      kind: "searching",
  1206→      depth: isQuickSignal ? "quick" : isDeepResearch ? "deep" : "quick",
  1207→      label: isDeepResearch
  1208→        ? "Es un tema para investigar a fondo. Dame unos segundos."
  1209→        : original.kind === "world_signal"
  1210→          ? "Busco senales reales."
  1211→          : "Consulto fuentes reales.",
  1212→    });
  1213→    const now = new Date().toISOString();
  1214→    const approvedAction: AssistantAction = {
  1215→      ...original,
  1216→      status: "approved",
  1217→      updatedAt: now,
  1218→    };
  1219→    commitDomainState((prev) => {
  1220→      const exists = prev.actions.some((action) => action.id === original.id);
  1221→      const next = {
  1222→        ...prev,
  1223→        actions: exists
  1224→          ? prev.actions.map((action) => (action.id === original.id ? approvedAction : action))
  1225→          : [approvedAction, ...prev.actions],
  1226→        updatedAt: now,
  1227→      };
  1228→      saveState(next);
  1229→      return next;
  1230→    });
  1231→    const waitingItem = actionToTurnItem(approvedAction);
  1232→    replaceActionInChat(waitingItem);
  1233→
  1234→    void (async () => {
  1235→      try {
  1236→        const result = await runWebNavigation(original.payload);
  1237→        const webPayload = webResultToPayload(result);
  1238→        const nextUiBlock = patchUiBlockWithWebResult(approvedAction, webPayload);
  1239→        const finishedAt = new Date().toISOString();
  1240→        const patchedAction: AssistantAction = {
  1241→          ...approvedAction,
  1242→          payload: {
  1243→            ...approvedAction.payload,
  1244→            ...webPayload,
  1245→            uiBlock: nextUiBlock ?? approvedAction.payload.uiBlock,
  1246→          },
  1247→          result: result.recommendation,
  1248→          status: "executed",
  1249→          updatedAt: finishedAt,
  1250→          executedAt: finishedAt,
  1251→        };
  1252→        const patchedItem = actionToTurnItem(patchedAction);
  1253→        commitDomainState((prev) => {
  1254→          const next = {
  1255→            ...prev,
  1256→            actions: prev.actions.map((action) => (action.id === original.id ? patchedAction : action)),
  1257→            updatedAt: finishedAt,
  1258→          };
  1259→          saveState(next);
  1260→          writeAuditEvent({
  1261→            type: "readonly_web_action_finished",
  1262→            action: {
  1263→              id: patchedAction.id,
  1264→              kind: patchedAction.kind,
  1265→              title: patchedAction.title,
  1266→              status: patchedAction.status,
  1267→              result: patchedAction.result,
  1268→              payload: patchedAction.payload,
  1269→            },
  1270→            item: auditTurnItems([patchedItem])[0],
  1271→            state: auditStateSnapshot(next),
  1272→            delta: auditStateDelta(prev, next),
  1273→          });
  1274→          return next;
  1275→        });
  1276→        replaceActionInChat(patchedItem);
  1277→      } finally {
  1278→        setActivity(null);
  1279→      }
  1280→    })();
  1281→    return waitingItem;
  1282→  }

  1284→  function reviewAction(id: string, approve: boolean): KoruTurnItem | null {
  1285→    const currentState = domainStateRef.current;
  1286→    const original = currentState.actions.find((action) => action.id === id);
  1287→    if (approve && (original?.kind === "web_research" || original?.kind === "world_signal")) {
  1288→      return runReadonlyWebAction(original);
  1289→    }
  1290→
  1291→    const nextState = approve ? approveAndExecuteAction(currentState, id) : rejectAction(currentState, id);
  1292→    const reviewed = nextState.actions.find((action) => action.id === id);
  1293→    if (!reviewed) return null;
  1294→    commitDomainState(nextState);
  1295→    const item = actionToTurnItem(reviewed);
  1296→    replaceActionInChat(item);
  1297→    writeAuditEvent({
  1298→      type: "action_reviewed",
  1299→      actionId: id,
  1300→      approved: approve,
  1301→      item: auditTurnItems([item])[0],
  1302→      state: auditStateSnapshot(nextState),
  1303→      delta: auditStateDelta(currentState, nextState),
  1304→    });
  1305→    appendKoruConfirmation(approve ? actionConfirmationText(item) : "Lo suelto. No hice nada con eso.");
  1306→
  1307→    return item;
  1308→  }
