    77→    77→function escapeHtml(s: string): string {
    78→    78→  return s.replace(/&/g, "&amp;").replace(/</g, "&lt;").replace(/>/g, "&gt;");
    79→    79→}
    80→    80→
    81→    81→function inlineMd(s: string): string {
    82→    82→  // s ya está escapado
    83→    83→  s = s.replace(/\[([^\]]+)\]\(([^)]+)\)/g, '<a href="$2" target="_blank" rel="noopener noreferrer">$1</a>');
    84→    84→  s = s.replace(/\*\*([^*]+)\*\*/g, "<strong>$1</strong>");
    85→    85→  s = s.replace(/\*([^*]+)\*/g, "<em>$1</em>");
    86→    86→  return s;
    87→    87→}
    88→    88→
    89→    89→function renderMarkdown(text: string): string {
    90→    90→  if (!text.trim()) return "";
    91→    91→  const lines = text.split("\n");
    92→    92→  const out: string[] = [];
    93→    93→  let inList = false;
    94→    94→  let para: string[] = [];
    95→    95→  const flushPara = () => {
    96→    96→    if (para.length > 0) {
    97→    97→      out.push(`<p>${para.map(p => inlineMd(escapeHtml(p))).join("<br />")}</p>`);
    98→    98→      para = [];
    99→    99→    }
   100→   100→  };
   101→   101→  for (const line of lines) {
   102→   102→    const listMatch = line.match(/^\s*-\s+(.*)$/);
   103→   103→    if (listMatch) {
   104→   104→      flushPara();
   105→   105→      if (!inList) { out.push("<ul>"); inList = true; }
   106→   106→      out.push(`<li>${inlineMd(escapeHtml(listMatch[1]))}</li>`);
   107→   107→      continue;
   108→   108→    }
   109→   109→    if (inList) { out.push("</ul>"); inList = false; }
   110→   110→    para.push(line);
   111→   111→  }
   112→   112→  flushPara();
   113→   113→  if (inList) out.push("</ul>");
   114→   114→  return out.join("");
   115→   115→}

   551→   551→  function insertWrap(wrap: string, placeholder: string) {
   552→   552→    const ta = notesRef.current;
   553→   553→    if (!ta) return;
   554→   554→    const start = ta.selectionStart;
   555→   555→    const end = ta.selectionEnd;
   556→   556→    const sel = notes.slice(start, end) || placeholder;
   557→   557→    const next = notes.slice(0, start) + wrap + sel + wrap + notes.slice(end);
   558→   558→    setNotes(next);
   559→   559→    requestAnimationFrame(() => {
   560→   560→      ta.focus();
   561→   561→      const pos = start + wrap.length;
   562→   562→      ta.setSelectionRange(pos, pos + sel.length);
   563→   563→    });
   564→   564→  }

   566→   566→  function insertPrefix(prefix: string) {
   567→   567→    const ta = notesRef.current;
   568→   568→    if (!ta) return;
   569→   569→    const start = ta.selectionStart;
   570→   570→    const before = notes.slice(0, start);
   571→   571→    const lineStart = before.lastIndexOf("\n") + 1;
   572→   572→    const next = notes.slice(0, lineStart) + prefix + notes.slice(lineStart);
   573→   573→    setNotes(next);
   574→   574→    requestAnimationFrame(() => {
   575→   575→      ta.focus();
   576→   576→      const pos = start + prefix.length;
   577→   577→      ta.setSelectionRange(pos, pos);
   578→   578→    });
   579→   579→  }

   582→   582→    const ta = notesRef.current;
   583→   583→    if (!ta) return;
   584→   584→    const start = ta.selectionStart;
   585→   585→    const end = ta.selectionEnd;
   586→   586→    const sel = notes.slice(start, end) || "texto";

   594→   594→    });
