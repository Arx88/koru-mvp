    77→function escapeHtml(s: string): string {
    78→  return s.replace(/&/g, "&amp;").replace(/</g, "&lt;").replace(/>/g, "&gt;");
    79→}
    80→
    81→function inlineMd(s: string): string {
    82→  // s ya está escapado
    83→  s = s.replace(/\[([^\]]+)\]\(([^)]+)\)/g, '<a href="$2" target="_blank" rel="noopener noreferrer">$1</a>');
    84→  s = s.replace(/\*\*([^*]+)\*\*/g, "<strong>$1</strong>");
    85→  s = s.replace(/\*([^*]+)\*/g, "<em>$1</em>");
    86→  return s;
    87→}

    89→function renderMarkdown(text: string): string {
    90→  if (!text.trim()) return "";
    91→  const lines = text.split("\n");
    92→  const out: string[] = [];
    93→  let inList = false;
    94→  let para: string[] = [];
    95→  const flushPara = () => {
    96→    if (para.length > 0) {
    97→      out.push(`<p>${para.map(p => inlineMd(escapeHtml(p))).join("<br />")}</p>`);
    98→      para = [];
    99→    }
   100→  };
   101→  for (const line of lines) {
   102→    const listMatch = line.match(/^\s*-\s+(.*)$/);
   103→    if (listMatch) {
   104→      flushPara();
   105→      if (!inList) { out.push("<ul>"); inList = true; }
   106→      out.push(`<li>${inlineMd(escapeHtml(listMatch[1]))}</li>`);
   107→      continue;
   108→    }
   109→    if (inList) { out.push("</ul>"); inList = false; }
   110→    para.push(line);
   111→  }
   112→  flushPara();
   113→  if (inList) out.push("</ul>");
   114→  return out.join("");
   115→}

   400→        title: title.trim() || "Plan de entrenamiento",
   401→        collection: finalCollection,
   402→        notes: ejText,
   403→        kind,
   404→        sourceBlock: buildSavedRecordBlock({
   405→          title: title.trim() || "Plan de entrenamiento",
   406→          collection: finalCollection,
   407→          notes: ejText,
   408→          kind,
   409→        }),
   410→      });
   411→      return;
   412→    }
   413→
   414→    if (selected === "memoria") {
   415→      const memText = [
   416→        memoriaText.trim(),
   417→        `Tipo: ${memoriaKind}`,

   447→        label: o.label,
   448→        factorScores: Object.fromEntries(facIds.map(id => [id, o.factorScores[id] ?? 0])),
   449→      }));

   454→        collection: finalCollection,
   455→        notes: fullNotes,
   456→        kind,
   457→        sourceBlock: buildSavedRecordBlock({
   458→          title: title.trim() || "Decisión",
   459→          collection: finalCollection,
   460→          notes: fullNotes,
   461→          kind,
   462→        }),
   463→      });
   464→      return;
   465→    }
   466→
   467→    // nota (default)
   468→    await persist({
   469→      title: title.trim() || "Nota sin título",
   470→      collection: finalCollection,
   471→      notes: notes.trim() || undefined,
   472→      kind,
   473→    });
   474→  }

   517→  function applySuggestion(s: AiSuggestion) {
   518→    const v = s.value;
   519→    switch (s.field) {
   520→      case "title": setTitle(v); break;
   521→      case "notes": setNotes(v); break;
   522→      case "amount": setAmount(v); break;
   523→      case "currency": setCurrency(v); break;
   524→      case "url": setUrl(v); break;
   525→      case "collection": setCollection(v); break;
   526→      case "prepTime": setPrepTime(v); break;
   527→      case "cookTime": setCookTime(v); break;
   528→      case "servings": setServings(v); break;
   529→      case "rutinaCadence": setRutinaCadence((v as RutinaCadence) || "daily"); break;
   530→      case "rutinaTarget": setRutinaTarget(v); break;
   531→      case "rutinaUnit": setRutinaUnit(v); break;
   532→      case "rutinaAnchorTime": setRutinaAnchorTime(v); break;
   533→      case "weeksTotal": setWeeksTotal(v); break;
   534→      case "memoriaText": setMemoriaText(v); break;
   535→      case "memoriaKind": setMemoriaKind((v as MemoriaKind) || "profile"); break;
   536→      case "memoriaConfidence": setMemoriaConfidence(parseFloat(v) || 0.5); break;
   537→      case "deadline": setDeadline(v); break;
   538→      default: break;
   539→    }
   540→    setAiSuggestions(prev => prev.filter(p => p.field !== s.field));
   541→  }

   551→  function insertWrap(wrap: string, placeholder: string) {
   552→    const ta = notesRef.current;
   553→    if (!ta) return;
   554→    const start = ta.selectionStart;
   555→    const end = ta.selectionEnd;
   556→    const sel = notes.slice(start, end) || placeholder;
   557→    const next = notes.slice(0, start) + wrap + sel + wrap + notes.slice(end);
   558→    setNotes(next);
   559→    requestAnimationFrame(() => {
   560→      ta.focus();
   561→      const pos = start + wrap.length;
   562→      ta.setSelectionRange(pos, pos + sel.length);
   563→    });
   564→  }
   565→
   566→  function insertPrefix(prefix: string) {
   567→    const ta = notesRef.current;
   568→    if (!ta) return;
   569→    const start = ta.selectionStart;
   570→    const before = notes.slice(0, start);
   571→    const lineStart = before.lastIndexOf("\n") + 1;
   572→    const next = notes.slice(0, lineStart) + prefix + notes.slice(lineStart);
   573→    setNotes(next);
   574→    requestAnimationFrame(() => {
   575→      ta.focus();
   576→      const pos = start + prefix.length;
   577→      ta.setSelectionRange(pos, pos);
   578→    });
   579→  }
   580→
   581→  function insertLink() {
   582→    const ta = notesRef.current;
   583→    if (!ta) return;
   584→    const start = ta.selectionStart;
   585→    const end = ta.selectionEnd;
   586→    const sel = notes.slice(start, end) || "texto";
   587→    const link = `[${sel}](https://)`;
   588→    const next = notes.slice(0, start) + link + notes.slice(end);
   589→    setNotes(next);
   590→    requestAnimationFrame(() => {
   591→      ta.focus();
   592→      const urlStart = start + sel.length + 3; // "[" + sel + "]("
   593→      ta.setSelectionRange(urlStart, urlStart + 8); // selecciona "https://"
   594→    });
   595→  }
