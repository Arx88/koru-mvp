   724вҶ’   724вҶ’function savedRecord(b: Of<"saved_record">): KoruPresentation {
   725вҶ’   725вҶ’  const records = b.records ?? [];
   726вҶ’   726вҶ’  const first = records[0];
   727вҶ’   727вҶ’  // La colecciГіn donde quedГі lo guardado: es el corazГіn de la promesa
   728вҶ’   728вҶ’  // "Listo, guardado en Sitios de IA" вҶ’ CTA "Ver colecciГіn" abre Mis Colecciones.
   729вҶ’   729вҶ’  // Solo usar collection si first.collection existe; NO usar b.title como fallback
   730вҶ’   730вҶ’  // porque b.title suele ser "Guardado" (genГ©rico) y produce "Guardado en Guardado".
   731вҶ’   731вҶ’  const collection = clean(first?.collection);
   732вҶ’   732вҶ’  const isLink = Boolean(first?.url);
   733вҶ’   733вҶ’  // desc: construir de forma robusta sin duplicados (case-insensitive + accent-insensitive).
   734вҶ’   734вҶ’  // Para "CafГ©" + "cafe" + "2000 ARS": solo mostrar "CafГ© В· 2000 ARS".
   735вҶ’   735вҶ’  let desc: string | undefined;
   736вҶ’   736вҶ’  if (records.length > 1) {
   737вҶ’   737вҶ’    desc = `${records.length} registros guardados`;
   738вҶ’   738вҶ’  } else if (first) {
   739вҶ’   739вҶ’    const parts: string[] = [];
   740вҶ’   740вҶ’    const seen = new Set<string>();
   741вҶ’   741вҶ’    const title = clean(first.title);
   742вҶ’   742вҶ’    const notes = clean(first.notes);
   743вҶ’   743вҶ’    const value = clean(first.value);
   744вҶ’   744вҶ’    // Normaliza: lowercase + sin acentos para comparar
   745вҶ’   745вҶ’    const norm = (s: string) => s.toLowerCase().normalize("NFD").replace(/[\u0300-\u036f]/g, "");
   746вҶ’   746вҶ’    const tryAdd = (s: string | undefined) => {
   747вҶ’   747вҶ’      if (!s) return;
   748вҶ’   748вҶ’      const key = norm(s);
   749вҶ’   749вҶ’      if (seen.has(key)) return;
   750вҶ’   750вҶ’      if (key === norm(collection ?? "")) return;
   751вҶ’   751вҶ’      seen.add(key);
   752вҶ’   752вҶ’      parts.push(s);
   753вҶ’   753вҶ’    };
   754вҶ’   754вҶ’    tryAdd(title);
   755вҶ’   755вҶ’    tryAdd(notes);
   756вҶ’   756вҶ’    tryAdd(value);
   757вҶ’   757вҶ’    desc = parts.length > 0 ? parts.join(" В· ") : undefined;
   758вҶ’   758вҶ’  }
   759вҶ’   759вҶ’  // Kicker: solo "Guardado en <colecciГіn>" si hay collection real; si no, "Guardado"
   760вҶ’   760вҶ’  const kicker = collection ? `Guardado en ${collection}` : "Guardado";
   761вҶ’   761вҶ’  // Title: si hay collection, mostrarlo. Si no, mostrar el title del primer record.
   762вҶ’   762вҶ’  const heroTitle = collection || clean(first?.title) || "Registro";
   763вҶ’   763вҶ’  return {
   764вҶ’   764вҶ’    hero: {
   765вҶ’   765вҶ’      kicker,
   766вҶ’   766вҶ’      title: heroTitleFrom(heroTitle, "Registro"),
   767вҶ’   767вҶ’      desc,
   768вҶ’   768вҶ’      icon: isLink ? "link" : "bookmark",
   769вҶ’   769вҶ’      accent: A.violet,
   770вҶ’   770вҶ’    },
   771вҶ’   771вҶ’    cta: {
   772вҶ’   772вҶ’      label: collection ? "Ver colecciГіn" : "Ver mis guardados",
   773вҶ’   773вҶ’      screen: "collections",
   774вҶ’   774вҶ’      collection: collection || undefined,
