   292→    void loadPersistedState().then(async (persisted) => {

   294→        const nudges = buildHeartbeatNudges(persisted);
   295→        commitDomainState(nudges.length > 0 ? applyHeartbeatNudges(persisted, nudges) : persisted);
   296→        // 🔴 Sincronizar reminders con notificaciones del navegador

   302→        // 🔴 Morning Brief: si es mañana y no se mostró hoy, generar

   305→        const today = now.toISOString().slice(0, 10);
   306→        const lastBriefDate = localStorage.getItem("koru.lastBriefDate");
   307→        if (hour >= 5 && hour <= 11 && lastBriefDate !== today) {
   308→          try {
   309→            const stateForBrief = {

   311→              commitments: (persisted.commitments ?? []).filter((c: any) => c.status === "open").slice(0, 5).map((c: any) => ({ title: c.title, dueHint: c.dueHint, dueAt: c.dueAt })),

   316→              method: "POST",
   317→              headers: { "Content-Type": "application/json" },
   318→              body: JSON.stringify({ state: stateForBrief }),
   319→            });

   321→              const data = await res.json();
   322→              if (data.shouldShow && data.brief) {
   323→                setMorningBrief(data.brief);
   324→                localStorage.setItem("koru.lastBriefDate", data.date ?? today);
   325→              }
   326→            }
   327→          } catch {
   328→            // silent — brief is optional
   329→          }
   330→        }
   331→      }
   332→    });
   333→    return () => { cancelled = true; };
   334→  }, []);
   335→

   338→      // 🔴 Proactividad: check due reminders y disparar notificaciones

   346→        const nudges = buildHeartbeatNudges(prev);
   347→        return nudges.length > 0 ? applyHeartbeatNudges(prev, nudges) : prev;

   351→      // Si hay nudges nuevos, inyectarlos como mensajes de Koru en el chat.

   353→      const state = domainStateRef.current;

   392→    };
   393→    document.addEventListener("visibilitychange", onVisibilityChange);

   429→          for (const msg of queue) {

   431→            try {
   432→              await sendMessageRef.current?.(msg.text);
   433→              await dequeueOfflineMessage(msg.id);
   434→            } catch {
   435→              // If still failing, leave it queued for the next retry.
   436→              break;
   437→            }
   438→          }
   439→        }).catch(() => { /* best-effort */ });
   440→      }
   441→    });

   452→    };

   454→    return () => window.removeEventListener("koru:proactive", onProactive);
   455→  }, []);
   456→
   457→  const energy = domainState.trustedEnergy;
   458→  const roots = domainState.memories.filter((m) => m.status === "confirmed").length;
   459→  const stage = domainStageToNew(domainState.stage);
   460→  const ephemeral = domainState.ephemeralMode;
   461→
   462→  const priorities: Priority[] = useMemo(() => {
   463→    const actionItems = domainState.actions
   464→      .filter((a) => a.status === "executed")
   465→      .slice(0, 1)
   466→      .map((a) => ({
   467→        id: a.id,
   468→        label: a.title,
   469→        detail: a.result,
   470→        done: true,
   471→        kind: "action" as const,
   472→      }));

   474→      .filter((nudge) => !nudge.dismissed)
   475→      .slice(0, 2)
   476→      .map((nudge) => ({
   477→        id: nudge.id,
   478→        label: nudge.title,
   479→        detail: nudge.body || nudge.reason,
   480→        done: false,
   481→        kind: "nudge" as const,
   482→      }));
   483→    const commitments = domainState.commitments
   484→      .filter((c) => c.status === "open")

   769→  async function submitEntry(
   770→    text: string,
   771→    transcriptSource: DailyEntry["transcriptSource"] = "typed",
   772→    history: KoruChatTurn[] = chatTurnsRef.current,
   773→  ) {
   774→    setProcessing(true);
   775→    let koruTurnId: string | null = null;
   776→    // Trackea los items creados durante el stream para preservar sus ids al
   777→    // reemplazar con el resultado final. Sin esto, el resultado final crea items
   778→    // con ids nuevos → el WebNavCardA se desmonta/remonta → pierde su estado
   779→    // (animación de progreso, visibleCount) → reinicia a "Iniciando búsqueda...".
   780→    const streamedItemsByType = new Map<string, KoruTurnItem>();
   781→    try {
   782→      const previousState = domainStateRef.current;
   783→      const domainHistory = history
   784→        .filter((turn) => turn.text.trim().length > 0)
   785→        .slice(-12)
   786→        .map((turn) => ({
   787→          role: turn.role === "koru" ? ("assistant" as const) : ("user" as const),
   788→          content: turn.text,
   789→          createdAt: turn.createdAt,
   790→        }));
   791→      const agentResult = await runBackendAgentTurn(text, previousState, domainHistory, selectedModel ?? undefined, (chunk) => {
   792→        // Fase real del pipeline: tomamos el último stateEvent para la píldora/barra.
   793→        if (chunk.stateEvents?.length) {
   794→          const last = chunk.stateEvents[chunk.stateEvents.length - 1];
   795→          setPhase(last.kind === "done" ? null : last.kind);
   796→        }
   797→        const blocksToItems = (blocks: UiBlock[]): KoruTurnItem[] =>
   798→          blocks.map((block) => ({
   799→            id: createId("item"),
   800→            kind: "action" as const,
   801→            tag: block.type,
   802→            text: (block as any).title ?? (block as any).query ?? "",
   803→            status: "working" as const,
   804→            uiBlock: block,
   805→          }));
   806→        if (!koruTurnId) {
   807→          koruTurnId = createId("turn");
   808→          const koruTurn: KoruChatTurn = {
   809→            id: koruTurnId,
   810→            role: "koru",
   811→            text: chunk.reply,
   812→            createdAt: new Date().toISOString(),
   813→            items: blocksToItems(chunk.uiBlocks),
   814→            status: "working" as const,
   815→            mascotState: chunk.mascotState ?? "working",
   816→          };
   817→          (koruTurn.items ?? []).forEach((it) => { if (it.uiBlock?.type) streamedItemsByType.set(it.uiBlock.type, it); });
   818→          commitChatTurns((prev) => [...prev, koruTurn].slice(-120));
   819→        } else {
   820→          const isDone = chunk.stateEvents?.some((e) => e.kind === "done");
   821→          commitChatTurns((prev) =>
   822→            prev.map((turn) => {
   823→              if (turn.id !== koruTurnId) return turn;
   824→              const existingItems = turn.items ?? [];
   825→              const newBlocks = chunk.uiBlocks ?? [];
   826→              // Merge por TIPO de block (no por índice). Así, si el orden cambia
   827→              // entre chunks (ej: aparece un data_card antes que el web_nav),
   828→              // cada item mantiene su identidad y estado (animación de progreso,
   829→              // contadores internos). Antes se matcheaba por idx, lo que hacía
   830→              // que un cambio de orden desmontara el componente y reiniciara la
   831→              // animación ("Iniciando búsqueda..." colgado).
   832→              const consumedExisting = new Set<string>();
   833→              const mergedItems: KoruTurnItem[] = newBlocks.map((block) => {
   834→                const existing = existingItems.find(
   835→                  (it) => it.uiBlock?.type === block.type && !consumedExisting.has(it.uiBlock!.type),
   836→                );
   837→                if (existing) {
   838→                  consumedExisting.add(existing.uiBlock!.type);
   839→                  return { ...existing, uiBlock: block, text: (block as any).title ?? (block as any).query ?? "" };
   840→                }
   841→                return {
   842→                  id: createId("item"),
   843→                  kind: "action" as const,
   844→                  tag: block.type,
   845→                  text: (block as any).title ?? (block as any).query ?? "",
   846→                  status: "working" as const,
   847→                  uiBlock: block,
   848→                };
   849→              });
   850→              mergedItems.forEach((it) => { if (it.uiBlock?.type) streamedItemsByType.set(it.uiBlock.type, it); });
   851→              return {
   852→                ...turn,
   853→                text: chunk.reply,
   854→                items: mergedItems,
   855→                mascotState: chunk.mascotState ?? turn.mascotState,
   856→                status: isDone ? ("done" as const) : ("working" as const),
   857→              };
   858→            }),
   859→          );
   860→        }
   861→      });
   862→      const result = applyBackendTurnToState(previousState, text, transcriptSource, agentResult);
   863→      commitDomainState(result.state);
   864→
   865→      // 🔴 Programar notificaciones para nuevos commitments con dueAt
   866→      const newCommitments = (result.state.commitments ?? []).filter(
   867→        (c) => c.dueAt && !previousState.commitments?.find((pc) => pc.id === c.id)
   868→      );
   869→      if (newCommitments.length > 0) {
   870→        const userId = result.state.userId ?? "default";
   871→        for (const c of newCommitments) {
   872→          scheduleReminderNotification(c as any, userId);
   873→          // 🔴 NUEVO: programar setTimeout preciso para que la notificación
   874→          // dispare EXACTO cuando llega la hora, no en el próximo heartbeat tick.
   875→          schedulePreciseTimeout(c.id, c.dueAt!, c.title, userId);
   876→        }
   877→        // 🔴 Request notification permission — best effort
   878→        requestNotificationPermission().catch(() => {});
   879→      }
   880→      if (koruTurnId) {
   881→        // Preservar ids de los items del stream: si el resultado final tiene un item
   882→        // del mismo tipo que uno ya renderizado en el stream, mantener su id (e
   883→        // identidad de React) para no desmontar/remontar el componente y perder su
   884→        // estado interno (animación de progreso del WebNavCardA, visibleCount, etc.).
   885→        const consumedTypes = new Set<string>();
   886→        const stableItems = result.items.map((item) => {
   887→          const type = item.uiBlock?.type ?? item.tag;
   888→          if (type && !consumedTypes.has(type)) {
   889→            const streamed = streamedItemsByType.get(type);
   890→            if (streamed) {
   891→              consumedTypes.add(type);
   892→              return { ...item, id: streamed.id };
   893→            }
   894→          }
   895→          return item;
   896→        });
   897→        commitChatTurns((prev) =>
   898→          prev.map((turn) =>
   899→            turn.id === koruTurnId
   900→              ? { ...turn, text: agentResult.reply, items: stableItems, status: "done" as const, mascotState: agentResult.mascotState ?? "idle" }
   901→              : turn,
   902→          ),
   903→        );
   904→        // 🔴 Voice: si voice está activado, Koru habla la respuesta
   905→        if (voiceEnabled && agentResult.reply) {
   906→          speak(agentResult.reply);
   907→        }
   908→        // 🔴 Memory toast: si hay memoryCandidates en la respuesta, mostrar toast.
   909→        // Buscar en memoryCandidates top-level Y dentro de toolResults (save_memory tool).
   910→        const topCandidates = (agentResult.memoryCandidates ?? []);
   911→        const toolCandidates = (agentResult.toolResults ?? []).flatMap((tr: any) => {
   912→          const data = tr.data;
   913→          if (data?.type === "memory_capture" && Array.isArray(data.memoryCandidates)) return data.memoryCandidates;
   914→          if (data?.type === "personal_capture" && Array.isArray(data.memoryCandidates)) return data.memoryCandidates;
   915→          return [];
   916→        });
   917→        const allCandidates = [...topCandidates, ...toolCandidates].filter(
   918→          (m: any) => m && typeof m.text === "string" && m.text.length > 4,
   919→        );
   920→        if (allCandidates.length > 0 && !previousState.ephemeralMode && previousState.durableMemoryEnabled) {
   921→          const first = allCandidates[0];
   922→          // Solo mostrar si NO existe ya una memoria con el mismo texto (deduplicación visual)
   923→          const normalizedNew = first.text.toLowerCase().replace(/[áéíóú]/g, (m: string) => ({ á: "a", é: "e", í: "i", ó: "o", ú: "u" }[m]!)).replace(/[^a-z0-9\s]/g, "").trim();
   924→          const alreadyExists = previousState.memories.some((existing: any) => {
   925→            const norm = existing.text.toLowerCase().replace(/[áéíóú]/g, (m: string) => ({ á: "a", é: "e", í: "i", ó: "o", ú: "u" }[m]!)).replace(/[^a-z0-9\s]/g, "").trim();
   926→            return norm === normalizedNew;
   927→          });
   928→          if (!alreadyExists) {
   929→            setMemoryToast({
   930→              id: `toast_${Date.now()}`,
   931→              kind: first.kind,
   932→              text: first.text,
   933→            });
   934→            // Auto-dismiss después de 6 segundos
   935→            setTimeout(() => {
   936→              setMemoryToast((current) => (current?.text === first.text ? null : current));
   937→            }, 6000);
   938→          }
   939→        }
   940→      }
   941→      writeAuditEvent({
   942→        type: "turn_analyzed",
   943→        engine: "backend_agent_loop",
   944→        input: text,
   945→        transcriptSource,
   946→        historyUsed: domainHistory,
   947→        response: agentResult.reply,
   948→        understanding: agentResult.understanding,
   949→        suggestedActions: agentResult.suggestedActions,
   950→        toolResults: agentResult.toolResults,
   951→        provider: agentResult.provider,
   952→        model: agentResult.model,
   953→        fallbackReason: agentResult.fallbackReason,
   954→        items: auditTurnItems(result.items),
   955→        state: auditStateSnapshot(result.state),
   956→        delta: auditStateDelta(previousState, result.state),
   957→      });
   958→      return { response: agentResult.reply, items: result.items, state: result.state, mascotState: agentResult.mascotState, koruTurnId };
   959→    } finally {
   960→      setProcessing(false);
   961→      setActivity(null);
   962→      setPhase(null);
   963→    }
   964→  }

   966→  async function sendMessage(text: string, transcriptSource: DailyEntry["transcriptSource"] = "typed") {
   967→    const cleanText = text.trim();
   968→    if (!cleanText) return null;
   969→    const previousState = domainStateRef.current;
   970→    const userTurn: KoruChatTurn = {
   971→      id: createId("turn"),
   972→      role: "user",
   973→      text: cleanText,
   974→      createdAt: new Date().toISOString(),
   975→      status: "done",
   976→    };
   977→    const historyBeforeUser = chatTurnsRef.current;
   978→    commitChatTurns((prev) => [...prev, userTurn].slice(-120));
   979→    writeAuditEvent({
   980→      type: "user_message",
   981→      turn: userTurn,
   982→      transcriptSource,

   984→        id: turn.id,

   986→        text: turn.text,
   987→        createdAt: turn.createdAt,
   988→        status: turn.status,
   989→        items: auditTurnItems(turn.items),
   990→      })),
   991→      stateBefore: auditStateSnapshot(previousState),
   992→    });
   993→    setActivity(inferActivity(cleanText));
   994→    try {
   995→      const result = await submitEntry(cleanText, transcriptSource, historyBeforeUser);
   996→      let koruTurn: KoruChatTurn | undefined;
   997→      if (!result.koruTurnId) {
   998→        koruTurn = {
   999→          id: createId("turn"),
  1000→          role: "koru",
  1001→          text: result.response,
  1002→          createdAt: new Date().toISOString(),
  1003→          items: result.items,
  1004→          status: "done",
  1005→          mascotState: result.mascotState ?? "idle",
  1006→        };
  1007→        commitChatTurns((prev) => [...prev, koruTurn!].slice(-120));
  1008→        writeAuditEvent({
  1009→          type: "koru_message",
  1010→          turn: {
  1011→            id: koruTurn.id,
  1012→            role: koruTurn.role,
  1013→            text: koruTurn.text,
  1014→            createdAt: koruTurn.createdAt,
  1015→            status: koruTurn.status,
  1016→            items: auditTurnItems(koruTurn.items),
  1017→          },
  1018→          stateAfter: auditStateSnapshot(result.state),
  1019→        });
  1020→      } else {
  1021→        commitChatTurns((prev) =>
  1022→          prev.map((turn) => {
  1023→            if (turn.id !== result.koruTurnId) return turn;
  1024→            const existingItems = turn.items ?? [];
  1025→            const mergedItems: KoruTurnItem[] = result.items.map((newItem, idx) => {
  1026→              const existing = existingItems[idx];
  1027→              if (existing && existing.uiBlock?.type === newItem.uiBlock?.type) {
  1028→                return { ...newItem, id: existing.id };
  1029→              }
  1030→              return newItem;
  1031→            });
  1032→            return { ...turn, items: mergedItems, status: "done" as const, mascotState: result.mascotState ?? "idle" };
  1033→          }),
  1034→        );
  1035→        koruTurn = chatTurnsRef.current.find((t) => t.id === result.koruTurnId);
  1036→        writeAuditEvent({
  1037→          type: "koru_message",
  1038→          turn: {
  1039→            id: result.koruTurnId,
  1040→            role: "koru",
  1041→            text: result.response,
  1042→            createdAt: new Date().toISOString(),
  1043→            status: "done",
  1044→            items: auditTurnItems(result.items),
  1045→          },
  1046→          stateAfter: auditStateSnapshot(result.state),
  1047→        });
  1048→      }
  1049→      const autoWebAction = result.state.actions
  1050→        .filter((action) => !previousState.actions.some((existing) => existing.id === action.id))
  1051→        .find(shouldAutoRunAction);
  1052→      if (autoWebAction) {
  1053→        window.setTimeout(() => runReadonlyWebAction(autoWebAction), 0);
  1054→      }
  1055→      return koruTurn ?? null;
  1056→    } catch (error) {
  1057→      const message = error instanceof Error ? error.message : "Fallo desconocido del agente.";
  1058→      const errorTurn: KoruChatTurn = {
  1059→        id: createId("turn"),
  1060→        role: "koru",
  1061→        text: `No pude contactar bien al agente ahora. No voy a inventarte una respuesta. Detalle: ${message}`,
  1062→        createdAt: new Date().toISOString(),
  1063→        status: "error",
  1064→      };
  1065→      commitChatTurns((prev) => [...prev, errorTurn].slice(-120));
  1066→      writeAuditEvent({
  1067→        type: "turn_error",
  1068→        input: cleanText,
  1069→        error: message,
  1070→        turn: errorTurn,
  1071→        state: auditStateSnapshot(domainStateRef.current),
  1072→      });
  1073→      setActivity(null);
  1074→      return errorTurn;
  1075→    }
  1076→  }

  1109→  function runReadonlyWebAction(original: AssistantAction): KoruTurnItem {
  1110→    // Dato en vivo (clima/trafico/mercado) = quick, sin panel largo. Un
  1111→    // informe/investigacion real (webMode "research", o web_research sin
  1112→    // modo puntual) = deep, se gana el panel "Trabajando...". Ver
  1113→    // flujo-informe-aoe2.html pasos 3 (informe) vs 7 (clima).
  1114→    const webMode = original.payload.webMode;
  1115→    const isQuickSignal = webMode === "weather" || webMode === "traffic" || webMode === "market";
  1116→    const isDeepResearch = webMode === "research" || (original.kind === "web_research" && !webMode);
  1117→    setActivity({
  1118→      kind: "searching",
  1119→      depth: isQuickSignal ? "quick" : isDeepResearch ? "deep" : "quick",
  1120→      label: isDeepResearch
  1121→        ? "Es un tema para investigar a fondo. Dame unos segundos."
  1122→        : original.kind === "world_signal"
  1123→          ? "Busco senales reales."
  1124→          : "Consulto fuentes reales.",
  1125→    });
  1126→    const now = new Date().toISOString();
  1127→    const approvedAction: AssistantAction = {
  1128→      ...original,
  1129→      status: "approved",
  1130→      updatedAt: now,
  1131→    };
  1132→    commitDomainState((prev) => {
  1133→      const exists = prev.actions.some((action) => action.id === original.id);
  1134→      const next = {
  1135→        ...prev,
  1136→        actions: exists
  1137→          ? prev.actions.map((action) => (action.id === original.id ? approvedAction : action))
  1138→          : [approvedAction, ...prev.actions],
  1139→        updatedAt: now,
  1140→      };
  1141→      saveState(next);
  1142→      return next;
  1143→    });
  1144→    const waitingItem = actionToTurnItem(approvedAction);
  1145→    replaceActionInChat(waitingItem);
  1146→
  1147→    void (async () => {
  1148→      try {
  1149→        const result = await runWebNavigation(original.payload);
  1150→        const webPayload = webResultToPayload(result);
  1151→        const nextUiBlock = patchUiBlockWithWebResult(approvedAction, webPayload);
  1152→        const finishedAt = new Date().toISOString();
  1153→        const patchedAction: AssistantAction = {
  1154→          ...approvedAction,
  1155→          payload: {
  1156→            ...approvedAction.payload,
  1157→            ...webPayload,
  1158→            uiBlock: nextUiBlock ?? approvedAction.payload.uiBlock,
  1159→          },
  1160→          result: result.recommendation,
  1161→          status: "executed",
  1162→          updatedAt: finishedAt,
  1163→          executedAt: finishedAt,
  1164→        };
  1165→        const patchedItem = actionToTurnItem(patchedAction);
  1166→        commitDomainState((prev) => {
  1167→          const next = {
  1168→            ...prev,
  1169→            actions: prev.actions.map((action) => (action.id === original.id ? patchedAction : action)),
  1170→            updatedAt: finishedAt,
  1171→          };
  1172→          saveState(next);
  1173→          writeAuditEvent({
  1174→            type: "readonly_web_action_finished",
  1175→            action: {
  1176→              id: patchedAction.id,
  1177→              kind: patchedAction.kind,
  1178→              title: patchedAction.title,
  1179→              status: patchedAction.status,
  1180→              result: patchedAction.result,
  1181→              payload: patchedAction.payload,
  1182→            },
  1183→            item: auditTurnItems([patchedItem])[0],
  1184→            state: auditStateSnapshot(next),
  1185→            delta: auditStateDelta(prev, next),
  1186→          });
  1187→          return next;
  1188→        });
  1189→        replaceActionInChat(patchedItem);
  1190→      } finally {
  1191→        setActivity(null);
  1192→      }
  1193→    })();
  1194→    return waitingItem;
  1195→  }

  1197→  function reviewAction(id: string, approve: boolean): KoruTurnItem | null {
  1198→    const currentState = domainStateRef.current;
  1199→    const original = currentState.actions.find((action) => action.id === id);
  1200→    if (approve && (original?.kind === "web_research" || original?.kind === "world_signal")) {
  1201→      return runReadonlyWebAction(original);
  1202→    }
  1203→
  1204→    const nextState = approve ? approveAndExecuteAction(currentState, id) : rejectAction(currentState, id);
  1205→    const reviewed = nextState.actions.find((action) => action.id === id);
  1206→    if (!reviewed) return null;
  1207→    commitDomainState(nextState);
  1208→    const item = actionToTurnItem(reviewed);
  1209→    replaceActionInChat(item);
  1210→    writeAuditEvent({
  1211→      type: "action_reviewed",
  1212→      actionId: id,
  1213→      approved: approve,
  1214→      item: auditTurnItems([item])[0],
  1215→      state: auditStateSnapshot(nextState),
  1216→      delta: auditStateDelta(currentState, nextState),
  1217→    });
  1218→    appendKoruConfirmation(approve ? actionConfirmationText(item) : "Lo suelto. No hice nada con eso.");
  1219→
  1220→    return item;
  1221→  }
