   378→  function completeOnboarding(name: string, facts: string[] = []) {
   379→    const cleanName = name.trim() || "amigo";
   380→    const previousState = domainStateRef.current;
   381→    setUserName(cleanName);
   382→    setOnboarded(true);
   383→    localStorage.setItem("koru.onboarded", "true");
   384→    localStorage.setItem("koru.username", cleanName);
   385→    commitChatTurns((prev) => {
   386→      if (prev.length === 0) return [greetingTurn(cleanName)];
   387→      if (prev.length === 1 && prev[0].role === "koru") {
   388→        return [{ ...prev[0], text: `Hola, ${cleanName}. Cuéntame cómo estás.` }];
   389→      }
   390→      return prev;
   391→    });
   392→    commitDomainState((prev) => {
   393→      const withName = { ...prev, userName: cleanName };
   394→      const next = facts.length ? addOnboardingMemories(withName, facts) : withName;
   395→      if (!facts.length) saveState(next);
   396→      writeAuditEvent({
   397→        type: "onboarding_completed",
   398→        name: cleanName,
   399→        facts,
   400→        state: auditStateSnapshot(next),
   401→        delta: auditStateDelta(previousState, next),
   402→      });
   403→      return next;
   404→    });
   405→  }

   579→  async function submitEntry(
   580→    text: string,
   581→    transcriptSource: DailyEntry["transcriptSource"] = "typed",
   582→    history: KoruChatTurn[] = chatTurnsRef.current,
   583→  ) {
   584→    setProcessing(true);
   585→    let koruTurnId: string | null = null;
   586→    // Trackea los items creados durante el stream para preservar sus ids al
   587→    // reemplazar con el resultado final. Sin esto, el resultado final crea items
   588→    // con ids nuevos → el WebNavCardA se desmonta/remonta → pierde su estado
   589→    // (animación de progreso, visibleCount) → reinicia a "Iniciando búsqueda...".
   590→    const streamedItemsByType = new Map<string, KoruTurnItem>();
   591→    try {
   592→      const previousState = domainStateRef.current;
   593→      const domainHistory = history
   594→        .filter((turn) => turn.text.trim().length > 0)
   595→        .slice(-12)
   596→        .map((turn) => ({
   597→          role: turn.role === "koru" ? ("assistant" as const) : ("user" as const),
   598→          content: turn.text,
   599→          createdAt: turn.createdAt,
   600→        }));
   601→      const agentResult = await runBackendAgentTurn(text, previousState, domainHistory, selectedModel ?? undefined, (chunk) => {
   602→        // Fase real del pipeline: tomamos el último stateEvent para la píldora/barra.
   603→        if (chunk.stateEvents?.length) {
   604→          const last = chunk.stateEvents[chunk.stateEvents.length - 1];
   605→          setPhase(last.kind === "done" ? null : last.kind);
   606→        }
   607→        const blocksToItems = (blocks: UiBlock[]): KoruTurnItem[] =>
   608→          blocks.map((block) => ({
   609→            id: createId("item"),
   610→            kind: "action" as const,
   611→            tag: block.type,
   612→            text: (block as any).title ?? (block as any).query ?? "",
   613→            status: "working" as const,
   614→            uiBlock: block,
   615→          }));
   616→        if (!koruTurnId) {
   617→          koruTurnId = createId("turn");
   618→          const koruTurn: KoruChatTurn = {
   619→            id: koruTurnId,
   620→            role: "koru",
   621→            text: chunk.reply,
   622→            createdAt: new Date().toISOString(),
   623→            items: blocksToItems(chunk.uiBlocks),
   624→            status: "working" as const,
   625→            mascotState: chunk.mascotState ?? "working",
   626→          };
   627→          (koruTurn.items ?? []).forEach((it) => { if (it.uiBlock?.type) streamedItemsByType.set(it.uiBlock.type, it); });
   628→          commitChatTurns((prev) => [...prev, koruTurn].slice(-120));
   629→        } else {
   630→          const isDone = chunk.stateEvents?.some((e) => e.kind === "done");
   631→          commitChatTurns((prev) =>
   632→            prev.map((turn) => {
   633→              if (turn.id !== koruTurnId) return turn;
   634→              const existingItems = turn.items ?? [];
   635→              const newBlocks = chunk.uiBlocks ?? [];
   636→              // Merge por TIPO de block (no por índice). Así, si el orden cambia
   637→              // entre chunks (ej: aparece un data_card antes que el web_nav),
   638→              // cada item mantiene su identidad y estado (animación de progreso,
   639→              // contadores internos). Antes se matcheaba por idx, lo que hacía
   640→              // que un cambio de orden desmontara el componente y reiniciara la
   641→              // animación ("Iniciando búsqueda..." colgado).
   642→              const consumedExisting = new Set<string>();
   643→              const mergedItems: KoruTurnItem[] = newBlocks.map((block) => {
   644→                const existing = existingItems.find(
   645→                  (it) => it.uiBlock?.type === block.type && !consumedExisting.has(it.uiBlock!.type),
   646→                );
   647→                if (existing) {
   648→                  consumedExisting.add(existing.uiBlock!.type);
   649→                  return { ...existing, uiBlock: block, text: (block as any).title ?? (block as any).query ?? "" };
   650→                }
   651→                return {
   652→                  id: createId("item"),
   653→                  kind: "action" as const,
   654→                  tag: block.type,
   655→                  text: (block as any).title ?? (block as any).query ?? "",
   656→                  status: "working" as const,
   657→                  uiBlock: block,
   658→                };
   659→              });
   660→              mergedItems.forEach((it) => { if (it.uiBlock?.type) streamedItemsByType.set(it.uiBlock.type, it); });
   661→              return {
   662→                ...turn,
   663→                text: chunk.reply,
   664→                items: mergedItems,
   665→                mascotState: chunk.mascotState ?? turn.mascotState,
   666→                status: isDone ? ("done" as const) : ("working" as const),
   667→              };
   668→            }),
   669→          );
   670→        }
   671→      });
   672→      const result = applyBackendTurnToState(previousState, text, transcriptSource, agentResult);
   673→      commitDomainState(result.state);
   674→      if (koruTurnId) {
   675→        // Preservar ids de los items del stream: si el resultado final tiene un item
   676→        // del mismo tipo que uno ya renderizado en el stream, mantener su id (e
   677→        // identidad de React) para no desmontar/remontar el componente y perder su
   678→        // estado interno (animación de progreso del WebNavCardA, visibleCount, etc.).
   679→        const consumedTypes = new Set<string>();
   680→        const stableItems = result.items.map((item) => {
   681→          const type = item.uiBlock?.type ?? item.tag;
   682→          if (type && !consumedTypes.has(type)) {
   683→            const streamed = streamedItemsByType.get(type);
   684→            if (streamed) {
   685→              consumedTypes.add(type);
   686→              return { ...item, id: streamed.id };
   687→            }
   688→          }
   689→          return item;
   690→        });
   691→        commitChatTurns((prev) =>
   692→          prev.map((turn) =>
   693→            turn.id === koruTurnId
   694→              ? { ...turn, text: agentResult.reply, items: stableItems, status: "done" as const, mascotState: agentResult.mascotState ?? "idle" }
   695→              : turn,
   696→          ),
   697→        );
   698→      }
   699→      writeAuditEvent({
   700→        type: "turn_analyzed",
   701→        engine: "backend_agent_loop",
   702→        input: text,
   703→        transcriptSource,
   704→        historyUsed: domainHistory,
   705→        response: agentResult.reply,
   706→        understanding: agentResult.understanding,
   707→        suggestedActions: agentResult.suggestedActions,
   708→        toolResults: agentResult.toolResults,
   709→        provider: agentResult.provider,
   710→        model: agentResult.model,
   711→        fallbackReason: agentResult.fallbackReason,
   712→        items: auditTurnItems(result.items),
   713→        state: auditStateSnapshot(result.state),
   714→        delta: auditStateDelta(previousState, result.state),
   715→      });
   716→      return { response: agentResult.reply, items: result.items, state: result.state, mascotState: agentResult.mascotState, koruTurnId };
   717→    } finally {
   718→      setProcessing(false);
   719→      setActivity(null);
   720→      setPhase(null);
   721→    }
   722→  }

   724→  async function sendMessage(text: string, transcriptSource: DailyEntry["transcriptSource"] = "typed") {
   725→    const cleanText = text.trim();
   726→    if (!cleanText) return null;
   727→    const previousState = domainStateRef.current;
   728→    const userTurn: KoruChatTurn = {
   729→      id: createId("turn"),
   730→      role: "user",
   731→      text: cleanText,
   732→      createdAt: new Date().toISOString(),
   733→      status: "done",
   734→    };
   735→    const historyBeforeUser = chatTurnsRef.current;
   736→    commitChatTurns((prev) => [...prev, userTurn].slice(-120));
   737→    writeAuditEvent({
   738→      type: "user_message",
   739→      turn: userTurn,
   740→      transcriptSource,
   741→      historyBeforeUser: historyBeforeUser.map((turn) => ({
   742→        id: turn.id,
   743→        role: turn.role,
   744→        text: turn.text,
   745→        createdAt: turn.createdAt,
   746→        status: turn.status,
   747→        items: auditTurnItems(turn.items),
   748→      })),
   749→      stateBefore: auditStateSnapshot(previousState),
   750→    });
   751→    setActivity(inferActivity(cleanText));
   752→    try {
   753→      const result = await submitEntry(cleanText, transcriptSource, historyBeforeUser);
   754→      let koruTurn: KoruChatTurn | undefined;
   755→      if (!result.koruTurnId) {
   756→        koruTurn = {
   757→          id: createId("turn"),
   758→          role: "koru",
   759→          text: result.response,
   760→          createdAt: new Date().toISOString(),
   761→          items: result.items,
   762→          status: "done",
   763→          mascotState: result.mascotState ?? "idle",
   764→        };
   765→        commitChatTurns((prev) => [...prev, koruTurn!].slice(-120));
   766→        writeAuditEvent({
   767→          type: "koru_message",
   768→          turn: {
   769→            id: koruTurn.id,
   770→            role: koruTurn.role,
   771→            text: koruTurn.text,
   772→            createdAt: koruTurn.createdAt,
   773→            status: koruTurn.status,
   774→            items: auditTurnItems(koruTurn.items),
   775→          },
   776→          stateAfter: auditStateSnapshot(result.state),
   777→        });
   778→      } else {
   779→        commitChatTurns((prev) =>
   780→          prev.map((turn) => {
   781→            if (turn.id !== result.koruTurnId) return turn;
   782→            const existingItems = turn.items ?? [];
   783→            const mergedItems: KoruTurnItem[] = result.items.map((newItem, idx) => {
   784→              const existing = existingItems[idx];
   785→              if (existing && existing.uiBlock?.type === newItem.uiBlock?.type) {
   786→                return { ...newItem, id: existing.id };
   787→              }
   788→              return newItem;
   789→            });
   790→            return { ...turn, items: mergedItems, status: "done" as const, mascotState: result.mascotState ?? "idle" };
   791→          }),
   792→        );
   793→        koruTurn = chatTurnsRef.current.find((t) => t.id === result.koruTurnId);
   794→        writeAuditEvent({
   795→          type: "koru_message",
   796→          turn: {
   797→            id: result.koruTurnId,
   798→            role: "koru",
   799→            text: result.response,
   800→            createdAt: new Date().toISOString(),
   801→            status: "done",
   802→            items: auditTurnItems(result.items),
   803→          },
   804→          stateAfter: auditStateSnapshot(result.state),
   805→        });
   806→      }
   807→      const autoWebAction = result.state.actions
   808→        .filter((action) => !previousState.actions.some((existing) => existing.id === action.id))
   809→        .find(shouldAutoRunAction);
   810→      if (autoWebAction) {
   811→        window.setTimeout(() => runReadonlyWebAction(autoWebAction), 0);
   812→      }
   813→      return koruTurn ?? null;
   814→    } catch (error) {
   815→      const message = error instanceof Error ? error.message : "Fallo desconocido del agente.";
   816→      const errorTurn: KoruChatTurn = {
   817→        id: createId("turn"),
   818→        role: "koru",
   819→        text: `No pude contactar bien al agente ahora. No voy a inventarte una respuesta. Detalle: ${message}`,
   820→        createdAt: new Date().toISOString(),
   821→        status: "error",
   822→      };
   823→      commitChatTurns((prev) => [...prev, errorTurn].slice(-120));
   824→      writeAuditEvent({
   825→        type: "turn_error",
   826→        input: cleanText,
   827→        error: message,
   828→        turn: errorTurn,
   829→        state: auditStateSnapshot(domainStateRef.current),
   830→      });
   831→      setActivity(null);
   832→      return errorTurn;
   833→    }
   834→  }

   861→  function runReadonlyWebAction(original: AssistantAction): KoruTurnItem {
   862→    // Dato en vivo (clima/trafico/mercado) = quick, sin panel largo. Un
   863→    // informe/investigacion real (webMode "research", o web_research sin
   864→    // modo puntual) = deep, se gana el panel "Trabajando...". Ver
   865→    // flujo-informe-aoe2.html pasos 3 (informe) vs 7 (clima).
   866→    const webMode = original.payload.webMode;
   867→    const isQuickSignal = webMode === "weather" || webMode === "traffic" || webMode === "market";
   868→    const isDeepResearch = webMode === "research" || (original.kind === "web_research" && !webMode);
   869→    setActivity({
   870→      kind: "searching",
   871→      depth: isQuickSignal ? "quick" : isDeepResearch ? "deep" : "quick",
   872→      label: isDeepResearch
   873→        ? "Es un tema para investigar a fondo. Dame unos segundos."
   874→        : original.kind === "world_signal"
   875→          ? "Busco senales reales."
   876→          : "Consulto fuentes reales.",
   877→    });
   878→    const now = new Date().toISOString();
   879→    const approvedAction: AssistantAction = {
   880→      ...original,
   881→      status: "approved",
   882→      updatedAt: now,
   883→    };
   884→    commitDomainState((prev) => {
   885→      const exists = prev.actions.some((action) => action.id === original.id);
   886→      const next = {
   887→        ...prev,
   888→        actions: exists
   889→          ? prev.actions.map((action) => (action.id === original.id ? approvedAction : action))
   890→          : [approvedAction, ...prev.actions],
   891→        updatedAt: now,
   892→      };
   893→      saveState(next);
   894→      return next;
   895→    });
   896→    const waitingItem = actionToTurnItem(approvedAction);
   897→    replaceActionInChat(waitingItem);
   898→
   899→    void (async () => {
   900→      try {
   901→        const result = await runWebNavigation(original.payload);
   902→        const webPayload = webResultToPayload(result);
   903→        const nextUiBlock = patchUiBlockWithWebResult(approvedAction, webPayload);
   904→        const finishedAt = new Date().toISOString();
   905→        const patchedAction: AssistantAction = {
   906→          ...approvedAction,
   907→          payload: {
   908→            ...approvedAction.payload,
   909→            ...webPayload,
   910→            uiBlock: nextUiBlock ?? approvedAction.payload.uiBlock,
   911→          },
   912→          result: result.recommendation,
   913→          status: "executed",
   914→          updatedAt: finishedAt,
   915→          executedAt: finishedAt,
   916→        };
   917→        const patchedItem = actionToTurnItem(patchedAction);
   918→        commitDomainState((prev) => {
   919→          const next = {
   920→            ...prev,
   921→            actions: prev.actions.map((action) => (action.id === original.id ? patchedAction : action)),
   922→            updatedAt: finishedAt,
   923→          };
   924→          saveState(next);
   925→          writeAuditEvent({
   926→            type: "readonly_web_action_finished",
   927→            action: {
   928→              id: patchedAction.id,
   929→              kind: patchedAction.kind,
   930→              title: patchedAction.title,
   931→              status: patchedAction.status,
   932→              result: patchedAction.result,
   933→              payload: patchedAction.payload,
   934→            },
   935→            item: auditTurnItems([patchedItem])[0],
   936→            state: auditStateSnapshot(next),
   937→            delta: auditStateDelta(prev, next),
   938→          });
   939→          return next;
   940→        });
   941→        replaceActionInChat(patchedItem);
   942→      } finally {
   943→        setActivity(null);
   944→      }
   945→    })();
   946→    return waitingItem;
   947→  }

   949→  function reviewAction(id: string, approve: boolean): KoruTurnItem | null {
   950→    const currentState = domainStateRef.current;
   951→    const original = currentState.actions.find((action) => action.id === id);
   952→    if (approve && (original?.kind === "web_research" || original?.kind === "world_signal")) {
   953→      return runReadonlyWebAction(original);
   954→    }
   955→
   956→    const nextState = approve ? approveAndExecuteAction(currentState, id) : rejectAction(currentState, id);
   957→    const reviewed = nextState.actions.find((action) => action.id === id);
   958→    if (!reviewed) return null;
   959→    commitDomainState(nextState);
   960→    const item = actionToTurnItem(reviewed);
   961→    replaceActionInChat(item);
   962→    writeAuditEvent({
   963→      type: "action_reviewed",
   964→      actionId: id,
   965→      approved: approve,
   966→      item: auditTurnItems([item])[0],
   967→      state: auditStateSnapshot(nextState),
   968→      delta: auditStateDelta(currentState, nextState),
   969→    });
   970→    appendKoruConfirmation(approve ? actionConfirmationText(item) : "Lo suelto. No hice nada con eso.");
   971→
   972→    return item;
   973→  }
